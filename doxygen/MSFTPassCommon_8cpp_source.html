16 void MSFTPassCommon::dedupInputs(MSFTModuleOp mod) {
17  const auto &instantiations = moduleInstantiations[mod];
18  // TODO: remove this limitation. This would involve looking at the common
19  // loopbacks for all the instances.
20  if (instantiations.size() != 1)
21  return;
22  InstanceOp inst =
23  dyn_cast<InstanceOp>(static_cast<Operation *>(instantiations[0]));
24  if (!inst)
25  return;
26 
27  // Find all the arguments which are driven by the same signal. Remap them
28  // appropriately within the module, and mark that input port for deletion.
29  Block *body = mod.getBodyBlock();
30  DenseMap<Value, unsigned> valueToInput;
31  llvm::BitVector argsToErase(body->getNumArguments());
32  for (OpOperand &oper : inst->getOpOperands()) {
33  auto existingValue = valueToInput.find(oper.get());
34  if (existingValue != valueToInput.end()) {
35  unsigned operNum = oper.getOperandNumber();
36  unsigned duplicateInputNum = existingValue->second;
37  body->getArgument(operNum).replaceAllUsesWith(
38  body->getArgument(duplicateInputNum));
39  argsToErase.set(operNum);
40  } else {
41  valueToInput[oper.get()] = oper.getOperandNumber();
42  }
43  }
44 
45  // Remove the ports.
46  auto remappedResults =
47  mod.removePorts(argsToErase, llvm::BitVector(inst.getNumResults()));
48  // and update the instantiations.
49  auto getOperands = [&](InstanceOp newInst, InstanceOp oldInst,
50  SmallVectorImpl<Value> &newOperands) {
51  for (unsigned argNum = 0, e = oldInst.getNumOperands(); argNum < e;
52  ++argNum)
53  if (!argsToErase.test(argNum))
54  newOperands.push_back(oldInst.getOperand(argNum));
55  };
56  inst = updateInstances(mod, remappedResults, getOperands)[0];
57 
58  SmallVector<Attribute, 32> newArgNames;
59  std::string buff;
60  for (Value oper : inst->getOperands()) {
61  newArgNames.push_back(StringAttr::get(
62  mod.getContext(), getValueName(oper, topLevelSyms, buff)));
63  }
64  mod.setArgNamesAttr(ArrayAttr::get(mod.getContext(), newArgNames));
65 }

67 /// Sink all the instance connections which are loops.
68 void MSFTPassCommon::sinkWiresDown(MSFTModuleOp mod) {
69  const auto &instantiations = moduleInstantiations[mod];
70  // TODO: remove this limitation. This would involve looking at the common
71  // loopbacks for all the instances.
72  if (instantiations.size() != 1)
73  return;
74  InstanceOp inst =
75  dyn_cast<InstanceOp>(static_cast<Operation *>(instantiations[0]));
76  if (!inst)
77  return;
78 
79  // Find all the "loopback" connections in the instantiation. Populate
80  // 'inputToOutputLoopback' with a mapping of input port to output port which
81  // drives it. Populate 'resultsToErase' with output ports which only drive
82  // input ports.
83  DenseMap<unsigned, unsigned> inputToOutputLoopback;
84  llvm::BitVector resultsToErase(inst.getNumResults());
85  for (unsigned resNum = 0, e = inst.getNumResults(); resNum < e; ++resNum) {
86  bool allLoops = true;
87  for (auto &use : inst.getResult(resNum).getUses()) {
88  if (use.getOwner() != inst.getOperation())
89  allLoops = false;
90  else
91  inputToOutputLoopback[use.getOperandNumber()] = resNum;
92  }
93  if (allLoops)
94  resultsToErase.set(resNum);
95  }
96 
97  // Add internal connections to replace the instantiation's loop back
98  // connections.
99  Block *body = mod.getBodyBlock();
100  Operation *terminator = body->getTerminator();
101  llvm::BitVector argsToErase(body->getNumArguments());
102  for (auto resOper : inputToOutputLoopback) {
103  body->getArgument(resOper.first)
104  .replaceAllUsesWith(terminator->getOperand(resOper.second));
105  argsToErase.set(resOper.first);
106  }
107 
108  // Remove the ports.
109  SmallVector<unsigned> newToOldResultMap =
110  mod.removePorts(argsToErase, resultsToErase);
111  // and update the instantiations.
112  auto getOperands = [&](InstanceOp newInst, InstanceOp oldInst,
113  SmallVectorImpl<Value> &newOperands) {
114  // Use sort-merge-join to compute the new operands;
115  for (unsigned argNum = 0, e = oldInst.getNumOperands(); argNum < e;
116  ++argNum)
117  if (!argsToErase.test(argNum))
118  newOperands.push_back(oldInst.getOperand(argNum));
119  };
120  updateInstances(mod, newToOldResultMap, getOperands);
121 }

135 SmallVector<InstanceOp, 1> MSFTPassCommon::updateInstances(
136  MSFTModuleOp mod, ArrayRef<unsigned> newToOldResultMap,
137  llvm::function_ref<void(InstanceOp, InstanceOp, SmallVectorImpl<Value> &)>
138  getOperandsFunc) {
139 
140  SmallVector<hw::HWInstanceLike, 1> newInstances;
141  SmallVector<InstanceOp, 1> newMsftInstances;
142  for (hw::HWInstanceLike instLike : moduleInstantiations[mod]) {
143  assert(instLike->getParentOp());
144  auto inst = dyn_cast<InstanceOp>(instLike.getOperation());
145  if (!inst) {
146  instLike.emitWarning("Can not update hw.instance ops");
147  continue;
148  }
149 
150  OpBuilder b(inst);
151  auto newInst = b.create<InstanceOp>(inst.getLoc(), mod.getResultTypes(),
152  inst.getOperands(), inst->getAttrs());
153 
154  SmallVector<Value> newOperands;
155  getOperandsFunc(newInst, inst, newOperands);
156  newInst->setOperands(newOperands);
157 
158  for (auto oldResult : llvm::enumerate(newToOldResultMap))
159  if (oldResult.value() < inst.getNumResults())
160  inst.getResult(oldResult.value())
161  .replaceAllUsesWith(newInst.getResult(oldResult.index()));
162 
163  newInstances.push_back(newInst);
164  newMsftInstances.push_back(newInst);
165  inst->dropAllUses();
166  inst->erase();
167  }
168  moduleInstantiations[mod].swap(newInstances);
169  return newMsftInstances;
170 }
