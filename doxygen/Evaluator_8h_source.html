26 /// A value of an object in memory. It is either a composite Object, or a
27 /// primitive Attribute. Further refinement is expected.
28 using ObjectValue = std::variant<std::shared_ptr<struct Object>, Attribute>;
29 
30 /// The fields of a composite Object, currently represented as a map. Further
31 /// refinement is expected.
32 using ObjectFields = SmallDenseMap<StringAttr, ObjectValue>;
33 
34 /// An Evaluator, which is constructed with an IR module and can instantiate
35 /// Objects. Further refinement is expected.
36 struct Evaluator {
37  /// Construct an Evaluator with an IR module.
38  Evaluator(ModuleOp mod);
39 
40  /// Instantiate an Object with its class name and actual parameters.
41  FailureOr<std::shared_ptr<Object>>
42  instantiate(StringAttr className, ArrayRef<ObjectValue> actualParams);
43 
44  /// Get the Module this Evaluator is built from.
45  mlir::ModuleOp getModule();
46 
47 private:
48  /// Evaluate a Value in a Class body according to the small expression grammar
49  /// described in the rationale document. The actual parameters are the values
50  /// supplied at the current instantiation of the Class being evaluated.
51  FailureOr<ObjectValue> evaluateValue(Value value,
52  ArrayRef<ObjectValue> actualParams);
53 
54  /// Evaluator dispatch functions for the small expression grammar.
55  FailureOr<ObjectValue> evaluateParameter(BlockArgument formalParam,
56  ArrayRef<ObjectValue> actualParams);
57  FailureOr<ObjectValue> evaluateConstant(ConstantOp op,
58  ArrayRef<ObjectValue> actualParams);
59  FailureOr<ObjectValue>
60  evaluateObjectInstance(ObjectOp op, ArrayRef<ObjectValue> actualParams);
61  FailureOr<ObjectValue>
62  evaluateObjectField(ObjectFieldOp op, ArrayRef<ObjectValue> actualParams);
63 
64  /// The symbol table for the IR module the Evaluator was constructed with.
65  /// Used to look up class definitions.
66  SymbolTable symbolTable;
67 
68  /// Object storage. Currently used for memoizing calls to evaluateValue.
69  /// Further refinement is expected.
70  mlir::DenseMap<Value, std::shared_ptr<Object>> objects;
71 };

96 /// Helper to enable printing objects in Diagnostics.
97 static inline mlir::Diagnostic &operator<<(mlir::Diagnostic &diag,
98  const ObjectValue &objectValue) {
99  if (auto *object = std::get_if<std::shared_ptr<Object>>(&objectValue))
100  diag << *object;
101  if (auto *attribute = std::get_if<Attribute>(&objectValue))
102  diag << *attribute;
103  return diag;
104 }
