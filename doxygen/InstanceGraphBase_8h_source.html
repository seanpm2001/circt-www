27 /// This just maps a iterator of references to an iterator of addresses.
28 template <typename It>
29 struct AddressIterator
30  : public llvm::mapped_iterator<It, typename It::pointer (*)(
31  typename It::reference)> {
32  // This using statement is to get around a bug in MSVC. Without it, it
33  // tries to look up "It" as a member type of the parent class.
34  using Iterator = It;
35  static typename Iterator::value_type *
36  addrOf(typename Iterator::value_type &v) noexcept {
37  return std::addressof(v);
38  }
39  /* implicit */ AddressIterator(Iterator iterator)
40  : llvm::mapped_iterator<It, typename Iterator::pointer (*)(
41  typename Iterator::reference)>(iterator,
42  addrOf) {}
43 };

48 /// This is an edge in the InstanceGraph. This tracks a specific instantiation
49 /// of a module.
50 class InstanceRecord

88 /// This is a Node in the InstanceGraph. Each node represents a Module in a
89 /// Circuit. Both external modules and regular modules can be represented by
90 /// this class. It is possible to efficiently iterate all modules instantiated
91 /// by this module, as well as all instantiations of this module.
92 class InstanceGraphNode : public llvm::ilist_node<InstanceGraphNode> {
93  using InstanceList = llvm::iplist<InstanceRecord>;
94 
95 public:
96  InstanceGraphNode() : module(nullptr) {}
97 
98  /// Get the module that this node is tracking.
99  HWModuleLike getModule() const { return module; }
100 
101  /// Iterate the instance records in this module.
102  using iterator = detail::AddressIterator<InstanceList::iterator>;
103  iterator begin() { return instances.begin(); }
104  iterator end() { return instances.end(); }
105 
106  /// Return true if there are no more instances of this module.
107  bool noUses() { return !firstUse; }
108 
109  /// Return true if this module has exactly one use.
110  bool hasOneUse() { return llvm::hasSingleElement(uses()); }
111 
112  /// Get the number of direct instantiations of this module.
113  size_t getNumUses() { return std::distance(usesBegin(), usesEnd()); }
114 
115  /// Iterator for module uses.
116  struct UseIterator
117  : public llvm::iterator_facade_base<
118  UseIterator, std::forward_iterator_tag, InstanceRecord *> {
119  UseIterator() : current(nullptr) {}
120  UseIterator(InstanceGraphNode *node) : current(node->firstUse) {}
121  InstanceRecord *operator*() const { return current; }
122  using llvm::iterator_facade_base<UseIterator, std::forward_iterator_tag,
123  InstanceRecord *>::operator++;
124  UseIterator &operator++() {
125  assert(current && "incrementing past end");
126  current = current->nextUse;
127  return *this;
128  }
129  bool operator==(const UseIterator &other) const {
130  return current == other.current;
131  }
132 
133  private:
134  InstanceRecord *current;
135  };
136 
137  /// Iterate the instance records which instantiate this module.
138  UseIterator usesBegin() { return {this}; }
139  UseIterator usesEnd() { return {}; }
140  llvm::iterator_range<UseIterator> uses() {
141  return llvm::make_range(usesBegin(), usesEnd());
142  }
143 
144  /// Record a new instance op in the body of this module. Returns a newly
145  /// allocated InstanceRecord which will be owned by this node.
146  InstanceRecord *addInstance(HWInstanceLike instance,
147  InstanceGraphNode *target);
148 
149 private:
150  friend class InstanceRecord;
151 
152  InstanceGraphNode(const InstanceGraphNode &) = delete;
153 
154  /// Record that a module instantiates this module.
155  void recordUse(InstanceRecord *record);
156 
157  /// The module.
158  HWModuleLike module;
159 
160  /// List of instance operations in this module. This member owns the
161  /// InstanceRecords, which may be pointed to by other InstanceGraphNode's use
162  /// lists.
163  InstanceList instances;
164 
165  /// List of instances which instantiate this module.
166  InstanceRecord *firstUse = nullptr;
167 
168  // Provide access to the constructor.
169  friend class InstanceGraphBase;
170 };

172 /// This graph tracks modules and where they are instantiated. This is intended
173 /// to be used as a cached analysis on circuits. This class can be used
174 /// to walk the modules efficiently in a bottom-up or top-down order.
175 ///
176 /// To use this class, retrieve a cached copy from the analysis manager:
177 /// auto &instanceGraph = getAnalysis<InstanceGraph>(getOperation());
178 class InstanceGraphBase {
179  /// This is the list of InstanceGraphNodes in the graph.
180  using NodeList = llvm::iplist<InstanceGraphNode>;
181 
182 public:
183  virtual ~InstanceGraphBase();
184 
185  /// Look up an InstanceGraphNode for a module.
186  InstanceGraphNode *lookup(HWModuleLike op);
187 
188  /// Lookup an module by name.
189  InstanceGraphNode *lookup(StringAttr name);
190 
191  /// Lookup an InstanceGraphNode for a module.
192  InstanceGraphNode *operator[](HWModuleLike op) { return lookup(op); }
193 
194  /// Look up the referenced module from an InstanceOp. This will use a
195  /// hashtable lookup to find the module, where
196  /// InstanceOp.getReferencedModule() will be a linear search through the IR.
197  HWModuleLike getReferencedModule(HWInstanceLike op);
198 
199  /// Check if child is instantiated by a parent.
200  bool isAncestor(HWModuleLike child, HWModuleLike parent);
201 
202  /// Get the node corresponding to the top-level module of a circuit.
203  virtual InstanceGraphNode *getTopLevelNode() = 0;
204 
205  /// Get the nodes corresponding to the inferred top-level modules of a
206  /// circuit.
207  FailureOr<llvm::ArrayRef<InstanceGraphNode *>> getInferredTopLevelNodes();
208 
209  /// Return the parent under which all nodes are nested.
210  Operation *getParent() { return parent; }
211 
212  /// Returns pointer to member of operation list.
213  static NodeList InstanceGraphBase::*getSublistAccess(Operation *) {
214  return &InstanceGraphBase::nodes;
215  }
216 
217  /// Iterate through all modules.
218  using iterator = detail::AddressIterator<NodeList::iterator>;
219  iterator begin() { return nodes.begin(); }
220  iterator end() { return nodes.end(); }

222  //===-------------------------------------------------------------------------
223  // Methods to keep an InstanceGraph up to date.
224  //
225  // These methods are not thread safe. Make sure that modifications are
226  // properly synchronized or performed in a serial context. When the

229 
230  /// Add a newly created module to the instance graph.

232

245  InstanceGraphBase(Operation *parent);
246  InstanceGraphBase(const InstanceGraphBase &) = delete;
247 
248  /// Get the node corresponding to the module. If the node has does not exist
249  /// yet, it will be created.
250  InstanceGraphNode *getOrAddNode(StringAttr name);
251 
252  /// The node under which all modules are nested.
253  Operation *parent;
254 
255  /// The storage for graph nodes, with deterministic iteration.
256  NodeList nodes;
257 
258  /// This maps each operation to its graph node.
259  llvm::DenseMap<Attribute, InstanceGraphNode *> nodeMap;
260 
261  /// A caching of the inferred top level module(s).
262  llvm::SmallVector<InstanceGraphNode *> inferredTopLevelNodes;

268 template <typename T>
269 inline static T &formatInstancePath(T &into, const InstancePath &path) {
270  into << "$root";
271  for (auto inst : path)
272  into << "/" << inst.getInstanceName() << ":"
273  << inst.getReferencedModuleName();
274  return into;
275 }

278 static T &operator<<(T &os, const InstancePath &path) {
279  return formatInstancePath(os, path);
