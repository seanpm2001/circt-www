23 ParseResult circt::om::ClassOp::parse(OpAsmParser &parser,
24  OperationState &state) {
25  // Parse the Class symbol name.
26  StringAttr symName;
27  if (parser.parseSymbolName(symName, mlir::SymbolTable::getSymbolAttrName(),
28  state.attributes))
29  return failure();
30 
31  // Parse the formal parameters.
32  SmallVector<OpAsmParser::Argument> args;
33  if (parser.parseArgumentList(args, OpAsmParser::Delimiter::Paren,
34  /*allowType=*/true, /*allowAttrs=*/false))
35  return failure();
36 
37  // Parse the optional attribute dictionary.
38  if (failed(parser.parseOptionalAttrDictWithKeyword(state.attributes)))
39  return failure();
40 
41  // Parse the body.
42  Region *region = state.addRegion();
43  if (parser.parseRegion(*region, args))
44  return failure();
45 
46  // If the region was empty, add an empty block so it's still a SizedRegion<1>.
47  if (region->empty())
48  region->emplaceBlock();
49 
50  // Remember the formal parameter names in an attribute.
51  auto argNames = llvm::map_range(args, [&](OpAsmParser::Argument arg) {
52  return StringAttr::get(parser.getContext(), arg.ssaName.name.drop_front());
53  });
54  state.addAttribute(
55  "formalParamNames",
56  ArrayAttr::get(parser.getContext(), SmallVector<Attribute>(argNames)));
57 
58  return success();
59 }

74 void circt::om::ClassOp::print(OpAsmPrinter &printer) {
75  // Print the Class symbol name.
76  printer << " @";
77  printer << getSymName();
78 
79  // Retrieve the formal parameter names and values.
80  auto argNames = SmallVector<StringRef>(
81  getFormalParamNames().getAsValueRange<StringAttr>());
82  ArrayRef<BlockArgument> args = getBodyBlock()->getArguments();
83 
84  // Print the formal parameters.
85  printer << '(';
86  for (size_t i = 0, e = args.size(); i < e; ++i) {
87  printer << '%' << argNames[i] << ": " << args[i].getType();
88  if (i < e - 1)
89  printer << ", ";
90  }
91  printer << ") ";
92 
93  // Print the optional attribute dictionary.
94  SmallVector<StringRef> elidedAttrs{getSymNameAttrName(),
95  getFormalParamNamesAttrName()};
96  printer.printOptionalAttrDictWithKeyword(getOperation()->getAttrs(),
97  elidedAttrs);
98 
99  // Print the body.
100  printer.printRegion(getBody(), /*printEntryBlockArgs=*/false,
101  /*printBlockTerminators=*/true);
102 }

104 LogicalResult circt::om::ClassOp::verify() {
105  // Verify the formal parameter names match up with the values.
106  if (getFormalParamNames().size() != getBodyBlock()->getArguments().size()) {
107  auto error = emitOpError(
108  "formal parameter name list doesn't match formal parameter value list");
109  error.attachNote(getLoc())
110  << "formal parameter names: " << getFormalParamNames();
111  error.attachNote(getLoc())
112  << "formal parameter values: " << getBodyBlock()->getArguments();
113  return error;
114  }
115 
116  return success();
117 }

145 LogicalResult
146 circt::om::ObjectOp::verifySymbolUses(SymbolTableCollection &symbolTable) {
147  // Get the containing ModuleOp.
148  auto moduleOp = getOperation()->getParentOfType<ModuleOp>();
149 
150  // Verify the result type is the same as the referred-to class.
151  StringAttr resultClassName = getResult().getType().getClassName().getAttr();
152  StringAttr className = getClassNameAttr();
153  if (resultClassName != className)
154  return emitOpError("result type (")
155  << resultClassName << ") does not match referred to class ("
156  << className << ')';
157 
158  // Verify the referred to ClassOp exists.
159  auto classDef = dyn_cast_or_null<ClassOp>(
160  symbolTable.lookupSymbolIn(moduleOp, className));
161  if (!classDef)
162  return emitOpError("refers to non-existant class (") << className << ')';
163 
164  auto actualTypes = getActualParams().getTypes();
165  auto formalTypes = classDef.getBodyBlock()->getArgumentTypes();
166 
167  // Verify the actual parameter list matches the formal parameter list.
168  if (actualTypes.size() != formalTypes.size()) {
169  auto error = emitOpError(
170  "actual parameter list doesn't match formal parameter list");
171  error.attachNote(classDef.getLoc())
172  << "formal parameters: " << classDef.getBodyBlock()->getArguments();
173  error.attachNote(getLoc()) << "actual parameters: " << getActualParams();
174  return error;
175  }
176 
177  // Verify the actual parameter types match the formal parameter types.
178  for (size_t i = 0, e = actualTypes.size(); i < e; ++i) {
179  if (actualTypes[i] != formalTypes[i]) {
180  return emitOpError("actual parameter type (")
181  << actualTypes[i] << ") doesn't match formal parameter type ("
182  << formalTypes[i] << ')';
183  }
184  }
185 
186  return success();
187 }

193 LogicalResult
194 circt::om::ObjectFieldOp::verifySymbolUses(SymbolTableCollection &symbolTable) {
195  // Get the containing ModuleOp.
196  auto moduleOp = getOperation()->getParentOfType<ModuleOp>();
197 
198  // Get the ObjectInstOp and the ClassOp it is an instance of.
199  ObjectOp objectInst = getObject().getDefiningOp<ObjectOp>();
200  ClassOp classDef = cast<ClassOp>(
201  symbolTable.lookupSymbolIn(moduleOp, objectInst.getClassNameAttr()));
202 
203  // Traverse the field path, verifying each field exists.
204  Value finalField;
205  auto fields = SmallVector<FlatSymbolRefAttr>(
206  getFieldPath().getAsRange<FlatSymbolRefAttr>());
207  for (size_t i = 0, e = fields.size(); i < e; ++i) {
208  // Verify the field exists on the ClassOp.
209  auto field = fields[i];
210  ClassFieldOp fieldDef =
211  cast_or_null<ClassFieldOp>(symbolTable.lookupSymbolIn(classDef, field));
212  if (!fieldDef) {
213  auto error = emitOpError("referenced non-existant field ") << field;
214  error.attachNote(classDef.getLoc()) << "class defined here";
215  return error;
216  }
217 
218  // If there are more fields, verify the current field is of ClassType, and
219  // look up the ClassOp for that field.
220  if (i < e - 1) {
221  auto classType = fieldDef.getValue().getType().dyn_cast<ClassType>();
222  if (!classType)
223  return emitOpError("nested field access into ")
224  << field << " requires a ClassType, but found "
225  << fieldDef.getValue().getType();
226 
227  // The nested ClassOp must exist, since a field with ClassType must be
228  // an ObjectInstOp, which already verifies the class exists.
229  classDef = cast<ClassOp>(
230  symbolTable.lookupSymbolIn(moduleOp, classType.getClassName()));
231 
232  // Proceed to the next field in the path.
233  continue;
234  }
235 
236  // On the last iteration down the path, save the final field being accessed.
237  finalField = fieldDef.getValue();
238  }
239 
240  // Verify the accessed field type matches the result type.
241  if (finalField.getType() != getResult().getType())
242  return emitOpError("expected type ")
243  << getResult().getType() << ", but accessed field has type "
244  << finalField.getType();
245 
246  return success();
247 }
