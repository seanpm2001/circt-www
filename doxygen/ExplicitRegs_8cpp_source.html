22 class ExplicitRegsPass : public ExplicitRegsBase<ExplicitRegsPass> {
23 public:
24  void runOnOperation() override;

26 private:
27  // Recursively routes value v backwards through the pipeline, adding new
28  // registers to 'stage' if the value was not already registered in the stage.
29  // Returns the registerred version of 'v' through 'stage'.
30  Value routeThroughStage(OpOperand &v, PipelineStageOp stage);
31 
32  // A mapping storing whether a given stage register constains a registerred
33  // version of a given value. The registered version will be a backedge during
34  // pipeline body analysis. Once the entire body has been analyzed, the
35  // pipeline.stage operations will be replaced with pipeline.stage.register
36  // operations containing the requested regs, and the backedge will be
37  // replaced. MapVector ensures deterministic iteration order, which in turn
38  // ensures determinism during stage op IR emission.
39  DenseMap<PipelineStageOp, llvm::MapVector<Value, Backedge>> stageRegMap;
40 
41  // A linked list of stages in the pipeline. Allows for easily looking up the
42  // predecessor stage of a given stage.
43  DenseMap<PipelineStageOp, PipelineStageOp> stagePredecessor;
44 
45  std::shared_ptr<BackedgeBuilder> bb;
46 };

51 Value ExplicitRegsPass::routeThroughStage(OpOperand &v, PipelineStageOp stage) {
52  Value retVal = v.get();
53  if (!stage) {
54  // Recursive base case - nothing to route (v is a block operand).
55  return retVal;
56  }
57 
58  auto regIt = stageRegMap[stage].find(retVal);
59  if (regIt != stageRegMap[stage].end()) {
60  // 'v' is already registered in 'stage'.
61  return regIt->second;
62  }
63 
64  auto *definingOp = retVal.getDefiningOp();
65  // Value is a block arg, a constant or defined after the provided stage
66  // - early exit here.
67  if (definingOp && (definingOp->hasTrait<OpTrait::ConstantLike>() ||
68  !definingOp->isBeforeInBlock(stage)))
69  return retVal;
70 
71  // Value is defined before the provided stage - route it through the stage.
72  auto regBackedge = bb->get(retVal.getType());
73  stageRegMap[stage].insert({retVal, regBackedge});
74  retVal = regBackedge;
75 
76  // Recurse - recursion will only create a new backedge if necessary.
77  auto predStageIt = stagePredecessor.find(stage);
78  assert(predStageIt != stagePredecessor.end() &&
79  "stage should have been registered before calling this function");
80  PipelineStageOp predecessorStage = predStageIt->second;
81  if (predecessorStage)
82  routeThroughStage(v, predecessorStage);
83 
84  return retVal;
85 }

87 void ExplicitRegsPass::runOnOperation() {
88  auto pipeline = getOperation();
89  OpBuilder b(getOperation().getContext());
90  bb = std::make_shared<BackedgeBuilder>(b, getOperation().getLoc());
91 
92  // A list of stages in the pipeline in the order which they appear.
93  SmallVector<PipelineStageOp> stageList;
94  PipelineStageOp currStage = nullptr;
95  // Iterate over the pipeline body in-order (!).
96  for (auto &op : *pipeline.getBodyBlock()) {
97  if (auto stageOp = dyn_cast<PipelineStageOp>(&op)) {
98  stagePredecessor[stageOp] = currStage;
99  currStage = stageOp;
100  continue;
101  }
102 
103  // Check the operands of this operation to see if any of them cross a
104  // stage boundary.
105  for (OpOperand &operand : op.getOpOperands()) {
106  Value reroutedValue = routeThroughStage(operand, currStage);
107  if (reroutedValue != operand.get())
108  op.setOperand(operand.getOperandNumber(), reroutedValue);
109  }
110  }
111 
112  // All values have been recorded through the stages. Now, replace the
113  // stages with new stage operations containing the required registers.
114  for (auto &[stageOp, regMap] : stageRegMap) {
115  b.setInsertionPoint(stageOp);
116  auto predStageOp = stagePredecessor[stageOp];
117 
118  // Gather register inputs to this stage, either from a predecessor stage
119  // or from the original op.
120  llvm::SmallVector<Value> regIns;
121  for (auto &[value, backedge] : regMap) {
122  if (predStageOp) {
123  // Grab the value if registerred through the predecessor op, else,
124  // use the raw value.
125  auto predRegIt = stageRegMap[predStageOp].find(value);
126  if (predRegIt != stageRegMap[predStageOp].end()) {
127  regIns.push_back(predRegIt->second);
128  continue;
129  }
130  }
131  // Not in predecessor stage - must be the original value.
132  regIns.push_back(value);
133  }
134 
135  auto newStageOp = b.create<PipelineStageRegisterOp>(
136  stageOp.getLoc(), stageOp.getWhen(), regIns);
137  stageOp.getValid().replaceAllUsesWith(newStageOp.getValid());
138 
139  // Replace backedges with the outputs of the new stage.
140  for (auto it : llvm::enumerate(regMap)) {
141  auto index = it.index();
142  auto &[value, backedge] = it.value();
143  backedge.setValue(newStageOp.getRegOuts()[index]);
144  }
145  stageOp.erase();
146  }
147 
148  // Clear internal state. See https://github.com/llvm/circt/issues/3235
149  stageRegMap.clear();
150  stagePredecessor.clear();
151 }
