23 LogicalResult firtool::populateCHIRRTLToLowFIRRTL(mlir::PassManager &pm,
24  const FirtoolOptions &opt,
25  ModuleOp module,
26  StringRef inputFilename) {
27  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createLowerIntrinsicsPass());
28 
29  if (!opt.disableOptimization)
30  pm.nest<firrtl::CircuitOp>().nest<firrtl::FModuleOp>().addPass(
31  mlir::createCSEPass());
32 
33  pm.nest<firrtl::CircuitOp>().nest<firrtl::FModuleOp>().addPass(
34  firrtl::createDropNamesPass(opt.getPreserveMode()));
35 
36  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createInjectDUTHierarchyPass());
37 
38  pm.nest<firrtl::CircuitOp>().nest<firrtl::FModuleOp>().addPass(
39  firrtl::createLowerCHIRRTLPass());
40 
41  // Run LowerMatches before InferWidths, as the latter does not support the
42  // match statement, but it does support what they lower to.
43  pm.nest<firrtl::CircuitOp>().nest<firrtl::FModuleOp>().addPass(
44  firrtl::createLowerMatchesPass());
45 
46  // Width inference creates canonicalization opportunities.
47  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createInferWidthsPass());
48 
49  pm.nest<firrtl::CircuitOp>().addPass(
50  firrtl::createMemToRegOfVecPass(opt.replSeqMem, opt.ignoreReadEnableMem));
51 
52  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createInferResetsPass());
53 
54  if (opt.exportChiselInterface) {
55  if (opt.chiselInterfaceOutDirectory.empty()) {
56  pm.nest<firrtl::CircuitOp>().addPass(createExportChiselInterfacePass());
57  } else {
58  pm.nest<firrtl::CircuitOp>().addPass(createExportSplitChiselInterfacePass(
59  opt.chiselInterfaceOutDirectory));
60  }
61  }
62 
63  pm.nest<firrtl::CircuitOp>().nestAny().addPass(firrtl::createDropConstPass());
64 
65  if (opt.dedup)
66  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createDedupPass());
67 
68  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createWireDFTPass());
69 
70  if (opt.vbToBV) {
71  pm.addNestedPass<firrtl::CircuitOp>(firrtl::createLowerFIRRTLTypesPass(
72  firrtl::PreserveAggregate::All, firrtl::PreserveAggregate::All));
73  pm.addNestedPass<firrtl::CircuitOp>(firrtl::createVBToBVPass());
74  }
75 
76  if (!opt.lowerMemories)
77  pm.nest<firrtl::CircuitOp>().nest<firrtl::FModuleOp>().addPass(
78  firrtl::createFlattenMemoryPass());
79 
80  // The input mlir file could be firrtl dialect so we might need to clean
81  // things up.
82  pm.addNestedPass<firrtl::CircuitOp>(firrtl::createLowerFIRRTLTypesPass(
83  opt.preserveAggregate, firrtl::PreserveAggregate::None));
84  // Only enable expand whens if lower types is also enabled.
85  auto &modulePM = pm.nest<firrtl::CircuitOp>().nest<firrtl::FModuleOp>();
86  modulePM.addPass(firrtl::createExpandWhensPass());
87  modulePM.addPass(firrtl::createSFCCompatPass());
88 
89  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createInlinerPass());
90 
91  // Preset the random initialization parameters for each module. The current
92  // implementation assumes it can run at a time where every register is
93  // currently in the final module it will be emitted in, all registers have
94  // been created, and no registers have yet been removed.
95  if (opt.isRandomEnabled(FirtoolOptions::RandomKind::Reg))
96  pm.nest<firrtl::CircuitOp>().nest<firrtl::FModuleOp>().addPass(
97  firrtl::createRandomizeRegisterInitPass());
98 
99  if (opt.useOldCheckCombCycles) {
100  if (opt.preserveAggregate == firrtl::PreserveAggregate::None)
101  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createCheckCombCyclesPass());
102  else
103  emitWarning(module.getLoc())
104  << "CheckCombCyclesPass doesn't support aggregate "
105  "values yet so it is skipped\n";
106  } else
107  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createCheckCombLoopsPass());
108 
109  // If we parsed a FIRRTL file and have optimizations enabled, clean it up.
110  if (!opt.disableOptimization)
111  pm.nest<firrtl::CircuitOp>().nest<firrtl::FModuleOp>().addPass(
112  createSimpleCanonicalizerPass());
113 
114  // Run the infer-rw pass, which merges read and write ports of a memory with
115  // mutually exclusive enables.
116  if (!opt.disableOptimization)
117  pm.nest<firrtl::CircuitOp>().nest<firrtl::FModuleOp>().addPass(
118  firrtl::createInferReadWritePass());
119 
120  if (opt.replSeqMem)
121  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createLowerMemoryPass());
122 
123  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createPrefixModulesPass());
124 
125  if (!opt.disableOptimization)
126  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createIMConstPropPass());
127 
128  pm.addNestedPass<firrtl::CircuitOp>(firrtl::createAddSeqMemPortsPass());
129 
130  pm.addPass(firrtl::createCreateSiFiveMetadataPass(
131  opt.replSeqMem, opt.replSeqMemCircuit, opt.replSeqMemFile));
132 
133  pm.addNestedPass<firrtl::CircuitOp>(firrtl::createExtractInstancesPass());
134 
135  // Run passes to resolve Grand Central features. This should run before
136  // BlackBoxReader because Grand Central needs to inform BlackBoxReader where
137  // certain black boxes should be placed. Note: all Grand Central Taps related
138  // collateral is resolved entirely by LowerAnnotations.
139  pm.addNestedPass<firrtl::CircuitOp>(
140  firrtl::createGrandCentralPass(opt.grandCentralInstantiateCompanionOnly));
141 
142  // Read black box source files into the IR.
143  StringRef blackBoxRoot = opt.blackBoxRootPath.empty()
144  ? llvm::sys::path::parent_path(inputFilename)
145  : opt.blackBoxRootPath;
146  pm.nest<firrtl::CircuitOp>().addPass(
147  firrtl::createBlackBoxReaderPass(blackBoxRoot));
148 
149  // Run SymbolDCE as late as possible, but before InnerSymbolDCE. This is for
150  // hierpathop's and just for general cleanup.
151  pm.addNestedPass<firrtl::CircuitOp>(mlir::createSymbolDCEPass());
152 
153  // Run InnerSymbolDCE as late as possible, but before IMDCE.
154  pm.addPass(firrtl::createInnerSymbolDCEPass());
155 
156  // The above passes, IMConstProp in particular, introduce additional
157  // canonicalization opportunities that we should pick up here before we
158  // proceed to output-specific pipelines.
159  if (!opt.disableOptimization) {
160  pm.nest<firrtl::CircuitOp>().nest<firrtl::FModuleOp>().addPass(
161  createSimpleCanonicalizerPass());
162  pm.nest<firrtl::CircuitOp>().nest<firrtl::FModuleOp>().addPass(
163  circt::firrtl::createRegisterOptimizerPass());
164  pm.addPass(firrtl::createIMDeadCodeElimPass());
165  }
166 
167  if (opt.emitOMIR)
168  pm.nest<firrtl::CircuitOp>().addPass(
169  firrtl::createEmitOMIRPass(opt.omirOutFile));
170 
171  // Always run this, required for legalization.
172  pm.nest<firrtl::CircuitOp>().nest<firrtl::FModuleOp>().addPass(
173  firrtl::createMergeConnectionsPass(
174  !opt.disableAggressiveMergeConnections.getValue()));
175 
176  if (!opt.disableOptimization)
177  pm.nest<firrtl::CircuitOp>().nest<firrtl::FModuleOp>().addPass(
178  firrtl::createVectorizationPass());
179 
180  return success();
181 }

183 LogicalResult firtool::populateLowFIRRTLToHW(mlir::PassManager &pm,
184  const FirtoolOptions &opt) {
185  // Remove TraceAnnotations and write their updated paths to an output
186  // annotation file.
187  if (opt.outputAnnotationFilename.empty())
188  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createResolveTracesPass());
189  else
190  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createResolveTracesPass(
191  opt.outputAnnotationFilename.getValue()));
192 
193  // Lower the ref.resolve and ref.send ops and remove the RefType ports.
194  // LowerToHW cannot handle RefType so, this pass must be run to remove all
195  // RefType ports and ops.
196  pm.nest<firrtl::CircuitOp>().addPass(firrtl::createLowerXMRPass());
197 
198  pm.addPass(createLowerFIRRTLToHWPass(
199  opt.enableAnnotationWarning.getValue(),
200  opt.emitChiselAssertsAsSVA.getValue(), opt.addMuxPragmas.getValue(),
201  !opt.isRandomEnabled(FirtoolOptions::RandomKind::Mem),
202  !opt.isRandomEnabled(FirtoolOptions::RandomKind::Reg)));
203 
204  if (!opt.disableOptimization) {
205  auto &modulePM = pm.nest<hw::HWModuleOp>();
206  modulePM.addPass(mlir::createCSEPass());
207  modulePM.addPass(createSimpleCanonicalizerPass());
208  }
209 
210  return success();
211 }

213 LogicalResult firtool::populateHWToSV(mlir::PassManager &pm,
214  const FirtoolOptions &opt) {
215  pm.nest<hw::HWModuleOp>().addPass(seq::createSeqFIRRTLLowerToSVPass(
216  {/*disableRandomization=*/!opt.isRandomEnabled(
217  FirtoolOptions::RandomKind::Reg),
218  /*emitSeparateAlwaysBlocks=*/
219  opt.emitSeparateAlwaysBlocks}));
220  pm.addPass(sv::createHWMemSimImplPass(
221  opt.replSeqMem, opt.ignoreReadEnableMem, opt.addMuxPragmas,
222  !opt.isRandomEnabled(FirtoolOptions::RandomKind::Mem),
223  !opt.isRandomEnabled(FirtoolOptions::RandomKind::Reg),
224  opt.addVivadoRAMAddressConflictSynthesisBugWorkaround));
225 
226  if (opt.extractTestCode)
227  pm.addPass(sv::createSVExtractTestCodePass(opt.etcDisableInstanceExtraction,
228  opt.etcDisableModuleInlining));
229 
230  // If enabled, run the optimizer.
231  if (!opt.disableOptimization) {
232  auto &modulePM = pm.nest<hw::HWModuleOp>();
233  modulePM.addPass(mlir::createCSEPass());
234  modulePM.addPass(createSimpleCanonicalizerPass());
235  modulePM.addPass(mlir::createCSEPass());
236  modulePM.addPass(sv::createHWCleanupPass(
237  /*mergeAlwaysBlocks=*/!opt.emitSeparateAlwaysBlocks));
238  }
239  return success();
240 }
