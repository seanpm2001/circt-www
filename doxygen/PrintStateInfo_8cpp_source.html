50 void PrintStateInfoPass::runOnOperation() {
51  // Print to the output file if one was given, or stdout otherwise.
52  if (stateFile.empty()) {
53  auto result = runOnOperation(llvm::outs());
54  llvm::outs() << "\n";
55  if (failed(result))
56  return signalPassFailure();
57  } else {
58  std::error_code ec;
59  llvm::ToolOutputFile outputFile(stateFile, ec,
60  llvm::sys::fs::OpenFlags::OF_None);
61  if (ec) {
62  mlir::emitError(getOperation().getLoc(), "unable to open state file: ")
63  << ec.message();
64  return signalPassFailure();
65  }
66  if (failed(runOnOperation(outputFile.os())))
67  return signalPassFailure();
68  outputFile.keep();
69  }
70 }

72 LogicalResult
73 PrintStateInfoPass::runOnOperation(llvm::raw_ostream &outputStream) {
74  llvm::json::OStream json(outputStream, 2);
75  bool anyFailed = false;
76  json.array([&] {
77  std::vector<StateInfo> states;
78  for (auto modelOp : getOperation().getOps<ModelOp>()) {
79  auto storageArg = modelOp.getBody().getArgument(0);
80  auto storageType = storageArg.getType().cast<StorageType>();
81  states.clear();
82  if (failed(collectStates(storageArg, 0, states))) {
83  anyFailed = true;
84  return;
85  }
86  llvm::sort(states, [](auto &a, auto &b) { return a.offset < b.offset; });
87 
88  json.object([&] {
89  json.attribute("name", modelOp.getName());
90  json.attribute("numStateBytes", storageType.getSize());
91  json.attributeArray("states", [&] {
92  for (const auto &state : states) {
93  json.object([&] {
94  json.attribute("name", state.name.getValue());
95  json.attribute("offset", state.offset);
96  json.attribute("numBits", state.numBits);
97  auto typeStr = [](StateInfo::Type type) {
98  switch (type) {
99  case StateInfo::Input:
100  return "input";
101  case StateInfo::Output:
102  return "output";
103  case StateInfo::Register:
104  return "register";
105  case StateInfo::Memory:
106  return "memory";
107  case StateInfo::Wire:
108  return "wire";
109  }
110  return "";
111  };
112  json.attribute("type", typeStr(state.type));
113  if (state.type == StateInfo::Memory) {
114  json.attribute("stride", state.memoryStride);
115  json.attribute("depth", state.memoryDepth);
116  }
117  });
118  }
119  });
120  });
121  }
122  });
123  return failure(anyFailed);
124 }

126 LogicalResult
127 PrintStateInfoPass::collectStates(Value storage, unsigned offset,
128  std::vector<StateInfo> &stateInfos) {
129  for (auto *op : storage.getUsers()) {
130  if (auto substorage = dyn_cast<AllocStorageOp>(op)) {
131  if (!substorage.getOffset().has_value()) {
132  substorage.emitOpError(
133  "without allocated offset; run state allocation first");
134  return failure();
135  }
136  if (failed(collectStates(substorage.getOutput(),
137  *substorage.getOffset() + offset, stateInfos)))
138  return failure();
139  continue;
140  }
141  if (!isa<AllocStateOp, RootInputOp, RootOutputOp, AllocMemoryOp>(op))
142  continue;
143  auto opName = op->getAttrOfType<StringAttr>("name");
144  if (!opName || opName.getValue().empty())
145  continue;
146  auto opOffset = op->getAttrOfType<IntegerAttr>("offset");
147  if (!opOffset) {
148  op->emitOpError("without allocated offset; run state allocation first");
149  return failure();
150  }
151  if (isa<AllocStateOp, RootInputOp, RootOutputOp>(op)) {
152  auto result = op->getResult(0);
153  auto &stateInfo = stateInfos.emplace_back();
154  stateInfo.type = StateInfo::Register;
155  if (isa<RootInputOp>(op))
156  stateInfo.type = StateInfo::Input;
157  else if (isa<RootOutputOp>(op))
158  stateInfo.type = StateInfo::Output;
159  else if (auto alloc = dyn_cast<AllocStateOp>(op)) {
160  if (alloc.getTap())
161  stateInfo.type = StateInfo::Wire;
162  }
163  stateInfo.name = opName;
164  stateInfo.offset = opOffset.getValue().getZExtValue() + offset;
165  stateInfo.numBits =
166  result.getType().cast<StateType>().getType().getWidth();
167  continue;
168  }
169  if (auto memOp = dyn_cast<AllocMemoryOp>(op)) {
170  auto stride = op->getAttrOfType<IntegerAttr>("stride");
171  if (!stride) {
172  op->emitOpError("without allocated stride; run state allocation first");
173  return failure();
174  }
175  auto memType = memOp.getType();
176  auto intType = memType.getWordType();
177  auto &stateInfo = stateInfos.emplace_back();
178  stateInfo.type = StateInfo::Memory;
179  stateInfo.name = opName;
180  stateInfo.offset = opOffset.getValue().getZExtValue() + offset;
181  stateInfo.numBits = intType.getWidth();
182  stateInfo.memoryStride = stride.getValue().getZExtValue();
183  stateInfo.memoryDepth = memType.getNumWords();
184  continue;
185  }
186  }
187  return success();
188 }
