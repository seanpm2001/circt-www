43  def instantiate(self, cls: type["DataclassInstance"],
44  *args: Any) -> "DataclassInstance":
45  """Instantiate an Object with a dataclass type and actual parameters."""
46 
47  # Convert the class name and actual parameters to Attributes within the
48  # Evaluator's context.
49  with self.module.context:
50  # Get the class name from the provided dataclass name.
51  class_name = StringAttr.get(cls.__name__)
52 
53  # Get the actual parameter Attributes from the supplied variadic
54  # arguments. This relies on the circt.support helpers to convert from
55  # Python objects to Attributes.
56  actual_params = var_to_attribute(list(args))
57 
58  # Call the base instantiate method.
59  obj = super().instantiate(class_name, actual_params)
60 
61  # Convert the field names of the class we are instantiating to StringAttrs
62  # within the Evaluator's context.
63  with self.module.context:
64  field_names = [StringAttr.get(field.name) for field in fields(cls)]
65 
66  # Convert the instantiated Object fields from Attributes to Python objects.
67  # This will be generalized to support Objects in fields soon.
68  object_fields = {}
69  for field_name in field_names:
70  # Get the field from the object.
71  field = obj.get_field(field_name)
72 
73  # Convert the field value to a Python object. This relies on the
74  # circt.support helpers to convert from Attribute to Python objects.
75  field_value = attribute_to_var(field)
76 
77  # Save this field in the keyword argument dictionary that will be passed
78  # to the dataclass constructor.
79  object_fields[field_name.value] = field_value
80 
81  # Instantiate a Python object of the requested class.
82  return cls(**object_fields)

84  def _handle_diagnostic(self, diagnostic: Diagnostic) -> bool:

86
