50 /// Return true if a value is module-scoped driven by a value of a specific
51 /// type.
52 template <typename A, typename... B>
53 static bool isModuleScopedDrivenBy(Value val, bool lookThroughWires,
54  bool lookThroughNodes,
55  bool lookThroughCasts) {
56  val = getModuleScopedDriver(val, lookThroughWires, lookThroughNodes,
57  lookThroughCasts);
58 
59  if (!val)
60  return false;
61 
62  auto *op = val.getDefiningOp();
63  if (!op)
64  return false;
65 
66  return isa<A, B...>(op);
67 }

136 /// If reftype, return wrapped base type. Otherwise (if base), return as-is.
137 inline FIRRTLBaseType getBaseType(FIRRTLType type) {
138  return TypeSwitch<FIRRTLType, FIRRTLBaseType>(type)
139  .Case<FIRRTLBaseType>([](auto base) { return base; })
140  .Case<RefType>([](auto ref) { return ref.getType(); });
141 }
142 
143 /// Return base type or passthrough if FIRRTLType, else null.
144 inline FIRRTLBaseType getBaseTypeOrNull(Type type) {
145  auto ftype = dyn_cast_or_null<FIRRTLType>(type);
146  if (!ftype)
147  return {};
148  return getBaseType(ftype);
149 }

151 /// Get base type if isa<> the requested type, else null.
152 template <typename T>
153 inline T getBaseOfType(Type type) {
154  return dyn_cast_or_null<T>(getBaseTypeOrNull(type));
155 }

157 /// Return a FIRRTLType with its base type component mutated by the given
158 /// function. (i.e., ref<T> -> ref<f(T)> and T -> f(T)).
159 inline FIRRTLType mapBaseType(FIRRTLType type,
160  function_ref<FIRRTLBaseType(FIRRTLBaseType)> fn) {
161  return TypeSwitch<FIRRTLType, FIRRTLType>(type)
162  .Case<FIRRTLBaseType>([&](auto base) { return fn(base); })
163  .Case<RefType>([&](auto ref) {
164  return RefType::get(fn(ref.getType()), ref.getForceable());
165  });
166 }

190 /// Wrapper for llvm::parallelTransformReduce that performs the transform_reduce
191 /// serially when MLIR multi-threading is disabled.
192 /// Does not add a ParallelDiagnosticHandler like mlir::parallelFor.
193 template <class IterTy, class ResultTy, class ReduceFuncTy,
194  class TransformFuncTy>
195 static ResultTy transformReduce(MLIRContext *context, IterTy begin, IterTy end,
196  ResultTy init, ReduceFuncTy reduce,
197  TransformFuncTy transform) {
198  // Parallel when enabled
199  if (context->isMultithreadingEnabled())
200  return llvm::parallelTransformReduce(begin, end, init, reduce, transform);
201 
202  // Serial fallback (from llvm::parallelTransformReduce)
203  for (IterTy i = begin; i != end; ++i)
204  init = reduce(std::move(init), transform(*i));
205  return std::move(init);
206 }
207 
208 /// Range wrapper
209 template <class RangeTy, class ResultTy, class ReduceFuncTy,
210  class TransformFuncTy>
211 static ResultTy transformReduce(MLIRContext *context, RangeTy &&r,
212  ResultTy init, ReduceFuncTy reduce,
213  TransformFuncTy transform) {
214  return transformReduce(context, std::begin(r), std::end(r), init, reduce,
215  transform);
216 }
