34 /// Expand a TraceNameAnnotation (which has don't touch semantics) into a
35 /// TraceAnnotation (which does NOT have don't touch semantics) and separate
36 /// DontTouchAnnotations for targets that are not modules, external modules, or
37 /// instances (as these targets are not valid for a don't touch).
38 LogicalResult circt::firrtl::applyTraceName(const AnnoPathValue &target,
39  DictionaryAttr anno,
40  ApplyState &state) {
41 
42  auto *context = anno.getContext();
43 
44  NamedAttrList trace, dontTouch;
45  for (auto namedAttr : anno.getValue()) {
46  if (namedAttr.getName() == "class") {
47  trace.append("class", StringAttr::get(context, traceAnnoClass));
48  dontTouch.append("class", StringAttr::get(context, dontTouchAnnoClass));
49  continue;
50  }
51  trace.append(namedAttr);
52 
53  // When we see the "target", check to see if this is not targeting a module,
54  // extmodule, or instance (as these are invalid "don't touch" targets). If
55  // it is not, then add a DontTouchAnnotation.
56  if (namedAttr.getName() == "target" &&
57  !target.isOpOfType<FModuleOp, FExtModuleOp, InstanceOp>()) {
58  dontTouch.append(namedAttr);
59  state.addToWorklistFn(DictionaryAttr::getWithSorted(context, dontTouch));
60  }
61  }
62 
63  state.addToWorklistFn(DictionaryAttr::getWithSorted(context, trace));
64 
65  return success();
66 }

68 struct ResolveTracesPass : public ResolveTracesBase<ResolveTracesPass> {
69  using ResolveTracesBase::outputAnnotationFilename;
70 
71  void runOnOperation() override;
72 
73 private:
74  /// Stores a pointer to an NLA Table. This is populated during
75  /// runOnOperation.
76  NLATable *nlaTable;
77 
78  /// Internal implementation that updates an Annotation to add a "target" field
79  /// based on the current location of the annotation in the circuit. The value
80  /// of the "target" will be a local target if the Annotation is local and a
81  /// non-local target if the Annotation is non-local.
82  bool updateTargetImpl(Annotation &anno, FModuleLike &module,
83  FIRRTLBaseType type, StringRef name) {
84  if (!anno.isClass(traceAnnoClass))
85  return false;
86 
87  LLVM_DEBUG(llvm::dbgs() << " - before: " << anno.getDict() << "\n");
88 
89  SmallString<64> newTarget("~");
90  newTarget.append(module->getParentOfType<CircuitOp>().getName());
91  newTarget.append("|");
92 
93  if (auto nla = anno.getMember<FlatSymbolRefAttr>("circt.nonlocal")) {
94  hw::HierPathOp path = nlaTable->getNLA(nla.getAttr());
95  for (auto part : path.getNamepath().getValue().drop_back()) {
96  auto inst = cast<hw::InnerRefAttr>(part);
97  newTarget.append(inst.getModule());
98  newTarget.append("/");
99  newTarget.append(inst.getName());
100  newTarget.append(":");
101  }
102  }
103  newTarget.append(module.getModuleName());
104  newTarget.append(">");
105 
106  newTarget.append(name);
107 
108  // Add information if this is an aggregate.
109  auto targetFieldID = anno.getFieldID();
110  while (targetFieldID) {
111  TypeSwitch<FIRRTLBaseType>(type)
112  .Case<FVectorType>([&](FVectorType vector) {
113  auto index = vector.getIndexForFieldID(targetFieldID);
114  newTarget.append("[");
115  newTarget.append(Twine(index).str());
116  newTarget.append("]");
117  type = vector.getElementType();
118  targetFieldID -= vector.getFieldID(index);
119  })
120  .Case<BundleType>([&](BundleType bundle) {
121  auto index = bundle.getIndexForFieldID(targetFieldID);
122  newTarget.append(".");
123  newTarget.append(bundle.getElementName(index));
124  type = bundle.getElementType(index);
125  targetFieldID -= bundle.getFieldID(index);
126  })
127  .Default([&](auto) { targetFieldID = 0; });
128  }
129 
130  anno.setMember("target", StringAttr::get(module->getContext(), newTarget));
131 
132  LLVM_DEBUG(llvm::dbgs() << " after: " << anno.getDict() << "\n");
133 
134  return true;
135  }
136 
137  /// Add a "target" field to a port Annotation that indicates the current
138  /// location of the port in the circuit.
139  bool updatePortTarget(FModuleLike &module, Annotation &anno,
140  unsigned portIdx) {
141  auto type = getBaseType(cast<FIRRTLType>(module.getPortType(portIdx)));
142  return updateTargetImpl(anno, module, type, module.getPortName(portIdx));
143  }
144 
145  /// Add a "target" field to an Annotation that indicates the current location
146  /// of a component in the circuit.
147  bool updateTarget(FModuleLike &module, Operation *op, Annotation &anno) {
148 
149  // If this operation doesn't have a name, then do nothing.
150  StringAttr name = op->getAttrOfType<StringAttr>("name");
151  if (!name)
152  return false;
153 
154  // Get the type of the operation either by checking for the
155  // result targeted by symbols on it (which are used to track the op)
156  // or by inspecting its single result.
157  auto is = dyn_cast<hw::InnerSymbolOpInterface>(op);
158  Type type;
159  if (is && is.getTargetResult())
160  type = is.getTargetResult().getType();
161  else {
162  if (op->getNumResults() != 1)
163  return false;
164  type = op->getResultTypes().front();
165  }
166 
167  auto baseType = getBaseType(cast<FIRRTLType>(type));
168  return updateTargetImpl(anno, module, baseType, name);
169  }
170 
171  /// Add a "target" field to an Annotation on a Module that indicates the
172  /// current location of the module. This will be local or non-local depending
173  /// on the Annotation.
174  bool updateModuleTarget(FModuleLike &module, Annotation &anno) {
175 
176  if (!anno.isClass(traceAnnoClass))
177  return false;
178 
179  LLVM_DEBUG(llvm::dbgs() << " - before: " << anno.getDict() << "\n");
180 
181  SmallString<64> newTarget("~");
182  newTarget.append(module->getParentOfType<CircuitOp>().getName());
183  newTarget.append("|");
184 
185  if (auto nla = anno.getMember<FlatSymbolRefAttr>("circt.nonlocal")) {
186  hw::HierPathOp path = nlaTable->getNLA(nla.getAttr());
187  for (auto part : path.getNamepath().getValue().drop_back()) {
188  auto inst = cast<hw::InnerRefAttr>(part);
189  newTarget.append(inst.getModule());
190  newTarget.append("/");
191  newTarget.append(inst.getName());
192  newTarget.append(":");
193  }
194  }
195  newTarget.append(module.getModuleName());
196 
197  anno.setMember("target", StringAttr::get(module->getContext(), newTarget));
198 
199  LLVM_DEBUG(llvm::dbgs() << " after: " << anno.getDict() << "\n");
200 
201  return true;
202  }
203 };

205 void ResolveTracesPass::runOnOperation() {
206  LLVM_DEBUG(
207  llvm::dbgs() << "==----- Running ResolveTraces "
208  "-----------------------------------------------===\n");
209 
210  // Populate the NLA Table.
211  nlaTable = &getAnalysis<NLATable>();
212 
213  // Grab the circuit (as this is used a few times below).
214  CircuitOp circuit = getOperation();
215  MLIRContext *context = circuit.getContext();
216 
217  // Function to find all Trace Annotations in the circuit, add a "target" field
218  // to them indicating the current local/non-local target of the operation/port
219  // the Annotation is attached to, copy the annotation into an
220  // "outputAnnotations" return vector, and delete the original Annotation. If
221  // a component or port is targeted by a Trace Annotation it will be given a
222  // symbol to prevent the output Trace Annotation from being made invalid by a
223  // later optimization.
224  auto onModule = [&](FModuleLike moduleLike) {
225  // Output Trace Annotations from this module only.
226  SmallVector<Annotation> outputAnnotations;
227 
228  // A lazily constructed module namespace.
229  std::optional<ModuleNamespace> moduleNamespace;
230 
231  // Return a cached module namespace, lazily constructing it if needed.
232  auto getNamespace = [&](FModuleLike module) -> ModuleNamespace & {
233  if (!moduleNamespace)
234  moduleNamespace = ModuleNamespace(module);
235  return *moduleNamespace;
236  };
237 
238  // Visit the module.
239  AnnotationSet::removeAnnotations(moduleLike, [&](Annotation anno) {
240  if (!updateModuleTarget(moduleLike, anno))
241  return false;
242 
243  outputAnnotations.push_back(anno);
244  return true;
245  });
246 
247  // Visit port annotations.
248  AnnotationSet::removePortAnnotations(
249  moduleLike, [&](unsigned portIdx, Annotation anno) {
250  if (!updatePortTarget(moduleLike, anno, portIdx))
251  return false;
252 
253  getOrAddInnerSym(moduleLike, portIdx, moduleLike.getPortName(portIdx),
254  getNamespace);
255  outputAnnotations.push_back(anno);
256  return true;
257  });
258 
259  // Visit component annotations.
260  moduleLike.walk([&](Operation *component) {
261  AnnotationSet::removeAnnotations(component, [&](Annotation anno) {
262  if (!updateTarget(moduleLike, component, anno))
263  return false;
264 
265  auto module = cast<FModuleOp>(moduleLike);
266  getOrAddInnerSym(
267  component, component->getAttrOfType<StringAttr>("name").getValue(),
268  module, getNamespace);
269  outputAnnotations.push_back(anno);
270  return true;
271  });
272  });
273 
274  return outputAnnotations;
275  };
276 
277  // Function to append one vector after another. This is used to merge results
278  // from parallel executions of "onModule".
279  auto appendVecs = [](auto &&a, auto &&b) {
280  a.append(b.begin(), b.end());
281  return std::forward<decltype(a)>(a);
282  };
283 
284  // Process all the modules in parallel or serially, depending on the
285  // multithreading context.
286  SmallVector<FModuleLike, 0> mods(circuit.getOps<FModuleLike>());
287  auto outputAnnotations = transformReduce(
288  context, mods, SmallVector<Annotation>{}, appendVecs, onModule);
289 
290  // Do not generate an output Annotation file if no Annotations exist.
291  if (outputAnnotations.empty())
292  return markAllAnalysesPreserved();
293 
294  // Write out all the Trace Annotations to a JSON buffer.
295  std::string jsonBuffer;
296  llvm::raw_string_ostream jsonStream(jsonBuffer);
297  llvm::json::OStream json(jsonStream, /*IndentSize=*/2);
298  json.arrayBegin();
299  for (auto anno : outputAnnotations) {
300  json.objectBegin();
301  json.attribute("class", anno.getClass());
302  json.attribute("target", anno.getMember<StringAttr>("target").getValue());
303  json.attribute("chiselTarget",
304  anno.getMember<StringAttr>("chiselTarget").getValue());
305  json.objectEnd();
306  }
307  json.arrayEnd();
308 
309  // Write the JSON-encoded Trace Annotation to a file called
310  // "$circuitName.anno.json". (This is implemented via an SVVerbatimOp that is
311  // inserted before the FIRRTL circuit.
312  OpBuilder b(circuit);
313  auto verbatimOp = b.create<sv::VerbatimOp>(b.getUnknownLoc(), jsonBuffer);
314  hw::OutputFileAttr fileAttr;
315  if (this->outputAnnotationFilename.empty())
316  fileAttr = hw::OutputFileAttr::getFromFilename(
317  context, circuit.getName() + ".anno.json",
318  /*excludeFromFilelist=*/true, false);
319  else
320  fileAttr = hw::OutputFileAttr::getFromFilename(
321  context, outputAnnotationFilename,
322  /*excludeFromFilelist=*/true, false);
323  verbatimOp->setAttr("output_file", fileAttr);
324 
325  return markAllAnalysesPreserved();
326 }
