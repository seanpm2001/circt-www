48 LogicalResult
49 LatencyRetimingPattern::matchAndRewrite(StateOp op,
50  PatternRewriter &rewriter) const {
51  unsigned minPrevLatency = UINT_MAX;
52  SetVector<StateOp> predecessors;
53 
54  if (op.getReset() || op.getEnable())
55  return failure();
56 
57  for (auto input : op.getInputs()) {
58  auto predState = input.getDefiningOp<StateOp>();
59  if (!predState)
60  return failure();
61 
62  if (predState->hasAttr("name") || predState->hasAttr("names"))
63  return failure();
64 
65  if (predState == op)
66  return failure();
67 
68  if (predState.getLatency() != 0 && op.getLatency() != 0 &&
69  predState.getClock() != op.getClock())
70  return failure();
71 
72  if (predState.getEnable() || predState.getReset())
73  return failure();
74 
75  if (llvm::any_of(predState->getUsers(),
76  [&](auto *user) { return user != op; }))
77  return failure();
78 
79  predecessors.insert(predState);
80  minPrevLatency = std::min(minPrevLatency, predState.getLatency());
81  }
82 
83  if (minPrevLatency == 0 || minPrevLatency == UINT_MAX)
84  return failure();
85 
86  op.setLatency(op.getLatency() + minPrevLatency);
87  for (auto prevStateOp : predecessors) {
88  if (!op.getClock() && !op->getParentOfType<ClockDomainOp>())
89  op.getClockMutable().assign(prevStateOp.getClock());
90 
91  statistics.latencyUnitsSaved += minPrevLatency;
92  auto newLatency = prevStateOp.getLatency() - minPrevLatency;
93  prevStateOp.setLatency(newLatency);
94 
95  if (newLatency > 0)
96  continue;
97 
98  prevStateOp.getClockMutable().clear();
99  if (cast<DefineOp>(symCache.getDefinition(prevStateOp.getArcAttr()))
100  .isPassthrough()) {
101  rewriter.replaceOp(prevStateOp, prevStateOp.getInputs());
102  ++statistics.numOpsRemoved;
103  }
104  }
105  statistics.latencyUnitsSaved -= minPrevLatency;
106 
107  return success();
108 }

120 void LatencyRetimingPass::runOnOperation() {
121  SymbolCache cache;
122  cache.addDefinitions(getOperation());
123 
124  LatencyRetimingStatistics statistics;
125 
126  RewritePatternSet patterns(&getContext());
127  patterns.add<LatencyRetimingPattern>(&getContext(), cache, statistics);
128 
129  if (failed(applyPatternsAndFoldGreedily(getOperation(), std::move(patterns))))
130  return signalPassFailure();
131 
132  numOpsRemoved = statistics.numOpsRemoved;
133  latencyUnitsSaved = statistics.latencyUnitsSaved;
