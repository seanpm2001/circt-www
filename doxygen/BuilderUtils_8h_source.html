17 /// A helper union that can represent a `StringAttr`, `StringRef`, or `Twine`.
18 /// It is intended to be used as arguments to an op's `build` function. This
19 /// allows a single builder to accept any flavor value for a string attribute.
20 /// The `get` function can then be used to obtain a `StringAttr` from any of the
21 /// possible variants `StringAttrOrRef` can take.
22 class StringAttrOrRef {
23  using Value = llvm::PointerUnion<StringAttr, StringRef *, Twine *>;
24  Value value;
25 
26 public:
27  StringAttrOrRef() : value() {}
28  StringAttrOrRef(StringAttr attr) : value(attr) {}
29  StringAttrOrRef(const StringRef &str)
30  : value(const_cast<StringRef *>(&str)) {}
31  StringAttrOrRef(const Twine &twine) : value(const_cast<Twine *>(&twine)) {}
32 
33  /// Return the represented string as a `StringAttr`.
34  StringAttr get(MLIRContext *context) const {
35  return TypeSwitch<Value, StringAttr>(value)
36  .Case<StringAttr>([&](auto value) { return value; })
37  .Case<StringRef *, Twine *>(
38  [&](auto value) { return StringAttr::get(context, *value); })
39  .Default([](auto) { return StringAttr{}; });
40  }
41 };
