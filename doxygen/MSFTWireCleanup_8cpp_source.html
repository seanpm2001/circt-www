39 void WireCleanupPass::runOnOperation() {
40  ModuleOp topMod = getOperation();
41  topLevelSyms.addDefinitions(topMod);
42  if (failed(verifyInstances(topMod))) {
43  signalPassFailure();
44  return;
45  }
46 
47  SmallVector<MSFTModuleOp> sortedMods;
48  getAndSortModules(topMod, sortedMods);
49 
50  for (auto mod : sortedMods) {
51  bubbleWiresUp(mod);
52  dedupOutputs(mod);
53  }
54 
55  for (auto mod : llvm::reverse(sortedMods)) {
56  sinkWiresDown(mod);
57  dedupInputs(mod);
58  }
59 }

61 /// Remove outputs driven by the same value.
62 void MSFTPassCommon::dedupOutputs(MSFTModuleOp mod) {
63  Block *body = mod.getBodyBlock();
64  Operation *terminator = body->getTerminator();
65 
66  DenseMap<Value, unsigned> valueToOutputIdx;
67  SmallVector<unsigned> outputMap;
68  llvm::BitVector outputPortsToRemove(terminator->getNumOperands());
69  for (OpOperand &outputVal : terminator->getOpOperands()) {
70  auto existing = valueToOutputIdx.find(outputVal.get());
71  if (existing != valueToOutputIdx.end()) {
72  outputMap.push_back(existing->second);
73  outputPortsToRemove.set(outputVal.getOperandNumber());
74  } else {
75  outputMap.push_back(valueToOutputIdx.size());
76  valueToOutputIdx[outputVal.get()] = valueToOutputIdx.size();
77  }
78  }
79 
80  mod.removePorts(llvm::BitVector(mod.getNumArguments()), outputPortsToRemove);
81  updateInstances(mod, makeSequentialRange(mod.getNumResults()),
82  [&](InstanceOp newInst, InstanceOp oldInst,
83  SmallVectorImpl<Value> &newOperands) {
84  // Operands don't change.
85  llvm::append_range(newOperands, oldInst.getOperands());
86  // The results have to be remapped.
87  for (OpResult res : oldInst.getResults())
88  res.replaceAllUsesWith(
89  newInst.getResult(outputMap[res.getResultNumber()]));
90  });
91 }

93 /// Push up any wires which are simply passed-through.
94 void MSFTPassCommon::bubbleWiresUp(MSFTModuleOp mod) {
95  Block *body = mod.getBodyBlock();
96  Operation *terminator = body->getTerminator();
97  hw::ModulePortInfo ports = mod.getPorts();
98 
99  // Find all "passthough" internal wires, filling 'inputPortsToRemove' as a
100  // side-effect.
101  DenseMap<Value, hw::PortInfo> passThroughs;
102  llvm::BitVector inputPortsToRemove(ports.inputs.size());
103  for (hw::PortInfo inputPort : ports.inputs) {
104  BlockArgument portArg = body->getArgument(inputPort.argNum);
105  bool removePort = true;
106  for (OpOperand user : portArg.getUsers()) {
107  if (user.getOwner() == terminator)
108  passThroughs[portArg] = inputPort;
109  else
110  removePort = false;
111  }
112  if (removePort)
113  inputPortsToRemove.set(inputPort.argNum);
114  }
115 
116  // Find all output ports which we can remove. Fill in 'outputToInputIdx' to
117  // help rewire instantiations later on.
118  DenseMap<unsigned, unsigned> outputToInputIdx;
119  llvm::BitVector outputPortsToRemove(ports.outputs.size());
120  for (hw::PortInfo outputPort : ports.outputs) {
121  assert(outputPort.argNum < terminator->getNumOperands() && "Invalid IR");
122  Value outputValue = terminator->getOperand(outputPort.argNum);
123  auto inputNumF = passThroughs.find(outputValue);
124  if (inputNumF == passThroughs.end())
125  continue;
126  hw::PortInfo inputPort = inputNumF->second;
127  outputToInputIdx[outputPort.argNum] = inputPort.argNum;
128  outputPortsToRemove.set(outputPort.argNum);
129  }
130 
131  // Use MSFTModuleOp's `removePorts` method to remove the ports. It returns a
132  // mapping of the new output port to old output port indices to assist in
133  // updating the instantiations later on.
134  auto newToOldResult =
135  mod.removePorts(inputPortsToRemove, outputPortsToRemove);
136 
137  // Update the instantiations.
138  auto setPassthroughsGetOperands = [&](InstanceOp newInst, InstanceOp oldInst,
139  SmallVectorImpl<Value> &newOperands) {
140  // Re-map the passthrough values around the instance.
141  for (auto idxPair : outputToInputIdx) {
142  size_t outputPortNum = idxPair.first;
143  assert(outputPortNum <= oldInst.getNumResults());
144  size_t inputPortNum = idxPair.second;
145  assert(inputPortNum <= oldInst.getNumOperands());
146  oldInst.getResult(outputPortNum)
147  .replaceAllUsesWith(oldInst.getOperand(inputPortNum));
148  }
149  // Use a sort-merge-join approach to figure out the operand mapping on the
150  // fly.
151  for (size_t operNum = 0, e = oldInst.getNumOperands(); operNum < e;
152  ++operNum)
153  if (!inputPortsToRemove.test(operNum))
154  newOperands.push_back(oldInst.getOperand(operNum));
155  };
156  updateInstances(mod, newToOldResult, setPassthroughsGetOperands);
157 }
