49 LogicalResult
50 InstanceOpLowering::matchAndRewrite(InstanceOp msftInst, OpAdaptor adaptor,
51  ConversionPatternRewriter &rewriter) const {
52  Operation *referencedModule = msftInst.getReferencedModule();
53  if (!referencedModule)
54  return rewriter.notifyMatchFailure(msftInst,
55  "Could not find referenced module");
56  if (!hw::isAnyModule(referencedModule))
57  return rewriter.notifyMatchFailure(
58  msftInst, "Referenced module was not an HW module");
59 
60  StringAttr instHierParamName = rewriter.getStringAttr("__INST_HIER");
61  // Does `mod` have either a parameter named __INST_HIER or implicitly has it?
62  auto hasInstHierParam = [instHierParamName](hw::HWModuleLike mod) {
63  if (!mod)
64  return false;
65  if (isa<MSFTModuleOp>(mod.getOperation()))
66  return true;
67  auto hwmod = dyn_cast<hw::HWModuleOp>(mod.getOperation());
68  if (!hwmod)
69  return false;
70  ArrayAttr params = hwmod.getParametersAttr();
71  return llvm::any_of(params.getAsRange<hw::ParamDeclAttr>(),
72  [instHierParamName](hw::ParamDeclAttr param) {
73  return param.getName() == instHierParamName;
74  });
75  };
76 
77  ArrayAttr paramValues;
78  if (isa<hw::HWModuleExternOp>(referencedModule)) {
79  paramValues = msftInst.getParametersAttr();
80  if (!paramValues)
81  paramValues = rewriter.getArrayAttr({});
82  } else {
83  // If the containing module supports instantiation with the instance path,
84  // use it.
85  auto mod = msftInst->getParentOfType<hw::HWModuleLike>();
86  if (hasInstHierParam(mod)) {
87  auto instAppendParam = hw::ParamExprAttr::get(
88  hw::PEO::StrConcat,
89  {hw::ParamDeclRefAttr::get(instHierParamName, rewriter.getNoneType()),
90  rewriter.getStringAttr("."), msftInst.getSymNameAttr()});
91  paramValues = rewriter.getArrayAttr(
92  {hw::ParamDeclAttr::get("__INST_HIER", instAppendParam)});
93  } else {
94  // Otherwise, create an instance path name assuming that the containing
95  // mod is the top level.
96  std::string instHier;
97  llvm::raw_string_ostream(instHier)
98  << mod.getModuleName() << "." << msftInst.getInstanceName();
99  paramValues = rewriter.getArrayAttr({hw::ParamDeclAttr::get(
100  "__INST_HIER", rewriter.getStringAttr(instHier))});
101  }
102  }
103 
104  auto hwInst = rewriter.create<hw::InstanceOp>(
105  msftInst.getLoc(), referencedModule, msftInst.getInstanceNameAttr(),
106  SmallVector<Value>(adaptor.getOperands().begin(),
107  adaptor.getOperands().end()),
108  paramValues, msftInst.getSymNameAttr());
109  hwInst->setDialectAttrs(msftInst->getDialectAttrs());
110  rewriter.replaceOp(msftInst, hwInst.getResults());
111  return success();
112 }

131 LogicalResult
132 ModuleOpLowering::matchAndRewrite(MSFTModuleOp mod, OpAdaptor adaptor,
133  ConversionPatternRewriter &rewriter) const {
134  if (mod.getBody().empty()) {
135  std::string comment;
136  llvm::raw_string_ostream(comment)
137  << "// Module not generated: \"" << mod.getName() << "\" params "
138  << mod.getParameters();
139  // TODO: replace this with proper comment op when it's created.
140  rewriter.replaceOpWithNewOp<sv::VerbatimOp>(mod, comment);
141  return success();
142  }
143 
144  ArrayAttr params = rewriter.getArrayAttr({hw::ParamDeclAttr::get(
145  rewriter.getStringAttr("__INST_HIER"),
146  rewriter.getStringAttr("INSTANTIATE_WITH_INSTANCE_PATH"))});
147  auto hwmod = rewriter.replaceOpWithNewOp<hw::HWModuleOp>(
148  mod, mod.getNameAttr(), mod.getPorts(), params);
149  rewriter.eraseBlock(hwmod.getBodyBlock());
150  rewriter.inlineRegionBefore(mod.getBody(), hwmod.getBody(),
151  hwmod.getBody().end());
152 
153  auto opOutputFile = mod.getFileName();
154  if (opOutputFile) {
155  auto outputFileAttr = hw::OutputFileAttr::getFromFilename(
156  rewriter.getContext(), *opOutputFile, false, true);
157  hwmod->setAttr("output_file", outputFileAttr);
158  } else if (!outputFile.empty()) {
159  auto outputFileAttr = hw::OutputFileAttr::getFromFilename(
160  rewriter.getContext(), outputFile, false, true);
161  hwmod->setAttr("output_file", outputFileAttr);
162  }
163  hwmod->setDialectAttrs(mod->getDialectAttrs());
164 
165  return success();
166 }

223 void LowerToHWPass::runOnOperation() {
224  auto top = getOperation();
225  auto *ctxt = &getContext();
226 
227  // The `hw::InstanceOp` (which `msft::InstanceOp` lowers to) convenience
228  // builder gets its argNames and resultNames from the `hw::HWModuleOp`. So we
229  // have to lower `msft::MSFTModuleOp` before we lower `msft::InstanceOp`.
230 
231  // Convert everything except instance ops first.
232 
233  ConversionTarget target(*ctxt);
234  target.addIllegalOp<MSFTModuleOp, MSFTModuleExternOp, OutputOp>();
235  target.addLegalDialect<hw::HWDialect>();
236  target.addLegalDialect<sv::SVDialect>();
237 
238  RewritePatternSet patterns(ctxt);
239  patterns.insert<ModuleOpLowering>(ctxt, verilogFile);
240  patterns.insert<ModuleExternOpLowering>(ctxt, verilogFile);
241  patterns.insert<OutputOpLowering>(ctxt);
242  patterns.insert<RemoveOpLowering<EntityExternOp>>(ctxt);
243  patterns.insert<RemoveOpLowering<DesignPartitionOp>>(ctxt);
244 
245  if (failed(applyPartialConversion(top, target, std::move(patterns))))
246  signalPassFailure();
247 
248  // Then, convert the InstanceOps
249  target.addDynamicallyLegalDialect<MSFTDialect>([](Operation *op) {
250  return isa<DynInstDataOpInterface, DeclPhysicalRegionOp,
251  InstanceHierarchyOp>(op);
252  });
253  RewritePatternSet instancePatterns(ctxt);
254  instancePatterns.insert<InstanceOpLowering>(ctxt);
255  if (failed(applyPartialConversion(top, target, std::move(instancePatterns))))
256  signalPassFailure();
257 }
