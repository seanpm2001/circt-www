28 /// Emit a connect between two values.
29 void circt::firrtl::emitConnect(ImplicitLocOpBuilder &builder, Value dst,
30  Value src) {
31  auto dstFType = dst.getType().cast<FIRRTLType>();
32  auto srcFType = src.getType().cast<FIRRTLType>();
33  auto dstType = dstFType.dyn_cast<FIRRTLBaseType>();
34  auto srcType = srcFType.dyn_cast<FIRRTLBaseType>();
35 
36  // Special Connects (non-base, foreign):
37  if (!dstType) {
38  // References use ref.define. Types should match, leave to verifier if not.
39  if (isa<RefType>(dstFType))
40  builder.create<RefDefineOp>(dst, src);
41  else // Other types, give up and leave a connect
42  builder.create<ConnectOp>(dst, src);
43  return;
44  }
45 
46  // If the types are the exact same we can just connect them.
47  if (dstType == srcType && dstType.isPassive() &&
48  !dstType.hasUninferredWidth()) {
49  builder.create<StrictConnectOp>(dst, src);
50  return;
51  }
52 
53  if (auto dstBundle = dstType.dyn_cast<BundleType>()) {
54  // Connect all the bundle elements pairwise.
55  auto numElements = dstBundle.getNumElements();
56  // Check if we are trying to create an illegal connect - just create the
57  // connect and let the verifier catch it.
58  auto srcBundle = srcType.dyn_cast<BundleType>();
59  if (!srcBundle || numElements != srcBundle.getNumElements()) {
60  builder.create<ConnectOp>(dst, src);
61  return;
62  }
63  for (size_t i = 0; i < numElements; ++i) {
64  auto dstField = builder.create<SubfieldOp>(dst, i);
65  auto srcField = builder.create<SubfieldOp>(src, i);
66  if (dstBundle.getElement(i).isFlip)
67  std::swap(dstField, srcField);
68  emitConnect(builder, dstField, srcField);
69  }
70  return;
71  }
72 
73  if (auto dstVector = dstType.dyn_cast<FVectorType>()) {
74  // Connect all the vector elements pairwise.
75  auto numElements = dstVector.getNumElements();
76  // Check if we are trying to create an illegal connect - just create the
77  // connect and let the verifier catch it.
78  auto srcVector = srcType.dyn_cast<FVectorType>();
79  if (!srcVector || numElements != srcVector.getNumElements()) {
80  builder.create<ConnectOp>(dst, src);
81  return;
82  }
83  for (size_t i = 0; i < numElements; ++i) {
84  auto dstField = builder.create<SubindexOp>(dst, i);
85  auto srcField = builder.create<SubindexOp>(src, i);
86  emitConnect(builder, dstField, srcField);
87  }
88  return;
89  }
90 
91  if ((dstType.hasUninferredReset() || srcType.hasUninferredReset()) &&
92  dstType != srcType) {
93  src = builder.create<UninferredResetCastOp>(dstType, src);
94  srcType = dstType;
95  }
96 
97  // Be sure uint, uint -> uint, (uir uint) since we are changing extneding
98  // connect to identity.
99  if (dstType.hasUninferredWidth() || srcType.hasUninferredWidth()) {
100  src = builder.create<UninferredWidthCastOp>(dstType, src);
101  srcType = dstType;
102  }
103 
104  // Handle ground types with possibly uninferred widths.
105  auto dstWidth = dstType.getBitWidthOrSentinel();
106  auto srcWidth = srcType.getBitWidthOrSentinel();
107 
108  // The source must be extended or truncated.
109  if (dstWidth < srcWidth) {
110  // firrtl.tail always returns uint even for sint operands.
111  IntType tmpType = dstType.cast<IntType>();
112  if (tmpType.isSigned())
113  tmpType = UIntType::get(dstType.getContext(), dstWidth);
114  src = builder.create<TailPrimOp>(tmpType, src, srcWidth - dstWidth);
115  // Insert the cast back to signed if needed.
116  if (tmpType != dstType)
117  src = builder.create<AsSIntPrimOp>(dstType, src);
118  } else if (srcWidth < dstWidth) {
119  // Need to extend arg.
120  src = builder.create<PadPrimOp>(src, dstWidth);
121  }
122 
123  // Strict connect requires the types to be completely equal, including
124  // connecting uint<1> to abstract reset types.
125  assert("Connect Types are equal" && dstType == src.getType());
126  builder.create<StrictConnectOp>(dst, src);
127 }

155 /// Return the value that drives another FIRRTL value within module scope. Only
156 /// look backwards through one connection. This is intended to be used in
157 /// situations where you only need to look at the most recent connect, e.g., to
158 /// know if a wire has been driven to a constant. Return null if no driver via
159 /// a connect was found.
160 Value circt::firrtl::getDriverFromConnect(Value val) {
161  for (auto *user : val.getUsers()) {
162  if (auto connect = dyn_cast<FConnectLike>(user)) {
163  if (connect.getDest() != val)
164  continue;
165  return connect.getSrc();
166  }
167  }
168  return nullptr;

171 Value circt::firrtl::getModuleScopedDriver(Value val, bool lookThroughWires,
172  bool lookThroughNodes,
173  bool lookThroughCasts) {
174  // Update `val` to the source of the connection driving `thisVal`. This walks
175  // backwards across users to find the first connection and updates `val` to
176  // the source. This assumes that only one connect is driving `thisVal`, i.e.,
177  // this pass runs after `ExpandWhens`.
178  auto updateVal = [&](Value thisVal) {
179  for (auto *user : thisVal.getUsers()) {
180  if (auto connect = dyn_cast<FConnectLike>(user)) {
181  if (connect.getDest() != val)
182  continue;
183  val = connect.getSrc();
184  return;
185  }
186  }
187  val = nullptr;
188  return;
189  };
190 
191  while (val) {
192  // The value is a port.
193  if (auto blockArg = val.dyn_cast<BlockArgument>()) {
194  FModuleOp op = cast<FModuleOp>(val.getParentBlock()->getParentOp());
195  auto direction = op.getPortDirection(blockArg.getArgNumber());
196  // Base case: this is one of the module's input ports.
197  if (direction == Direction::In)
198  return blockArg;
199  updateVal(blockArg);
200  continue;
201  }
202 
203  auto *op = val.getDefiningOp();
204 
205  // The value is an instance port.
206  if (auto inst = dyn_cast<InstanceOp>(op)) {
207  auto resultNo = val.cast<OpResult>().getResultNumber();
208  // Base case: this is an instance's output port.
209  if (inst.getPortDirection(resultNo) == Direction::Out)
210  return inst.getResult(resultNo);
211  updateVal(val);
212  continue;
213  }
214 
215  // If told to look through wires, continue from the driver of the wire.
216  if (lookThroughWires && isa<WireOp>(op)) {
217  updateVal(op->getResult(0));
218  continue;
219  }
220 
221  // If told to look through nodes, continue from the node input.
222  if (lookThroughNodes && isa<NodeOp>(op)) {
223  val = cast<NodeOp>(op).getInput();
224  continue;
225  }
226 
227  if (lookThroughCasts &&
228  isa<AsUIntPrimOp, AsSIntPrimOp, AsClockPrimOp, AsAsyncResetPrimOp>(
229  op)) {
230  val = op->getOperand(0);
231  continue;
232  }
233 
234  // Look through unary ops generated by emitConnect
235  if (isa<PadPrimOp, TailPrimOp>(op)) {
236  val = op->getOperand(0);
237  continue;
238  }
239 
240  // Base case: this is a constant/invalid or primop.
241  //
242  // TODO: If needed, this could be modified to look through unary ops which
243  // have an unambiguous single driver. This should only be added if a need
244  // arises for it.
245  break;
246  };
247  return val;
248 }

250 bool circt::firrtl::walkDrivers(FIRRTLBaseValue value, bool lookThroughWires,
251  bool lookThroughNodes, bool lookThroughCasts,
252  WalkDriverCallback callback) {
253  // TODO: what do we want to happen when there are flips in the type? Do we
254  // want to filter out fields which have reverse flow?
255  assert(value.getType().isPassive() && "this code was not tested with flips");
256 
257  // This method keeps a stack of wires (or ports) and subfields of those that
258  // it still has to process. It keeps track of which fields in the
259  // destination are attached to which fields of the source, as well as which
260  // subfield of the source we are currently investigating. The fieldID is
261  // used to filter which subfields of the current operation which we should
262  // visit. As an example, the src might be an aggregate wire, but the current
263  // value might be a subfield of that wire. The `src` FieldRef will represent
264  // all subaccesses to the target, but `fieldID` for the current op only needs
265  // to represent the all subaccesses between the current op and the target.
266  struct StackElement {
267  StackElement(FieldRef dst, FieldRef src, Value current, unsigned fieldID)
268  : dst(dst), src(src), current(current), it(current.user_begin()),
269  fieldID(fieldID) {}
270  // The elements of the destination that this refers to.
271  FieldRef dst;
272  // The elements of the source that this refers to.
273  FieldRef src;
274 
275  // These next fields are tied to the value we are currently iterating. This
276  // is used so we can check if a connect op is reading or driving from this
277  // value.
278  Value current;
279  // An iterator of the users of the current value. An end() iterator can be
280  // constructed from the `current` value.
281  Value::user_iterator it;
282  // A filter for which fields of the current value we care about.
283  unsigned fieldID;
284  };
285  SmallVector<StackElement> workStack;
286 
287  // Helper to add record a new wire to be processed in the worklist. This will
288  // add the wire itself to the worklist, which will lead to all subaccesses
289  // being eventually processed as well.
290  auto addToWorklist = [&](FieldRef dst, FieldRef src) {
291  auto value = src.getValue();
292  workStack.emplace_back(dst, src, value, src.getFieldID());
293  };
294 
295  // Create an initial fieldRef from the input value. As a starting state, the
296  // dst and src are the same value.
297  auto original = getFieldRefFromValue(value);
298  auto fieldRef = original;
299 
300  // This loop wraps the worklist, which processes wires. Initially the worklist
301  // is empty.
302  while (true) {
303  // This loop looks through simple operations like casts and nodes. If it
304  // encounters a wire it will stop and add the wire to the worklist.
305  while (true) {
306  auto val = fieldRef.getValue();
307 
308  // The value is a port.
309  if (auto blockArg = val.dyn_cast<BlockArgument>()) {
310  FModuleOp op = cast<FModuleOp>(val.getParentBlock()->getParentOp());
311  auto direction = op.getPortDirection(blockArg.getArgNumber());
312  // Base case: this is one of the module's input ports.
313  if (direction == Direction::In) {
314  if (!callback(original, fieldRef))
315  return false;
316  break;
317  }
318  addToWorklist(original, fieldRef);
319  break;
320  }
321 
322  auto *op = val.getDefiningOp();
323 
324  // The value is an instance port.
325  if (auto inst = dyn_cast<InstanceOp>(op)) {
326  auto resultNo = val.cast<OpResult>().getResultNumber();
327  // Base case: this is an instance's output port.
328  if (inst.getPortDirection(resultNo) == Direction::Out) {
329  if (!callback(original, fieldRef))
330  return false;
331  break;
332  }
333  addToWorklist(original, fieldRef);
334  break;
335  }
336 
337  // If told to look through wires, continue from the driver of the wire.
338  if (lookThroughWires && isa<WireOp>(op)) {
339  addToWorklist(original, fieldRef);
340  break;
341  }
342 
343  // If told to look through nodes, continue from the node input.
344  if (lookThroughNodes && isa<NodeOp>(op)) {
345  auto input = cast<NodeOp>(op).getInput();
346  auto next = getFieldRefFromValue(input);
347  fieldRef = next.getSubField(fieldRef.getFieldID());
348  continue;
349  }
350 
351  // If told to look through casts, continue from the cast input.
352  if (lookThroughCasts &&
353  isa<AsUIntPrimOp, AsSIntPrimOp, AsClockPrimOp, AsAsyncResetPrimOp>(
354  op)) {
355  auto input = op->getOperand(0);
356  auto next = getFieldRefFromValue(input);
357  fieldRef = next.getSubField(fieldRef.getFieldID());
358  continue;
359  }
360 
361  // Look through unary ops generated by emitConnect.
362  if (isa<PadPrimOp, TailPrimOp>(op)) {
363  auto input = op->getOperand(0);
364  auto next = getFieldRefFromValue(input);
365  fieldRef = next.getSubField(fieldRef.getFieldID());
366  continue;
367  }
368 
369  // Base case: this is a constant/invalid or primop.
370  //
371  // TODO: If needed, this could be modified to look through unary ops which
372  // have an unambiguous single driver. This should only be added if a need
373  // arises for it.
374  if (!callback(original, fieldRef))
375  return false;
376  break;
377  }
378 
379  // Process the next element on the stack.
380  while (true) {
381  // If there is nothing left in the workstack, we are done.
382  if (workStack.empty())
383  return true;
384  auto &back = workStack.back();
385  auto current = back.current;
386  // Pop the current element if we have processed all users.
387  if (back.it == current.user_end()) {
388  workStack.pop_back();
389  continue;
390  }
391 
392  original = back.dst;
393  fieldRef = back.src;
394  auto *user = *back.it++;
395  auto fieldID = back.fieldID;
396 
397  if (auto subfield = dyn_cast<SubfieldOp>(user)) {
398  auto bundleType = subfield.getInput().getType();
399  auto index = subfield.getFieldIndex();
400  auto subID = bundleType.getFieldID(index);
401  // If the index of this operation doesn't match the target, skip it.
402  if (fieldID && index != bundleType.getIndexForFieldID(fieldID))
403  continue;
404  auto subRef = fieldRef.getSubField(subID);
405  auto subOriginal = original.getSubField(subID);
406  auto value = subfield.getResult();
407  workStack.emplace_back(subOriginal, subRef, value, fieldID - subID);
408  } else if (auto subindex = dyn_cast<SubindexOp>(user)) {
409  auto vectorType = subindex.getInput().getType();
410  auto index = subindex.getIndex();
411  auto subID = vectorType.getFieldID(index);
412  // If the index of this operation doesn't match the target, skip it.
413  if (fieldID && index != vectorType.getIndexForFieldID(fieldID))
414  continue;
415  auto subRef = fieldRef.getSubField(subID);
416  auto subOriginal = original.getSubField(subID);
417  auto value = subindex.getResult();
418  workStack.emplace_back(subOriginal, subRef, value, fieldID - subID);
419  } else if (auto connect = dyn_cast<FConnectLike>(user)) {
420  // Make sure that this connect is driving the value.
421  if (connect.getDest() != current)
422  continue;
423  // If the value is driven by a connect, we don't have to recurse,
424  // just update the current value.
425  fieldRef = getFieldRefFromValue(connect.getSrc());
426  break;
427  }
428  }
429  }
430 }

436 FieldRef circt::firrtl::getFieldRefFromValue(Value value) {
437  // This code walks upwards from the subfield and calculates the field ID at
438  // each level. At each stage, it must take the current id, and re-index it as
439  // a nested bundle under the parent field.. This is accomplished by using the
440  // parent field's ID as a base, and adding the field ID of the child.
441  unsigned id = 0;
442  while (value) {
443  Operation *op = value.getDefiningOp();
444 
445  // If this is a block argument, we are done.
446  if (!op)
447  break;
448 
449  auto handled = TypeSwitch<Operation *, bool>(op)
450  .Case<SubfieldOp, OpenSubfieldOp>([&](auto subfieldOp) {
451  value = subfieldOp.getInput();
452  auto bundleType = subfieldOp.getInput().getType();
453  // Rebase the current index on the parent field's
454  // index.
455  id +=
456  bundleType.getFieldID(subfieldOp.getFieldIndex());
457  return true;
458  })
459  .Case<SubindexOp, OpenSubindexOp>([&](auto subindexOp) {
460  value = subindexOp.getInput();
461  auto vecType = subindexOp.getInput().getType();
462  // Rebase the current index on the parent field's
463  // index.
464  id += vecType.getFieldID(subindexOp.getIndex());
465  return true;
466  })
467  .Default(false);
468  if (!handled)
469  break;
470  }
471  return {value, id};
472 }

474 /// Get the string name of a value which is a direct child of a declaration op.
475 static void getDeclName(Value value, SmallString<64> &string, bool nameSafe) {
476  // Treat the value as a worklist to allow for recursion.
477  while (value) {
478  if (auto arg = value.dyn_cast<BlockArgument>()) {
479  // Get the module ports and get the name.
480  auto module = cast<FModuleOp>(arg.getOwner()->getParentOp());
481  SmallVector<PortInfo> ports = module.getPorts();
482  string += ports[arg.getArgNumber()].name.getValue();
483  return;
484  }
485 
486  auto *op = value.getDefiningOp();
487  TypeSwitch<Operation *>(op)
488  .Case<InstanceOp, MemOp>([&](auto op) {
489  string += op.getName();
490  string += nameSafe ? "_" : ".";
491  string += op.getPortName(value.cast<OpResult>().getResultNumber())
492  .getValue();
493  value = nullptr;
494  })
495  .Case<FNamableOp>([&](auto op) {
496  string += op.getName();
497  value = nullptr;
498  })
499  .Case<mlir::UnrealizedConversionCastOp>(
500  [&](mlir::UnrealizedConversionCastOp cast) {
501  // Forward through 1:1 conversion cast ops.
502  if (cast.getNumResults() == 1 && cast.getNumOperands() == 1 &&
503  cast.getResult(0).getType() == cast.getOperand(0).getType()) {
504  value = cast.getInputs()[0];
505  } else {
506  // Can't name this.
507  string.clear();
508  value = nullptr;
509  }
510  })
511  .Default([&](auto) {
512  // Can't name this.
513  string.clear();
514  value = nullptr;
515  });
516  }
517 }

519 std::pair<std::string, bool>
520 circt::firrtl::getFieldName(const FieldRef &fieldRef, bool nameSafe) {
521  SmallString<64> name;
522  auto value = fieldRef.getValue();
523  getDeclName(value, name, nameSafe);
524  bool rootKnown = !name.empty();
525 
526  auto type = value.getType();
527  auto localID = fieldRef.getFieldID();
528  while (localID) {
529  // Index directly into ref inner type.
530  if (auto refTy = dyn_cast<RefType>(type))
531  type = refTy.getType();
532 
533  if (auto bundleType = type.dyn_cast<BundleType>()) {
534  auto index = bundleType.getIndexForFieldID(localID);
535  // Add the current field string, and recurse into a subfield.
536  auto &element = bundleType.getElements()[index];
537  if (!name.empty())
538  name += nameSafe ? "_" : ".";
539  name += element.name.getValue();
540  // Recurse in to the element type.
541  type = element.type;
542  localID = localID - bundleType.getFieldID(index);
543  } else if (auto vecType = type.dyn_cast<FVectorType>()) {
544  auto index = vecType.getIndexForFieldID(localID);
545  name += nameSafe ? "_" : "[";
546  name += std::to_string(index);
547  if (!nameSafe)
548  name += "]";
549  // Recurse in to the element type.
550  type = vecType.getElementType();
551  localID = localID - vecType.getFieldID(index);
552  } else {
553  // If we reach here, the field ref is pointing inside some aggregate type
554  // that isn't a bundle or a vector. If the type is a ground type, then the
555  // localID should be 0 at this point, and we should have broken from the
556  // loop.
557  llvm_unreachable("unsupported type");
558  }
559  }
560 
561  return {name.str().str(), rootKnown};
562 }

564 /// This gets the value targeted by a field id. If the field id is targeting
565 /// the value itself, it returns it unchanged. If it is targeting a single field
566 /// in a aggregate value, such as a bundle or vector, this will create the
567 /// necessary subaccesses to get the value.
568 Value circt::firrtl::getValueByFieldID(ImplicitLocOpBuilder builder,
569  Value value, unsigned fieldID) {
570  // When the fieldID hits 0, we've found the target value.
571  while (fieldID != 0) {
572  auto type = value.getType();
573  if (auto bundle = type.dyn_cast<BundleType>()) {
574  auto index = bundle.getIndexForFieldID(fieldID);
575  value = builder.create<SubfieldOp>(value, index);
576  fieldID -= bundle.getFieldID(index);
577  } else {
578  auto vector = type.cast<FVectorType>();
579  auto index = vector.getIndexForFieldID(fieldID);
580  value = builder.create<SubindexOp>(value, index);
581  fieldID -= vector.getFieldID(index);
582  }
583  }
584  return value;
585 }

587 /// Walk leaf ground types in the `firrtlType` and apply the function `fn`.
588 /// The first argument of `fn` is field ID, and the second argument is a
589 /// leaf ground type.
590 void circt::firrtl::walkGroundTypes(
591  FIRRTLType firrtlType,
592  llvm::function_ref<void(uint64_t, FIRRTLBaseType)> fn) {
593  auto type = getBaseType(firrtlType);
594  // If this is a ground type, don't call recursive functions.
595  if (type.isGround())
596  return fn(0, type);
597 
598  uint64_t fieldID = 0;
599  auto recurse = [&](auto &&f, FIRRTLBaseType type) -> void {
600  TypeSwitch<FIRRTLBaseType>(type)
601  .Case<BundleType>([&](BundleType bundle) {
602  for (size_t i = 0, e = bundle.getNumElements(); i < e; ++i) {
603  fieldID++;
604  f(f, bundle.getElementType(i));
605  }
606  })
607  .template Case<FVectorType>([&](FVectorType vector) {
608  for (size_t i = 0, e = vector.getNumElements(); i < e; ++i) {
609  fieldID++;
610  f(f, vector.getElementType());
611  }
612  })
613  .template Case<FEnumType>([&](FEnumType fenum) {
614  for (size_t i = 0, e = fenum.getNumElements(); i < e; ++i) {
615  fieldID++;
616  f(f, fenum.getElementType(i));
617  }
618  })
619  .Default([&](FIRRTLBaseType groundType) {

689 /// Parse a string that may encode a FIRRTL location into a LocationAttr.
690 std::pair<bool, std::optional<mlir::LocationAttr>>
691 circt::firrtl::maybeStringToLocation(StringRef spelling, bool skipParsing,
692  StringAttr &locatorFilenameCache,
693  FileLineColLoc &fileLineColLocCache,
694  MLIRContext *context) {
695  // The spelling of the token looks something like "@[Decoupled.scala 221:8]".
696  if (!spelling.startswith("@[") || !spelling.endswith("]"))
697  return {false, std::nullopt};
698 
699  spelling = spelling.drop_front(2).drop_back(1);
700 
701  // Decode the locator in "spelling", returning the filename and filling in
702  // lineNo and colNo on success. On failure, this returns an empty filename.
703  auto decodeLocator = [&](StringRef input, unsigned &resultLineNo,
704  unsigned &resultColNo) -> StringRef {
705  // Split at the last space.
706  auto spaceLoc = input.find_last_of(' ');
707  if (spaceLoc == StringRef::npos)
708  return {};
709 
710  auto filename = input.take_front(spaceLoc);
711  auto lineAndColumn = input.drop_front(spaceLoc + 1);
712 
713  // Decode the line/column. If the colon is missing, then it will be empty
714  // here.
715  StringRef lineStr, colStr;
716  std::tie(lineStr, colStr) = lineAndColumn.split(':');
717 
718  // Decode the line number and the column number if present.
719  if (lineStr.getAsInteger(10, resultLineNo))
720  return {};
721  if (!colStr.empty()) {
722  if (colStr.front() != '{') {
723  if (colStr.getAsInteger(10, resultColNo))
724  return {};
725  } else {
726  // compound locator, just parse the first part for now
727  if (colStr.drop_front().split(',').first.getAsInteger(10, resultColNo))
728  return {};
729  }
730  }
731  return filename;
732  };
733 
734  // Decode the locator spelling, reporting an error if it is malformed.
735  unsigned lineNo = 0, columnNo = 0;
736  StringRef filename = decodeLocator(spelling, lineNo, columnNo);
737  if (filename.empty())
738  return {false, std::nullopt};
739 
740  // If info locators are ignored, don't actually apply them. We still do all
741  // the verification above though.
742  if (skipParsing)
743  return {true, std::nullopt};
744 
745  /// Return an FileLineColLoc for the specified location, but use a bit of
746  /// caching to reduce thrasing the MLIRContext.
747  auto getFileLineColLoc = [&](StringRef filename, unsigned lineNo,
748  unsigned columnNo) -> FileLineColLoc {
749  // Check our single-entry cache for this filename.
750  StringAttr filenameId = locatorFilenameCache;
751  if (filenameId.str() != filename) {
752  // We missed! Get the right identifier.
753  locatorFilenameCache = filenameId = StringAttr::get(context, filename);
754 
755  // If we miss in the filename cache, we also miss in the FileLineColLoc
756  // cache.
757  return fileLineColLocCache =
758  FileLineColLoc::get(filenameId, lineNo, columnNo);
759  }
760 
761  // If we hit the filename cache, check the FileLineColLoc cache.
762  auto result = fileLineColLocCache;
763  if (result && result.getLine() == lineNo && result.getColumn() == columnNo)
764  return result;
765 
766  return fileLineColLocCache =
767  FileLineColLoc::get(filenameId, lineNo, columnNo);
768  };
769 
770  // Compound locators will be combined with spaces, like:
771  // @[Foo.scala 123:4 Bar.scala 309:14]
772  // and at this point will be parsed as a-long-string-with-two-spaces at
773  // 309:14. We'd like to parse this into two things and represent it as an
774  // MLIR fused locator, but we want to be conservatively safe for filenames
775  // that have a space in it. As such, we are careful to make sure we can
776  // decode the filename/loc of the result. If so, we accumulate results,
777  // backward, in this vector.
778  SmallVector<Location> extraLocs;
779  auto spaceLoc = filename.find_last_of(' ');
780  while (spaceLoc != StringRef::npos) {
781  // Try decoding the thing before the space. Validates that there is another
782  // space and that the file/line can be decoded in that substring.
783  unsigned nextLineNo = 0, nextColumnNo = 0;
784  auto nextFilename =
785  decodeLocator(filename.take_front(spaceLoc), nextLineNo, nextColumnNo);
786 
787  // On failure we didn't have a joined locator.
788  if (nextFilename.empty())
789  break;
790 
791  // On success, remember what we already parsed (Bar.Scala / 309:14), and
792  // move on to the next chunk.
793  auto loc =
794  getFileLineColLoc(filename.drop_front(spaceLoc + 1), lineNo, columnNo);
795  extraLocs.push_back(loc);
796  filename = nextFilename;
797  lineNo = nextLineNo;
798  columnNo = nextColumnNo;
799  spaceLoc = filename.find_last_of(' ');
800  }
801 
802  mlir::LocationAttr result = getFileLineColLoc(filename, lineNo, columnNo);
803  if (!extraLocs.empty()) {
804  extraLocs.push_back(result);
805  std::reverse(extraLocs.begin(), extraLocs.end());
806  result = FusedLoc::get(context, extraLocs);
807  }
808  return {true, result};
809 }

811 /// Given a type, return the corresponding lowered type for the HW dialect.
812 /// Non-FIRRTL types are simply passed through. This returns a null type if it
813 /// cannot be lowered.
814 Type circt::firrtl::lowerType(Type type) {
815  auto firType = type.dyn_cast<FIRRTLBaseType>();
816  if (!firType)
817  return type;
818 
819  // Ignore flip types.
820  firType = firType.getPassiveType();
821 
822  if (auto bundle = firType.dyn_cast<BundleType>()) {
823  mlir::SmallVector<hw::StructType::FieldInfo, 8> hwfields;
824  for (auto element : bundle) {
825  Type etype = lowerType(element.type);
826  if (!etype)
827  return {};
828  hwfields.push_back(hw::StructType::FieldInfo{element.name, etype});
829  }
830  return hw::StructType::get(type.getContext(), hwfields);
831  }
832  if (auto vec = firType.dyn_cast<FVectorType>()) {
833  auto elemTy = lowerType(vec.getElementType());
834  if (!elemTy)
835  return {};
836  return hw::ArrayType::get(elemTy, vec.getNumElements());
837  }
838  if (auto fenum = firType.dyn_cast<FEnumType>()) {
839  mlir::SmallVector<hw::UnionType::FieldInfo, 8> hwfields;
840  SmallVector<Attribute> names;
841  bool simple = true;
842  for (auto element : fenum) {
843  Type etype = lowerType(element.type);
844  if (!etype)
845  return {};
846  hwfields.push_back(hw::UnionType::FieldInfo{element.name, etype, 0});
847  names.push_back(element.name);
848  if (!element.type.isa<UIntType>() ||
849  element.type.getBitWidthOrSentinel() != 0)
850  simple = false;
851  }
852  auto tagTy = hw::EnumType::get(type.getContext(),
853  ArrayAttr::get(type.getContext(), names));
854  if (simple)
855  return tagTy;
856  auto bodyTy = hw::UnionType::get(type.getContext(), hwfields);
857  hw::StructType::FieldInfo fields[2] = {
858  {StringAttr::get(type.getContext(), "tag"), tagTy},
859  {StringAttr::get(type.getContext(), "body"), bodyTy}};
860  return hw::StructType::get(type.getContext(), fields);
861  }
862 
863  auto width = firType.getBitWidthOrSentinel();
864  if (width >= 0) // IntType, analog with known width, clock, etc.
865  return IntegerType::get(type.getContext(), width);
866 
867  return {};
868 }
