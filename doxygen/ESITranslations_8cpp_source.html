71 LogicalResult ExportCosimSchema::visitEndpoint(CosimEndpointOp ep) {
72  auto sendTypeSchema =
73  std::make_shared<capnp::CapnpTypeSchema>(ep.getSend().getType());
74  if (!sendTypeSchema->isSupported())
75  return ep.emitOpError("Type ")
76  << ep.getSend().getType() << " not supported.";
77  types.push_back(sendTypeSchema);
78 
79  auto recvTypeSchema =
80  std::make_shared<capnp::CapnpTypeSchema>(ep.getRecv().getType());
81  if (!recvTypeSchema->isSupported())
82  return ep.emitOpError("Type '")
83  << ep.getRecv().getType() << "' not supported.";
84  types.push_back(recvTypeSchema);
85 
86  os << "# Endpoint ";
87  StringAttr epName = ep->getAttrOfType<StringAttr>("name");
88  if (epName)
89  os << epName << " endpoint at " << ep.getLoc() << ":\n";
90  os << "# Send type: ";
91  sendTypeSchema->writeMetadata(os);
92  os << "\n";
93 
94  os << "# Recv type: ";
95  recvTypeSchema->writeMetadata(os);
96  os << "\n";
97 
98  return success();
99 }

113 LogicalResult ExportCosimSchema::emit() {
114  os << "#########################################################\n"
115  << "## ESI generated schema.\n"
116  << "#########################################################\n";
117 
118  // Walk and collect the type data.
119  auto walkResult = module.walk([this](CosimEndpointOp ep) {
120  if (failed(visitEndpoint(ep)))
121  return mlir::WalkResult::interrupt();
122  return mlir::WalkResult::advance();
123  });
124  if (walkResult.wasInterrupted())
125  return failure();
126  os << "#########################################################\n";
127 
128  // We need a sorted list to ensure determinism.
129  llvm::sort(types.begin(), types.end(),
130  [](auto &a, auto &b) { return a->typeID() > b->typeID(); });
131 
132  // Compute and emit the capnp file id.
133  uint64_t fileHash = 2544816649379317016; // Some random number.
134  for (auto &schema : types)
135  fileHash = llvm::hashing::detail::hash_16_bytes(fileHash, schema->typeID());
136  // Capnp IDs always have a '1' high bit.
137  fileHash |= 0x8000000000000000;
138  capnp::emitCapnpID(os, fileHash) << ";\n\n";
139 
140  os << "#########################################################\n"
141  << "## Types for your design.\n"
142  << "#########################################################\n\n";
143  // Iterate through the various types and emit their schemas.
144  auto end = std::unique(
145  types.begin(), types.end(),
146  [&](const auto &lhs, const auto &rhs) { return *lhs == *rhs; });
147  for (auto typeIter = types.begin(); typeIter != end; ++typeIter) {
148  if (failed((*typeIter)->write(os)))
149  // If we fail during an emission, dump out early since the output may be
150  // corrupted.
151  return failure();
152  }
153 
154  // Include the RPC schema in each generated file.
155  emitCosimSchemaBody(os);
156 
157  return errorCount == 0 ? success() : failure();
158 }
