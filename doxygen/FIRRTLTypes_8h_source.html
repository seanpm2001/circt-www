48 /// A collection of bits indicating the recursive properties of a type.
49 struct RecursiveTypeProperties {
50  /// Whether the type only contains passive elements.
51  bool isPassive : 1;
52  /// Whether the type contains a reference type.
53  bool containsReference : 1;
54  /// Whether the type contains an analog type.
55  bool containsAnalog : 1;
56  /// Whether the type contains a const type.
57  bool containsConst : 1;
58  /// Whether the type has any uninferred bit widths.
59  bool hasUninferredWidth : 1;
60  /// Whether the type has any uninferred reset.
61  bool hasUninferredReset : 1;
62 };

156  static bool classof(Type type) {
157  return llvm::isa<FIRRTLDialect>(type.getDialect()) &&
158  !type.isa<RefType, OpenBundleType, OpenVectorType, StringType>();

161  /// Returns true if this is a non-const "passive" that which is not analog.
162  bool isRegisterType() {
163  return isPassive() && !containsAnalog() && !containsConst();

167  bool isResetType();

173  /// Get the maximum field ID of this type. For integers and other ground
174  /// types, there are no subfields and the maximum field ID is 0. For bundle
175  /// types and vector types, each field is assigned a field ID in a depth-first
176  /// walk order. This function is used to calculate field IDs when this type is

242 /// Trait for types which have a width.
243 /// Users must implement:
244 /// ```c++
245 /// /// Return the width if known, or -1 if unknown.
246 /// int32_t getWidthOrSentinel();
247 /// ```
248 template <typename ConcreteType>
249 class WidthQualifiedTypeTrait
250  : public mlir::TypeTrait::TraitBase<ConcreteType, WidthQualifiedTypeTrait> {
251 public:
252  /// Return an optional containing the width, if the width is known (or empty
253  /// if width is unknown).
254  std::optional<int32_t> getWidth() {
255  auto width = static_cast<ConcreteType *>(this)->getWidthOrSentinel();
256  if (width < 0)
257  return std::nullopt;
258  return width;
259  }
260 
261  /// Return true if this integer type has a known width.
262  bool hasWidth() {
263  return 0 <= static_cast<ConcreteType *>(this)->getWidthOrSentinel();
264  }
265 };
