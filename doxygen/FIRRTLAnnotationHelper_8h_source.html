90  void insertOp(Operation *op) {
91  TypeSwitch<Operation *>(op)
92  .Case<InstanceOp, MemOp, NodeOp, RegOp, RegResetOp, WireOp,
93  chirrtl::CombMemOp, chirrtl::SeqMemOp, chirrtl::MemoryPortOp,
94  chirrtl::MemoryDebugPortOp, PrintFOp>([&](auto op) {
95  // To be safe, check attribute and non-empty name before adding.
96  if (auto name = op.getNameAttr(); name && !name.getValue().empty())
97  targets.insert({name, OpAnnoTarget(op)});
98  });
99  }
100 
101  /// Replace `oldOp` with `newOp` in the target cache. The new and old ops can
102  /// have different names.
103  void replaceOp(Operation *oldOp, Operation *newOp) {
104  if (auto name = oldOp->getAttrOfType<StringAttr>("name");
105  name && !name.getValue().empty())
106  targets.erase(name);
107  insertOp(newOp);
108  }
109 
110  /// Add a new module port to the target cache.
111  void insertPort(FModuleLike mod, size_t portNo) {
112  targets.insert({mod.getPortNameAttr(portNo), PortAnnoTarget(mod, portNo)});
113  }
114 
115 private:
116  /// Walk the module and add named things to 'targets'.
117  void gatherTargets(FModuleLike mod);
118 
119  llvm::DenseMap<StringRef, AnnoTarget> targets;
120 };

122 /// Cache AnnoTargets for a circuit's modules, walked as needed.
123 struct CircuitTargetCache {
124  /// Get cache for specified module, creating it as needed.
125  /// Returned reference may become invalidated by future calls.
126  const AnnoTargetCache &getOrCreateCacheFor(FModuleLike module) {
127  auto it = targetCaches.find(module);
128  if (it == targetCaches.end())
129  it = targetCaches.try_emplace(module, module).first;
130  return it->second;
131  }
132 
133  /// Lookup the target for 'name' in 'module'.
134  AnnoTarget lookup(FModuleLike module, StringRef name) {
135  return getOrCreateCacheFor(module).getTargetForName(name);
136  }
137 
138  /// Clear the cache completely.
139  void invalidate() { targetCaches.clear(); }
140 
141  /// Replace `oldOp` with `newOp` in the target cache. The new and old ops can
142  /// have different names.
143  void replaceOp(Operation *oldOp, Operation *newOp) {
144  auto mod = newOp->getParentOfType<FModuleOp>();
145  auto it = targetCaches.find(mod);
146  if (it == targetCaches.end())
147  return;
148  it->getSecond().replaceOp(oldOp, newOp);
149  }
150 
151  /// Add a new module port to the target cache.
152  void insertPort(FModuleLike mod, size_t portNo) {
153  auto it = targetCaches.find(mod);
154  if (it == targetCaches.end())
155  return;
156  it->getSecond().insertPort(mod, portNo);
157  }
158 
159  /// Add a new op to the target cache.
160  void insertOp(Operation *op) {
161  auto mod = op->getParentOfType<FModuleOp>();
162  auto it = targetCaches.find(mod);
163  if (it == targetCaches.end())
164  return;
165  it->getSecond().insertOp(op);
166  }
167 
168 private:
169  DenseMap<Operation *, AnnoTargetCache> targetCaches;
170 };

304 /// Implements the same behavior as DictionaryAttr::getAs<A> to return the
305 /// value of a specific type associated with a key in a dictionary. However,
306 /// this is specialized to print a useful error message, specific to custom
307 /// annotation process, on failure.
308 template <typename A>
309 A tryGetAs(DictionaryAttr &dict, const Attribute &root, StringRef key,
310  Location loc, Twine className, Twine path = Twine()) {
311  // Check that the key exists.
312  auto value = dict.get(key);
313  if (!value) {
314  SmallString<128> msg;
315  if (path.isTriviallyEmpty())
316  msg = ("Annotation '" + className + "' did not contain required key '" +
317  key + "'.")
318  .str();
319  else
320  msg = ("Annotation '" + className + "' with path '" + path +
321  "' did not contain required key '" + key + "'.")
322  .str();
323  mlir::emitError(loc, msg).attachNote()
324  << "The full Annotation is reproduced here: " << root << "\n";
325  return nullptr;
326  }
327  // Check that the value has the correct type.
328  auto valueA = value.dyn_cast_or_null<A>();
329  if (!valueA) {
330  SmallString<128> msg;
331  if (path.isTriviallyEmpty())
332  msg = ("Annotation '" + className +
333  "' did not contain the correct type for key '" + key + "'.")
334  .str();
335  else
336  msg = ("Annotation '" + className + "' with path '" + path +
337  "' did not contain the correct type for key '" + key + "'.")
338  .str();
339  mlir::emitError(loc, msg).attachNote()
340  << "The full Annotation is reproduced here: " << root << "\n";
341  return nullptr;
342  }
343  return valueA;
344 }
