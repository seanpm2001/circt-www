31 LogicalResult
32 ServiceGeneratorDispatcher::generate(ServiceImplementReqOp req,
33  ServiceDeclOpInterface decl) {
34  // Lookup based on 'impl_type' attribute and pass through the generate request
35  // if found.
36  auto genF = genLookupTable.find(req.getImplTypeAttr().getValue());
37  if (genF == genLookupTable.end()) {
38  if (failIfNotFound)
39  return req.emitOpError("Could not find service generator for attribute '")
40  << req.getImplTypeAttr() << "'";
41  return success();
42  }
43  return genF->second(req, decl);
44 }

46 /// The generator for the "cosim" impl_type.
47 static LogicalResult instantiateCosimEndpointOps(ServiceImplementReqOp req,
48  ServiceDeclOpInterface) {
49  auto *ctxt = req.getContext();
50  OpBuilder b(req);
51  Value clk = req.getOperand(0);
52  Value rst = req.getOperand(1);
53 
54  // Determine which EndpointID this generator should start with.
55  if (req.getImplOpts()) {
56  auto opts = req.getImplOpts()->getValue();
57  for (auto nameAttr : opts) {
58  return req.emitOpError("did not recognize option name ")
59  << nameAttr.getName();
60  }
61  }
62 
63  // Assemble the name to use for an endpoint.
64  auto toStringAttr = [&](ArrayAttr strArr) {
65  std::string buff;
66  llvm::raw_string_ostream os(buff);
67  llvm::interleave(strArr.getAsValueRange<StringAttr>(), os, ".");
68  return StringAttr::get(ctxt, os.str());
69  };
70 
71  llvm::DenseMap<RequestToClientConnectionOp, unsigned> toClientResultNum;
72  for (auto toClient : req.getOps<RequestToClientConnectionOp>())
73  toClientResultNum[toClient] = toClientResultNum.size();
74 
75  // Get the request pairs.
76  llvm::SmallVector<
77  std::pair<RequestToServerConnectionOp, RequestToClientConnectionOp>, 8>
78  reqPairs;
79  req.gatherPairedReqs(reqPairs);
80 
81  // Iterate through them, building a cosim endpoint for each one.
82  for (auto [toServer, toClient] : reqPairs) {
83  assert((toServer || toClient) &&
84  "At least one in all pairs must be non-null");
85  Location loc = toServer ? toServer.getLoc() : toClient.getLoc();
86  ArrayAttr clientNamePathAttr = toServer ? toServer.getClientNamePathAttr()
87  : toClient.getClientNamePathAttr();
88 
89  Value toServerValue;
90  if (toServer)
91  toServerValue = toServer.getToServer();
92  else
93  toServerValue =
94  b.create<NullSourceOp>(loc, ChannelType::get(ctxt, b.getI1Type()));
95 
96  Type toClientType;
97  if (toClient)
98  toClientType = toClient.getToClient().getType();
99  else
100  toClientType = ChannelType::get(ctxt, b.getI1Type());
101 
102  auto cosim =
103  b.create<CosimEndpointOp>(loc, toClientType, clk, rst, toServerValue,
104  toStringAttr(clientNamePathAttr));
105 
106  if (toClient) {
107  unsigned clientReqIdx = toClientResultNum[toClient];
108  req.getResult(clientReqIdx).replaceAllUsesWith(cosim.getRecv());
109  }
110  }
111 
112  // Erase the generation request.
113  req.erase();
114  return success();
115 }

117 // Generator for "sv_mem" implementation type. Emits SV ops for an unpacked
118 // array, hopefully inferred as a memory to the SV compiler.
119 static LogicalResult
120 instantiateSystemVerilogMemory(ServiceImplementReqOp req,
121  ServiceDeclOpInterface decl) {
122  if (!decl)
123  return req.emitOpError(
124  "Must specify a service declaration to use 'sv_mem'.");
125 
126  ImplicitLocOpBuilder b(req.getLoc(), req);
127  BackedgeBuilder bb(b, req.getLoc());
128 
129  RandomAccessMemoryDeclOp ramDecl =
130  dyn_cast<RandomAccessMemoryDeclOp>(decl.getOperation());
131  if (!ramDecl)
132  return req.emitOpError("'sv_mem' implementation type can only be used to "
133  "implement RandomAccessMemory declarations");
134 
135  if (req.getNumOperands() != 2)
136  return req.emitOpError("Implementation requires clk and rst operands");
137  auto clk = req.getOperand(0);
138  auto rst = req.getOperand(1);
139  auto write = b.getStringAttr("write");
140  auto read = b.getStringAttr("read");
141  auto none = b.create<hw::ConstantOp>(
142  APInt(/*numBits*/ 0, /*val*/ 0, /*isSigned*/ false));
143  auto i1 = b.getI1Type();
144  auto c0 = b.create<hw::ConstantOp>(i1, 0);
145 
146  // Assemble a mapping of toClient results to actual consumers.
147  DenseMap<Value, Value> outputMap;
148  for (auto [bout, reqout] : llvm::zip_longest(
149  req.getOps<RequestToClientConnectionOp>(), req.getResults())) {
150  assert(bout.has_value());
151  assert(reqout.has_value());
152  outputMap[*bout] = *reqout;
153  }
154 
155  // Create the SV memory.
156  hw::UnpackedArrayType memType =
157  hw::UnpackedArrayType::get(ramDecl.getInnerType(), ramDecl.getDepth());
158  auto mem = b.create<sv::RegOp>(memType, req.getServiceSymbolAttr().getAttr())
159  .getResult();
160 
161  // Get the request pairs.
162  llvm::SmallVector<
163  std::pair<RequestToServerConnectionOp, RequestToClientConnectionOp>, 8>
164  reqPairs;
165  req.gatherPairedReqs(reqPairs);
166 
167  // Do everything which doesn't actually write to the memory, store the signals
168  // needed for the actual memory writes for later.
169  SmallVector<std::tuple<Value, Value, Value>> writeGoAddressData;
170  for (auto [toServerReq, toClientReq] : reqPairs) {
171  assert(toServerReq && toClientReq); // All of our interfaces are inout.
172  assert(toServerReq.getServicePort() == toClientReq.getServicePort());
173  auto port = toServerReq.getServicePort().getName();
174  WrapValidReadyOp toClientResp;
175 
176  if (port == write) {
177  // If this pair is doing a write...
178 
179  // Construct the response channel.
180  auto doneValid = bb.get(i1);
181  toClientResp = b.create<WrapValidReadyOp>(none, doneValid);
182 
183  // Unwrap the write request and 'explode' the struct.
184  auto unwrap = b.create<UnwrapValidReadyOp>(toServerReq.getToServer(),
185  toClientResp.getReady());
186 
187  Value address = b.create<hw::StructExtractOp>(unwrap.getRawOutput(),
188  b.getStringAttr("address"));
189  Value data = b.create<hw::StructExtractOp>(unwrap.getRawOutput(),
190  b.getStringAttr("data"));
191 
192  // Determine if the write should occur this cycle.
193  auto go = b.create<comb::AndOp>(unwrap.getValid(), unwrap.getReady());
194  go->setAttr("sv.namehint", b.getStringAttr("write_go"));
195  // Register the 'go' signal and use it as the done message.
196  doneValid.setValue(
197  b.create<seq::CompRegOp>(go, clk, rst, c0, "write_done"));
198  // Store the necessary data for the 'always' memory writing block.
199  writeGoAddressData.push_back(std::make_tuple(go, address, data));
200 
201  } else if (port == read) {
202  // If it's a read...
203 
204  // Construct the response channel.
205  auto dataValid = bb.get(i1);
206  auto data = bb.get(ramDecl.getInnerType());
207  toClientResp = b.create<WrapValidReadyOp>(data, dataValid);
208 
209  // Unwrap the requested address and read from that memory location.
210  auto addressUnwrap = b.create<UnwrapValidReadyOp>(
211  toServerReq.getToServer(), toClientResp.getReady());
212  Value memLoc =
213  b.create<sv::ArrayIndexInOutOp>(mem, addressUnwrap.getRawOutput());
214  auto readData = b.create<sv::ReadInOutOp>(memLoc);
215 
216  // Set the data on the response.
217  data.setValue(readData);
218  dataValid.setValue(addressUnwrap.getValid());
219  } else {
220  assert(false && "Port should be either 'read' or 'write'");
221  }
222 
223  outputMap[toClientReq.getToClient()].replaceAllUsesWith(
224  toClientResp.getChanOutput());
225  }
226 
227  // Now construct the memory writes.
228  b.create<sv::AlwaysFFOp>(
229  sv::EventControl::AtPosEdge, clk, ResetType::SyncReset,
230  sv::EventControl::AtPosEdge, rst, [&] {
231  for (auto [go, address, data] : writeGoAddressData) {
232  Value a = address, d = data; // So the lambda can capture.
233  // If we're told to go, do the write.
234  b.create<sv::IfOp>(go, [&] {
235  Value memLoc = b.create<sv::ArrayIndexInOutOp>(mem, a);
236  b.create<sv::PAssignOp>(memLoc, d);
237  });
238  }
239  });
240 
241  req.erase();
242  return success();
243 }

264 /// Implements a pass to connect up ESI services clients to the nearest server
265 /// instantiation. Wires up the ports and generates a generation request to
266 /// call a user-specified generator.
267 struct ESIConnectServicesPass
268  : public ESIConnectServicesBase<ESIConnectServicesPass>,
269  msft::PassCommon {
270 
271  ESIConnectServicesPass(const ServiceGeneratorDispatcher &gen)
272  : genDispatcher(gen) {}
273  ESIConnectServicesPass()
274  : genDispatcher(ServiceGeneratorDispatcher::globalDispatcher()) {}
275 
276  void runOnOperation() override;
277 
278  /// "Bubble up" the specified requests to all of the instantiations of the
279  /// module specified. Create and connect up ports to tunnel the ESI channels
280  /// through.
281  LogicalResult surfaceReqs(hw::HWMutableModuleLike,
282  ArrayRef<RequestToClientConnectionOp>,
283  ArrayRef<RequestToServerConnectionOp>);
284 
285  /// Copy all service metadata up the instance hierarchy. Modify the service
286  /// name path while copying.
287  void copyMetadata(hw::HWModuleLike);
288 
289  /// For any service which is "local" (provides the requested service) in a
290  /// module, replace it with a ServiceImplementOp. Said op is to be replaced
291  /// with an instantiation by a generator.
292  LogicalResult replaceInst(ServiceInstanceOp, Block *portReqs);
293 
294  /// Figure out which requests are "local" vs need to be surfaced. Call
295  /// 'surfaceReqs' and/or 'replaceInst' as appropriate.
296  LogicalResult process(hw::HWModuleLike);
297 
298 private:
299  ServiceGeneratorDispatcher genDispatcher;
300 };
301 } // anonymous namespace

303 void ESIConnectServicesPass::runOnOperation() {
304  ModuleOp outerMod = getOperation();
305  topLevelSyms.addDefinitions(outerMod);
306  if (failed(verifyInstances(outerMod))) {
307  signalPassFailure();
308  return;
309  }
310 
311  // Get a partially-ordered list of modules based on the instantiation DAG.
312  // It's _very_ important that we process modules before their instantiations
313  // so that the modules where they're instantiated correctly process the
314  // surfaced connections.
315  SmallVector<hw::HWModuleLike, 64> sortedMods;
316  getAndSortModules(outerMod, sortedMods);
317 
318  // Process each module.
319  for (auto mod : sortedMods) {
320  hw::HWModuleLike mutableMod = dyn_cast<hw::HWModuleLike>(*mod);
321  if (mutableMod && failed(process(mutableMod))) {
322  signalPassFailure();
323  return;
324  }
325  }
326 }

328 LogicalResult ESIConnectServicesPass::process(hw::HWModuleLike mod) {
329  Block &modBlock = mod->getRegion(0).front();
330 
331  // Index the local services and create blocks in which to put the requests.
332  DenseMap<SymbolRefAttr, Block *> localImplReqs;
333  Block *anyServiceInst = nullptr;
334  for (auto instOp : modBlock.getOps<ServiceInstanceOp>()) {
335  auto *b = new Block();
336  localImplReqs[instOp.getServiceSymbolAttr()] = b;
337  if (!instOp.getServiceSymbol().has_value())
338  anyServiceInst = b;
339  }
340 
341  // Decompose the 'inout' requests int to 'in' and 'out' requests.
342  mod.walk([&](RequestInOutChannelOp reqInOut) {
343  ImplicitLocOpBuilder b(reqInOut.getLoc(), reqInOut);
344  b.create<RequestToServerConnectionOp>(reqInOut.getServicePortAttr(),
345  reqInOut.getToServer(),
346  reqInOut.getClientNamePathAttr());
347  auto toClientReq = b.create<RequestToClientConnectionOp>(
348  reqInOut.getToClient().getType(), reqInOut.getServicePortAttr(),
349  reqInOut.getClientNamePathAttr());
350  reqInOut.getToClient().replaceAllUsesWith(toClientReq.getToClient());
351  reqInOut.erase();
352  });
353 
354  // Find all of the "local" requests.
355  mod.walk([&](Operation *op) {
356  if (auto req = dyn_cast<RequestToClientConnectionOp>(op)) {
357  auto service = req.getServicePortAttr().getModuleRef();
358  auto implOpF = localImplReqs.find(service);
359  if (implOpF != localImplReqs.end())
360  req->moveBefore(implOpF->second, implOpF->second->end());
361  else if (anyServiceInst)
362  req->moveBefore(anyServiceInst, anyServiceInst->end());
363  } else if (auto req = dyn_cast<RequestToServerConnectionOp>(op)) {
364  auto service = req.getServicePortAttr().getModuleRef();
365  auto implOpF = localImplReqs.find(service);
366  if (implOpF != localImplReqs.end())
367  req->moveBefore(implOpF->second, implOpF->second->end());
368  else if (anyServiceInst)
369  req->moveBefore(anyServiceInst, anyServiceInst->end());
370  }
371  });
372 
373  // Replace each service instance with a generation request. If a service
374  // generator is registered, generate the server.
375  for (auto instOp :
376  llvm::make_early_inc_range(modBlock.getOps<ServiceInstanceOp>())) {
377  Block *portReqs = localImplReqs[instOp.getServiceSymbolAttr()];
378  if (failed(replaceInst(instOp, portReqs)))
379  return failure();
380  }
381 
382  // Copy any metadata up the instance hierarchy.
383  copyMetadata(mod);
384 
385  // Identify the non-local reqs which need to be surfaced from this module.
386  SmallVector<RequestToClientConnectionOp, 4> nonLocalToClientReqs;
387  SmallVector<RequestToServerConnectionOp, 4> nonLocalToServerReqs;
388  mod.walk([&](Operation *op) {
389  if (auto req = dyn_cast<RequestToClientConnectionOp>(op)) {
390  auto service = req.getServicePortAttr().getModuleRef();
391  auto implOpF = localImplReqs.find(service);
392  if (implOpF == localImplReqs.end())
393  nonLocalToClientReqs.push_back(req);
394  } else if (auto req = dyn_cast<RequestToServerConnectionOp>(op)) {
395  auto service = req.getServicePortAttr().getModuleRef();
396  auto implOpF = localImplReqs.find(service);
397  if (implOpF == localImplReqs.end())
398  nonLocalToServerReqs.push_back(req);
399  }
400  });
401 
402  // Surface all of the requests which cannot be fulfilled locally.
403  if (nonLocalToClientReqs.empty() && nonLocalToServerReqs.empty())
404  return success();
405 
406  if (auto mutableMod = dyn_cast<hw::HWMutableModuleLike>(mod.getOperation()))
407  return surfaceReqs(mutableMod, nonLocalToClientReqs, nonLocalToServerReqs);
408  return mod.emitOpError(
409  "Cannot surface requests through module without mutable ports");
410 }

412 void ESIConnectServicesPass::copyMetadata(hw::HWModuleLike mod) {
413  SmallVector<ServiceHierarchyMetadataOp, 8> metadataOps;
414  mod.walk([&](ServiceHierarchyMetadataOp op) { metadataOps.push_back(op); });
415 
416  for (auto inst : moduleInstantiations[mod]) {
417  OpBuilder b(inst);
418  auto instName = inst.getInstanceNameAttr();
419  for (auto metadata : metadataOps) {
420  SmallVector<Attribute, 4> path;
421  path.push_back(hw::InnerRefAttr::get(mod.getModuleNameAttr(), instName));
422  for (auto attr : metadata.getServerNamePathAttr())
423  path.push_back(attr);
424 
425  auto metadataCopy = cast<ServiceHierarchyMetadataOp>(b.clone(*metadata));
426  metadataCopy.setServerNamePathAttr(b.getArrayAttr(path));
427  }
428  }

431 /// Create an op which contains metadata about the soon-to-be implemented
432 /// service. To be used by later passes which require these data (e.g.
433 /// automated software API creation).
434 static void emitServiceMetadata(ServiceImplementReqOp implReqOp) {
435  ImplicitLocOpBuilder b(implReqOp.getLoc(), implReqOp);
436 
437  // Check if there are any "BSP" service providers -- ones which implement any
438  // service -- and create an implicit service declaration for them.
439  std::unique_ptr<Block> bspPorts = nullptr;
440  if (!implReqOp.getServiceSymbol().has_value()) {
441  bspPorts = std::make_unique<Block>();
442  b.setInsertionPointToStart(bspPorts.get());
443  }
444 
445  llvm::SmallVector<
446  std::pair<RequestToServerConnectionOp, RequestToClientConnectionOp>, 8>
447  reqPairs;
448  implReqOp.gatherPairedReqs(reqPairs);
449 
450  SmallVector<Attribute, 8> clients;
451  for (auto [toServer, toClient] : reqPairs) {
452  SmallVector<NamedAttribute, 4> clientAttrs;
453  Attribute servicePort, clientNamePath;
454  if (toServer) {
455  clientNamePath = toServer.getClientNamePathAttr();
456  servicePort = toServer.getServicePortAttr();
457  clientAttrs.push_back(b.getNamedAttr(
458  "to_server_type", TypeAttr::get(toServer.getToServer().getType())));
459  }
460  if (toClient) {
461  clientNamePath = toClient.getClientNamePathAttr();
462  servicePort = toClient.getServicePortAttr();
463  clientAttrs.push_back(b.getNamedAttr(
464  "to_client_type", TypeAttr::get(toClient.getToClient().getType())));
465  }
466 
467  clientAttrs.push_back(b.getNamedAttr("port", servicePort));
468  clientAttrs.push_back(b.getNamedAttr("client_name", clientNamePath));
469 
470  clients.push_back(b.getDictionaryAttr(clientAttrs));
471 
472  if (!bspPorts)
473  continue;
474 
475  if (toServer && toClient)
476  b.create<ServiceDeclInOutOp>(
477  toServer.getServicePort().getName(),
478  TypeAttr::get(toServer.getToServer().getType()),
479  TypeAttr::get(toClient.getToClient().getType()));
480  else if (toClient)
481  b.create<ToClientOp>(toClient.getServicePort().getName(),
482  TypeAttr::get(toClient.getToClient().getType()));
483  else
484  b.create<ToServerOp>(toServer.getServicePort().getName(),
485  TypeAttr::get(toServer.getToServer().getType()));
486  }
487 
488  if (bspPorts && !bspPorts->empty()) {
489  b.setInsertionPointToEnd(
490  implReqOp->getParentOfType<mlir::ModuleOp>().getBody());
491  // TODO: we currently only support one BSP. Should we support more?
492  auto decl = b.create<CustomServiceDeclOp>("BSP");
493  decl.getPorts().push_back(bspPorts.release());
494  implReqOp.setServiceSymbol(decl.getSymNameAttr().getValue());
495  }
496 
497  auto clientsAttr = b.getArrayAttr(clients);
498  auto nameAttr = b.getArrayAttr(ArrayRef<Attribute>{});
499  b.setInsertionPointAfter(implReqOp);
500  b.create<ServiceHierarchyMetadataOp>(
501  implReqOp.getServiceSymbolAttr(), nameAttr, implReqOp.getImplTypeAttr(),
502  implReqOp.getImplOptsAttr(), clientsAttr);
503 }

505 LogicalResult ESIConnectServicesPass::replaceInst(ServiceInstanceOp instOp,
506  Block *portReqs) {
507  assert(portReqs);
508  auto declSym = instOp.getServiceSymbolAttr();
509  ServiceDeclOpInterface decl;
510  if (declSym) {
511  decl = dyn_cast_or_null<ServiceDeclOpInterface>(
512  topLevelSyms.getDefinition(declSym));
513  if (!decl)
514  return instOp.emitOpError("Could not find service declaration ")
515  << declSym;
516  }
517 
518  // Compute the result types for the new op -- the instance op's output types
519  // + the to_client types.
520  SmallVector<Type, 8> resultTypes(instOp.getResultTypes().begin(),
521  instOp.getResultTypes().end());
522  for (auto toClient : portReqs->getOps<RequestToClientConnectionOp>())
523  resultTypes.push_back(toClient.getToClient().getType());
524 
525  // Create the generation request op.
526  OpBuilder b(instOp);
527  auto implOp = b.create<ServiceImplementReqOp>(
528  instOp.getLoc(), resultTypes, instOp.getServiceSymbolAttr(),
529  instOp.getImplTypeAttr(), instOp.getImplOptsAttr(), instOp.getOperands());
530  implOp->setDialectAttrs(instOp->getDialectAttrs());
531  implOp.getPortReqs().push_back(portReqs);
532 
533  // Update the users.
534  for (auto [n, o] : llvm::zip(implOp.getResults(), instOp.getResults()))
535  o.replaceAllUsesWith(n);
536  unsigned instOpNumResults = instOp.getNumResults();
537  for (auto e :
538  llvm::enumerate(portReqs->getOps<RequestToClientConnectionOp>()))
539  e.value().getToClient().replaceAllUsesWith(
540  implOp.getResult(e.index() + instOpNumResults));
541 
542  emitServiceMetadata(implOp);
543 
544  // Try to generate the service provider.
545  if (failed(genDispatcher.generate(implOp, decl)))
546  return instOp.emitOpError("failed to generate server");
547 
548  instOp.erase();
549  return success();
550 }

552 LogicalResult ESIConnectServicesPass::surfaceReqs(
553  hw::HWMutableModuleLike mod,
554  ArrayRef<RequestToClientConnectionOp> toClientReqs,
555  ArrayRef<RequestToServerConnectionOp> toServerReqs) {
556  auto *ctxt = mod.getContext();
557  Block *body = &mod->getRegion(0).front();
558 
559  // Track initial operand/result counts and the new IO.
560  unsigned origNumInputs = mod.getNumInputs();
561  SmallVector<std::pair<unsigned, hw::PortInfo>> newInputs;
562  unsigned origNumOutputs = mod.getNumOutputs();
563  SmallVector<std::pair<mlir::StringAttr, Value>> newOutputs;
564 
565  // Assemble a port name from an array.
566  auto getPortName = [&](ArrayAttr namePath) {
567  std::string portName;
568  llvm::raw_string_ostream nameOS(portName);
569  llvm::interleave(
570  namePath.getValue(), nameOS,
571  [&](Attribute attr) { nameOS << attr.cast<StringAttr>().getValue(); },
572  ".");
573  return StringAttr::get(ctxt, nameOS.str());
574  };
575 
576  // Insert new module input ESI ports.
577  for (auto toClient : toClientReqs) {
578  newInputs.push_back(std::make_pair(
579  origNumInputs,
580  hw::PortInfo{getPortName(toClient.getClientNamePath()),
581  hw::PortDirection::INPUT, toClient.getType(),
582  origNumInputs, nullptr, toClient.getLoc()}));
583  body->addArgument(toClient.getType(), toClient.getLoc());
584  }
585  mod.insertPorts(newInputs, {});
586 
587  // Replace uses with new block args which will correspond to said ports.
588  // Note: no zip or enumerate here because we need mutable access to
589  // toClientReqs.
590  int i = 0;
591  for (auto toClient : toClientReqs) {
592  toClient.replaceAllUsesWith(body->getArguments()[origNumInputs + i]);
593  ++i;
594  }
595 
596  // Append output ports to new port list and redirect toServer inputs to
597  // output op.
598  unsigned outputCounter = origNumOutputs;
599  for (auto toServer : toServerReqs)

611  };

616  StringAttr resultsAttrName = StringAttr::get(ctxt, "resultNames");

623  inst->getOperands().end());

625  inst->getResultTypes().end());
626

628  // into the module containing `inst`.

631  instToClient.setClientNamePathAttr(prependNamePart(

633  newOperands.push_back(instToClient.getToClient());
634  }

637  for (auto newPort : newOutputs)

641  SmallVector<NamedAttribute> newAttrs;

643  if (attr.getName() == argsAttrName)

645  else if (attr.getName() == resultsAttrName)
646  newAttrs.push_back(

648  else

650  }

652  inst->getLoc(), inst->getName(), newResultTypes, newOperands,
653  b.getDictionaryAttr(newAttrs), inst->getPropertiesStorage(),
654  inst->getSuccessors(), inst->getRegions()));

656 
657  // Replace all uses of the instance being replaced.
658  for (auto [newV, oldV] :
659  llvm::zip(newHWInst->getResults(), inst->getResults()))
660  oldV.replaceAllUsesWith(newV);
661 
662  // Clone the to_server requests and wire them up to the new instance.
663  outputCounter = origNumOutputs;
664  for (auto [toServer, newPort] : llvm::zip(toServerReqs, newOutputs)) {
665  auto instToServer = cast<RequestToServerConnectionOp>(b.clone(*toServer));
666  instToServer.setClientNamePathAttr(prependNamePart(
667  instToServer.getClientNamePath(), inst.getInstanceName()));
668  instToServer->setOperand(0, newHWInst->getResult(outputCounter++));
669  }
670  }
671 
672  // Replace the list of instantiations and erase the old ones.
673  moduleInstantiations[mod].swap(newModuleInstantiations);
674  for (auto oldInst : newModuleInstantiations)
675  oldInst->erase();
676 
677  // Erase the original requests since they have been cloned into the proper
678  // destination modules.
679  for (auto toClient : toClientReqs)
680  toClient.erase();
681  for (auto toServer : toServerReqs)
682  toServer.erase();
683  return success();
