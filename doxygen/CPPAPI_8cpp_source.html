58 static void emitCPPType(Type type, mlir::raw_indented_ostream &os) {
59  llvm::TypeSwitch<Type>(type)
60  .Case([&os](IntegerType intTy) {
61  auto w = intTy.getWidth();
62  if (w == 0) {
63  os << "void";
64  } else if (w == 1) {
65  os << "bool";
66  } else {
67  if (intTy.isSigned())
68  os << "int";
69  else
70  os << "uint";
71 
72  // Round up.
73  if (w <= 8)
74  os << "8";
75  else if (w <= 16)
76  os << "16";
77  else if (w <= 32)
78  os << "32";
79  else if (w <= 64)
80  os << "64";
81  else
82  assert(false && "Type not supported. Integer too wide. Please "
83  "check support first with isSupported()");
84 
85  os << "_t";
86  }
87  })
88  .Case<hw::ArrayType, hw::StructType>([](auto) {
89  assert(false && "Structs containing List and Structs not supported");
90  })
91  .Default([](Type) {
92  assert(false && "Type not supported. Please check support first with "
93  "isSupported()");
94  });
95 }

102 LogicalResult CPPType::write(mlir::raw_indented_ostream &os) const {
103 
104  os << "struct " << name() << " {\n";
105  os.indent();
106 
107  os << "// Data members:\n";
108  auto fieldTypes = getFields();
109  for (auto field : fieldTypes) {
110 
111  // Do not emit zero-width fields.
112  if (isZeroWidthInt(field.type))
113  os << "// Zero-width field: ";
114 
115  emitCPPType(field.type, os);
116  os << " " << field.name.getValue() << ";";
117  // Specify the mlir type.
118  os << "\t// MLIR type is " << field.type << "\n";
119  }
120  os << "\n";
121 
122  bool isI0 = false;
123  if (fieldTypes.size() == 1) {
124  os << "// Unary types have convenience constructors\n";
125  os << name() << "() = default;\n";
126 
127  auto fieldType = fieldTypes[0].type;
128  if (!isZeroWidthInt(fieldType)) {
129 
130  os << name() << "(";
131  auto fieldName = fieldTypes[0].name.getValue();
132  emitCPPType(fieldType, os);
133  os << " " << fieldName << ") : " << fieldName << "(" << fieldName << ")"
134  << " {}\n\n";
135 
136  os << "// Unary types have convenience conversion operators\n";
137  os << "operator ";
138  emitCPPType(fieldType, os);
139  os << "() const { return " << fieldName << "; }\n\n";
140  } else {
141  isI0 = true;
142  }
143  }
144 
145  // Comparison operator
146  os << "// Equality operator\n";
147  os << "// TODO (c++20): emit default <=> operator\n";
148  os << "auto operator==(const " << name() << " &other) const {\n";
149  os.indent();
150  os << "return ";
151  llvm::interleave(
152  fieldTypes, os,
153  [&](auto field) {
154  os << "(";
155  if (isZeroWidthInt(field.type))
156  os << "/*i0==*/true";
157  else
158  os << field.name.getValue() << " == other." << field.name.getValue();
159  os << ")";
160  },
161  " && ");
162  os << ";\n";
163  os.unindent();
164  os << "}\n\n";
165 
166  // != operator
167  os << "// Inequality operator\n";
168  os << "auto operator!=(const " << name() << " &other) const {\n";
169  os.indent();
170  os << "return !(*this == other);\n";
171  os.unindent();
172  os << "}\n\n";
173 
174  // Stream operator
175  os << "// Stream operator\n";
176  os << "friend std::ostream &operator<<(std::ostream &os, const " << name()
177  << " &val) {\n";
178  os.indent();
179  os << "os << \"" << name() << "(\";\n";
180  for (auto [idx, field] : llvm::enumerate(fieldTypes)) {
181  if (isZeroWidthInt(field.type))
182  continue;
183  os << "os << \"" << field.name.getValue() << ": \";\n";
184 
185  // A bit of a hack - (u)int8_t types will by default be printed as chars.
186  // We want to avoid this and just print the underlying value.
187  os << "os << ";
188  if (field.type.getIntOrFloatBitWidth() > 1 &&
189  field.type.getIntOrFloatBitWidth() <= 8)
190  os << "(uint32_t)";
191  os << "val." << field.name.getValue() << ";\n";
192  if (idx != fieldTypes.size() - 1)
193  os << "os << \", \";\n";
194  }
195  os << "os << \")\";\n";
196  os << "return os;\n";
197  os.unindent();
198  os << "}\n\n";
199 
200  // Capnproto type (todo: remove)
201  os << "// Cap'nProto type which this ESI type maps to\n";
202  os << "using CPType = ::";
203  if (isI0)
204  os << "UntypedData";
205  else
206  os << name();
207 
208  os << ";\n";
209 
210  os.unindent();
211  os << "};\n\n";
212 
213  return success();
214 }

250 LogicalResult CPPEndpoint::writeType(Location loc,
251  mlir::raw_indented_ostream &os) const {
252  auto emitType = [&](llvm::StringRef dir, mlir::Type type) -> LogicalResult {
253  type = esi::innerType(type);
254  auto cppTypeIt = types.find(type);
255  if (cppTypeIt == types.end()) {
256 
257  emitError(loc) << "Type " << type
258  << " not found in set of CPP API types!";
259  return failure();
260  }
261 
262  os << "/*" << dir << "Type=*/ ESITypes::" << cppTypeIt->second.name();
263  return success();
264  };
265 
266  if (portInfo.toClientType && portInfo.toServerType) {
267  os << "esi::runtime::ReadWritePort<";
268  if (failed(emitType("read", portInfo.toClientType)))
269  return failure();
270  os << ", ";
271  if (failed(emitType("write", portInfo.toServerType)))
272  return failure();
273  } else if (portInfo.toServerType) {
274  os << "esi::runtime::ReadPort<";
275  if (failed(emitType("read", portInfo.toServerType)))
276  return failure();
277  } else if (portInfo.toClientType) {
278  os << "esi::runtime::WritePort<";
279  if (failed(emitType("write", portInfo.toClientType)))
280  return failure();
281  } else {
282  llvm_unreachable("Port has no type");
283  }
284  os << ", TBackend>";
285  return success();
286 }

300  auto loc = service.getLoc();
301  os << "template <typename TBackend>\n";
302  os << "class " << name() << " : public esi::runtime::Module<TBackend> {\n";
303  os.indent();

305  << "public:\n";

307 
308  for (auto &ep : endpoints)

310  return failure();

315  os << ep->getPointerTypeName() << " " << ep->getName();
316  });
317  os << ")\n";
318  os << "// Initialize base class with knowledge of all endpoints.\n";
319  os << ": esi::runtime::Module<TBackend>({";
320  llvm::interleaveComma(endpoints, os, [&](auto &ep) { os << ep->getName(); });
321  os << "}),\n";
322 
323  os << "// Initialize each individual endpoint handle.\n";
324  llvm::interleaveComma(endpoints, os, [&](auto &ep) {
325  os << ep->getName() << "(" << ep->getName() << ")";
326  });
327  os << " {}\n\n";
328 
329  os << "// Endpoint handles\n";
330  for (auto &ep : endpoints)
331  os << ep->getPointerTypeName() << " " << ep->getName() << ";\n";
332 
333  os.unindent();
334  os << "};\n\n";

363 LogicalResult CPPDesignModule::write(mlir::raw_indented_ostream &ios) {
364  ios << "template <typename TBackend>\n";
365  ios << "class " << name() << " {\n";
366  ios << "public:\n";
367  ios.indent();
368 
369  // Locate the inner services.
370  struct ServiceInfo {
371  std::string cppMemberName;
372  ArrayAttr clients;
373  };
374  llvm::MapVector<CPPService *, ServiceInfo> innerServices;
375  for (auto &service : services) {
376  // lookup service in CPPServices
377  auto serviceSym = *service.getServiceSymbol();
378  auto *cppServiceIt = llvm::find_if(cppServices, [&](auto &cppService) {
379  auto sym = SymbolTable::getSymbolName(cppService.getService());
380  return sym == serviceSym;
381  });
382  if (cppServiceIt == cppServices.end())
383  return mod.emitOpError()
384  << "Service " << serviceSym << " not found in CPPServices.";
385 
386  ServiceInfo info;
387  info.cppMemberName = lowercase(cppServiceIt->name().str());
388  info.clients = service.getClients();
389  ios << "std::unique_ptr<" << cppServiceIt->name() << "<TBackend>> "
390  << info.cppMemberName << ";\n";
391  innerServices[cppServiceIt] = info;
392  }
393 
394  // Add constructor
395  constexpr static std::string_view backendName = "backend";
396  ios << name() << "(TBackend& " << backendName << ") {\n";
397  ios.indent();
398  for (auto &[innerCppService, serviceInfo] : innerServices) {
399  ios << "// " << innerCppService->name() << " initialization.\n";
400 
401  struct ClientPortName {
402  std::string servicePortName;
403  std::string clientPortName;
404  };
405  llvm::SmallVector<ClientPortName> portNames;
406 
407  for (auto [i, client] : llvm::enumerate(serviceInfo.clients)) {
408  // cast client to a dictionary attribute
409  auto clientDict = client.cast<DictionaryAttr>();
410  auto clientName = clientDict.getAs<ArrayAttr>("client_name");
411  auto clientNameRange = clientName.getAsValueRange<StringAttr>();
412  auto port = clientDict.getAs<hw::InnerRefAttr>("port");
413 
414  auto portName = joinStringAttrArray(clientName, "_");
415  auto *ep = innerCppService->getPort(port.getName());
416  ios << "auto " << portName << " = std::make_shared<";
417  portNames.push_back(ClientPortName{port.getName().str(), portName});
418  if (failed(ep->writeType(mod.getLoc(), ios)))
419  return failure();
420  ios << ">(std::vector<std::string>{";
421  llvm::interleaveComma(clientNameRange, ios, [&](auto clientHierName) {
422  ios << "\"" << clientHierName.str() << "\"";
423  });
424  ios << "}, " << backendName << ", \"cosim\");\n";
425  }
426 
427  // And instantiate the member using the instantiated ports. Make sure that
428  // they are in the same order as the endpoints in the service to match the
429  // order of the ports in the constructor.
430  auto servicePorts = innerCppService->getPorts();
431  llvm::sort(portNames, [&](auto &a, auto &b) {
432  auto aIt = llvm::find_if(servicePorts, [&](auto &port) {
433  return port.name == a.servicePortName;
434  });
435  auto bIt = llvm::find_if(servicePorts, [&](auto &port) {
436  return port.name == b.servicePortName;
437  });
438  auto aIdx = std::distance(servicePorts.begin(), aIt);
439  auto bIdx = std::distance(servicePorts.begin(), bIt);
440  return aIdx < bIdx;
441  });
442 
443  ios << serviceInfo.cppMemberName << " = std::make_unique<"
444  << innerCppService->name() << "<TBackend>>(";
445  llvm::interleaveComma(portNames, ios, [&](auto &portName) {
446  ios << portName.clientPortName;
447  });
448  ios << ");\n";
449  // and initialize...
450  ios << serviceInfo.cppMemberName << "->init();\n";
451  }
452 
453  ios.unindent();
454  ios << "}\n";
455  ios.unindent();
456  ios << "};\n";
457  return success();
458 }
