30 // Mapping from unique identifier to python callback. We use std::string
31 // pointers since we also need to allocate memory for the string.
32 llvm::DenseMap<std::string *, PyObject *> serviceGenFuncLookup;
33 static MlirLogicalResult serviceGenFunc(MlirOperation reqOp,
34  MlirOperation declOp, void *userData) {
35  std::string *name = static_cast<std::string *>(userData);
36  py::handle genFunc(serviceGenFuncLookup[name]);
37  py::gil_scoped_acquire();
38  py::object rc = genFunc(reqOp);
39  return rc.cast<bool>() ? mlirLogicalResultSuccess()
40  : mlirLogicalResultFailure();
41 }
