114 template <class ConcreteOp>
115 struct AllocStateLikeOpLowering : public OpConversionPattern<ConcreteOp> {
116  using OpConversionPattern<ConcreteOp>::OpConversionPattern;
117  using OpConversionPattern<ConcreteOp>::typeConverter;
118  using OpAdaptor = typename ConcreteOp::Adaptor;
119 
120  LogicalResult
121  matchAndRewrite(ConcreteOp op, OpAdaptor adaptor,
122  ConversionPatternRewriter &rewriter) const final {
123  // Get a pointer to the correct offset in the storage.
124  auto offsetAttr = op->template getAttrOfType<IntegerAttr>("offset");
125  if (!offsetAttr)
126  return failure();
127  Value ptr = rewriter.create<LLVM::GEPOp>(
128  op->getLoc(), adaptor.getStorage().getType(), adaptor.getStorage(),
129  LLVM::GEPArg(offsetAttr.getValue().getZExtValue()));
130 
131  // Cast the raw storage pointer to a pointer of the state's actual type.
132  auto type = typeConverter->convertType(op.getType());
133  if (type != ptr.getType())
134  ptr = rewriter.create<LLVM::BitcastOp>(op->getLoc(), type, ptr);
135 
136  rewriter.replaceOp(op, ptr);
137  return success();
138  }
139 };

215 static MemoryAccess prepareMemoryAccess(Location loc, Value memory,
216  Value address, MemoryType type,
217  ConversionPatternRewriter &rewriter) {
218  auto zextAddrType = rewriter.getIntegerType(
219  address.getType().cast<IntegerType>().getWidth() + 1);
220  Value addr = rewriter.create<LLVM::ZExtOp>(loc, zextAddrType, address);
221  Value addrLimit = rewriter.create<LLVM::ConstantOp>(
222  loc, zextAddrType, rewriter.getI32IntegerAttr(type.getNumWords()));
223  Value withinBounds = rewriter.create<LLVM::ICmpOp>(
224  loc, LLVM::ICmpPredicate::ult, addr, addrLimit);
225  auto ptrType = LLVM::LLVMPointerType::get(type.getWordType());
226  Value ptr =
227  rewriter.create<LLVM::GEPOp>(loc, ptrType, memory, ValueRange{addr});
228  return {ptr, withinBounds};
229 }

231 struct MemoryReadOpLowering : public OpConversionPattern<arc::MemoryReadOp> {
232  using OpConversionPattern::OpConversionPattern;
233  LogicalResult
234  matchAndRewrite(arc::MemoryReadOp op, OpAdaptor adaptor,
235  ConversionPatternRewriter &rewriter) const final {
236  auto type = typeConverter->convertType(op.getType());
237  auto access = prepareMemoryAccess(
238  op.getLoc(), adaptor.getMemory(), adaptor.getAddress(),
239  op.getMemory().getType().cast<MemoryType>(), rewriter);
240 
241  // Only attempt to read the memory if the address is within bounds,
242  // otherwise produce a zero value.
243  rewriter.replaceOpWithNewOp<scf::IfOp>(
244  op, access.withinBounds,
245  [&](auto &builder, auto loc) {
246  Value loadOp = builder.template create<LLVM::LoadOp>(loc, access.ptr);
247  builder.template create<scf::YieldOp>(loc, loadOp);
248  },
249  [&](auto &builder, auto loc) {
250  Value zeroValue = builder.template create<LLVM::ConstantOp>(
251  loc, type, builder.getI64IntegerAttr(0));
252  builder.template create<scf::YieldOp>(loc, zeroValue);
253  });
254  return success();
255  }
256 };

258 struct MemoryWriteOpLowering : public OpConversionPattern<arc::MemoryWriteOp> {
259  using OpConversionPattern::OpConversionPattern;
260  LogicalResult
261  matchAndRewrite(arc::MemoryWriteOp op, OpAdaptor adaptor,
262  ConversionPatternRewriter &rewriter) const final {
263  auto access = prepareMemoryAccess(
264  op.getLoc(), adaptor.getMemory(), adaptor.getAddress(),
265  op.getMemory().getType().cast<MemoryType>(), rewriter);
266  auto enable = access.withinBounds;
267  if (adaptor.getEnable())
268  enable = rewriter.create<LLVM::AndOp>(op.getLoc(), adaptor.getEnable(),
269  enable);
270 
271  // Only attempt to write the memory if the address is within bounds.
272  rewriter.replaceOpWithNewOp<scf::IfOp>(
273  op, enable, [&](auto &builder, auto loc) {
274  builder.template create<LLVM::StoreOp>(loc, adaptor.getData(),
275  access.ptr);
276  builder.template create<scf::YieldOp>(loc);
277  });
278  return success();
279  }

362 static void populateLegality(ConversionTarget &target) {
363  target.addLegalDialect<mlir::BuiltinDialect>();
364  target.addLegalDialect<hw::HWDialect>();
365  target.addLegalDialect<comb::CombDialect>();
366  target.addLegalDialect<func::FuncDialect>();
367  target.addLegalDialect<scf::SCFDialect>();
368  target.addLegalDialect<LLVM::LLVMDialect>();
369 
370  target.addIllegalOp<arc::DefineOp>();
371  target.addIllegalOp<arc::OutputOp>();
372  target.addIllegalOp<arc::StateOp>();
373  target.addIllegalOp<arc::ClockTreeOp>();
374  target.addIllegalOp<arc::PassThroughOp>();
375 
376  target.addDynamicallyLegalOp<func::FuncOp>([](func::FuncOp op) {
377  auto argsConverted = llvm::none_of(op.getBlocks(), [](auto &block) {
378  return hasArcType(block.getArguments());
379  });
380  auto resultsConverted = !hasArcType(op.getResultTypes());
381  return argsConverted && resultsConverted;
382  });
383  addGenericLegality<func::ReturnOp>(target);
384  addGenericLegality<func::CallOp>(target);
385 }

387 static void populateTypeConversion(TypeConverter &typeConverter) {
388  typeConverter.addConversion([&](StorageType type) {
389  return LLVM::LLVMPointerType::get(IntegerType::get(type.getContext(), 8));
390  });
391  typeConverter.addConversion([&](MemoryType type) {
392  return LLVM::LLVMPointerType::get(
393  IntegerType::get(type.getContext(), type.getStride() * 8));
394  });
395  typeConverter.addConversion([&](StateType type) {
396  return LLVM::LLVMPointerType::get(
397  typeConverter.convertType(type.getType()));
398  });
399  typeConverter.addConversion([](hw::ArrayType type) { return type; });
400  typeConverter.addConversion([](mlir::IntegerType type) { return type; });
401 }

445 void LowerArcToLLVMPass::runOnOperation() {
446  // Remove the models since we only care about the clock functions at this
447  // point.
448  // NOTE: In the future we may want to have an earlier pass lower the model
449  // into a separate `*_eval` function that checks for rising edges on clocks
450  // and then calls the appropriate function. At that point we won't have to
451  // delete models here anymore.
452  for (auto op : llvm::make_early_inc_range(getOperation().getOps<ModelOp>()))
453  op.erase();
454 
455  if (failed(lowerToMLIR()))
456  return signalPassFailure();
457 
458  if (failed(lowerArcToLLVM()))
459  return signalPassFailure();
460 }

462 /// Perform the lowering to Func and SCF.
463 LogicalResult LowerArcToLLVMPass::lowerToMLIR() {
464  LLVM_DEBUG(llvm::dbgs() << "Lowering arcs to Func/SCF dialects\n");
465  ConversionTarget target(getContext());
466  TypeConverter converter;
467  RewritePatternSet patterns(&getContext());

474 /// Perform lowering to LLVM.
475 LogicalResult LowerArcToLLVMPass::lowerArcToLLVM() {
476  LLVM_DEBUG(llvm::dbgs() << "Lowering to LLVM dialect\n");
477 
478  Namespace globals;
479  SymbolCache cache;
480  cache.addDefinitions(getOperation());
481  globals.add(cache);
482 
483  LLVMConversionTarget target(getContext());
484  LLVMTypeConverter converter(&getContext());
485  RewritePatternSet patterns(&getContext());
486  target.addLegalOp<mlir::ModuleOp>();
487  target.addIllegalOp<arc::ModelOp>();
488  populateSCFToControlFlowConversionPatterns(patterns);
489  populateFuncToLLVMConversionPatterns(converter, patterns);
490  cf::populateControlFlowToLLVMConversionPatterns(converter, patterns);
491 
492  DenseMap<std::pair<Type, ArrayAttr>, LLVM::GlobalOp> constAggregateGlobalsMap;
493  populateHWToLLVMConversionPatterns(converter, patterns, globals,
494  constAggregateGlobalsMap);
495  populateHWToLLVMTypeConversions(converter);
496  populateCombToLLVMConversionPatterns(converter, patterns);
497  arith::populateArithToLLVMConversionPatterns(converter, patterns);
498 
499  return applyFullConversion(getOperation(), target, std::move(patterns));
500 }
