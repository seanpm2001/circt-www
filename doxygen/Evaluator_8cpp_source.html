29 /// Instantiate an Object with its class name and actual parameters.
30 FailureOr<std::shared_ptr<Object>>
31 circt::om::Evaluator::instantiate(StringAttr className,
32  ArrayRef<ObjectValue> actualParams) {
33  ClassOp cls = symbolTable.lookup<ClassOp>(className);
34  if (!cls)
35  return symbolTable.getOp()->emitError("unknown class name ") << className;
36 
37  auto formalParamNames = cls.getFormalParamNames().getAsRange<StringAttr>();
38  auto formalParamTypes = cls.getBodyBlock()->getArgumentTypes();
39 
40  // Verify the actual parameters are the right size and types for this class.
41  if (actualParams.size() != formalParamTypes.size()) {
42  auto error = cls.emitError("actual parameter list length (")
43  << actualParams.size() << ") does not match formal "
44  << "parameter list length (" << formalParamTypes.size() << ")";
45  error.attachNote() << "actual parameters: " << actualParams;
46  error.attachNote(cls.getLoc()) << "formal parameters: " << formalParamTypes;
47  return error;
48  }
49 
50  // Verify the actual parameter types match.
51  for (auto [actualParam, formalParamName, formalParamType] :
52  llvm::zip(actualParams, formalParamNames, formalParamTypes)) {
53  Type actualParamType;
54  if (auto *attr = std::get_if<Attribute>(&actualParam))
55  if (auto typedActualParam = attr->dyn_cast_or_null<TypedAttr>())
56  actualParamType = typedActualParam.getType();
57  if (auto *object = std::get_if<std::shared_ptr<Object>>(&actualParam))
58  actualParamType = object->get()->getType();
59 
60  if (!actualParamType)
61  return cls.emitError("actual parameter for ")
62  << formalParamName << " is null";
63 
64  if (actualParamType != formalParamType) {
65  auto error = cls.emitError("actual parameter for ")
66  << formalParamName << " has invalid type";
67  error.attachNote() << "actual parameter: " << actualParam;
68  error.attachNote() << "format parameter type: " << formalParamType;
69  return error;
70  }
71  }
72 
73  // Instantiate the fields.
74  ObjectFields fields;
75  for (auto field : cls.getOps<ClassFieldOp>()) {
76  StringAttr name = field.getSymNameAttr();
77  Value value = field.getValue();
78 
79  FailureOr<ObjectValue> result = evaluateValue(value, actualParams);
80  if (failed(result))
81  return failure();
82 
83  fields[name] = result.value();
84  }
85 
86  // Allocate the Object. Further refinement is expected.
87  auto *object = new Object(cls, fields);
88 
89  return success(std::shared_ptr<Object>(object));
90 }

92 /// Evaluate a Value in a Class body according to the semantics of the IR. The
93 /// actual parameters are the values supplied at the current instantiation of
94 /// the Class being evaluated.
95 FailureOr<ObjectValue>
96 circt::om::Evaluator::evaluateValue(Value value,
97  ArrayRef<ObjectValue> actualParams) {
98  return TypeSwitch<Value, FailureOr<ObjectValue>>(value)
99  .Case([&](BlockArgument arg) {
100  return evaluateParameter(arg, actualParams);
101  })
102  .Case([&](OpResult result) {
103  return TypeSwitch<Operation *, FailureOr<ObjectValue>>(
104  result.getDefiningOp())
105  .Case([&](ConstantOp op) {
106  return evaluateConstant(op, actualParams);
107  })
108  .Case([&](ObjectOp op) {
109  return evaluateObjectInstance(op, actualParams);
110  })
111  .Case([&](ObjectFieldOp op) {
112  return evaluateObjectField(op, actualParams);
113  })
114  .Default([&](Operation *op) {
115  auto error = op->emitError("unable to evaluate value");
116  error.attachNote() << "value: " << value;
117  return error;
118  });
119  });
120 }

136 /// Evaluator dispatch function for Object instances.
137 FailureOr<ObjectValue> circt::om::Evaluator::evaluateObjectInstance(
138  ObjectOp op, ArrayRef<ObjectValue> actualParams) {
139  // First, check if we have already evaluated this object, and return it if so.
140  auto existingInstance = objects.find(op);
141  if (existingInstance != objects.end())
142  return success(existingInstance->second);
143 
144  // If we need to instantiate a new object, evaluate values for all of its
145  // actual parameters. Note that this is eager evaluation, which precludes
146  // creating cycles in the object model. Further refinement is expected.
147  SmallVector<ObjectValue> objectParams;
148  for (auto param : op.getActualParams()) {
149  FailureOr<ObjectValue> result = evaluateValue(param, actualParams);
150  if (failed(result))
151  return result;
152  objectParams.push_back(result.value());
153  }
154 
155  // Instantiate and return the new Object, saving the instance for later.
156  auto newInstance = instantiate(op.getClassNameAttr(), objectParams);
157  if (succeeded(newInstance))
158  objects[op.getResult()] = newInstance.value();
159  return newInstance;
160 }

162 /// Evaluator dispatch function for Object fields.
163 FailureOr<ObjectValue>
164 circt::om::Evaluator::evaluateObjectField(ObjectFieldOp op,
165  ArrayRef<ObjectValue> actualParams) {
166  // Evaluate the Object itself, in case it hasn't been evaluated yet.
167  FailureOr<ObjectValue> currentObjectResult =
168  evaluateValue(op.getObject(), actualParams);
169  if (failed(currentObjectResult))
170  return currentObjectResult;
171 
172  std::shared_ptr<Object> currentObject =
173  std::get<std::shared_ptr<Object>>(currentObjectResult.value());
174 
175  // Iteratively access nested fields through the path until we reach the final
176  // field in the path.
177  ObjectValue finalField;
178  for (auto field : op.getFieldPath().getAsRange<FlatSymbolRefAttr>()) {
179  auto currentField = currentObject->getField(field.getAttr());
180  finalField = currentField.value();
181  if (auto *nextObject = std::get_if<std::shared_ptr<Object>>(&finalField))
182  currentObject = *nextObject;
183  }
184 
185  // Return the field being accessed.
186  return finalField;
187 }
