31 /// Buffer tokens for clients that need to adjust things.
32 struct BufferingPP {
33  using BufferVec = SmallVectorImpl<Token>;
34  BufferVec &tokens;
35  bool hasEOF = false;
36 
37  BufferingPP(BufferVec &tokens) : tokens(tokens) {}
38 
39  void add(Token t) {
40  assert(!hasEOF);
41  tokens.push_back(t);
42  }
43 
44  /// Add a range of tokens.
45  template <typename R>
46  void addTokens(R &&newTokens) {
47  assert(!hasEOF);
48  llvm::append_range(tokens, newTokens);
49  }
50 
51  /// Buffer a final EOF, no tokens allowed after this.
52  void eof() {
53  assert(!hasEOF);
54  hasEOF = true;
55  }
56 
57  /// Flush buffered tokens to the specified pretty printer.
58  /// Emit the EOF is one was added.
59  void flush(PrettyPrinter &pp) {
60  pp.addTokens(tokens);
61  tokens.clear();
62  if (hasEOF) {
63  pp.eof();
64  hasEOF = false;
65  }
66  }
67 };

77 /// Add convenience methods for generating pretty-printing tokens.
78 template <typename PPTy = PrettyPrinter>

87  /// Add new token.
88  template <typename T, typename... Args>
89  typename std::enable_if_t<std::is_base_of_v<Token, T>> add(Args &&...args) {
90  pp.add(T(std::forward<Args>(args)...));
91  }

93 
94  /// End of a stream.
95  void eof() { pp.eof(); }

99  /// Add a literal (with external storage).
100  void literal(StringRef str) { add<StringToken>(str); }
101 
102  /// Add a non-breaking space.
103  void nbsp() { literal(" "); }
104 
105  /// Add multiple non-breaking spaces as a single token.
106  void nbsp(unsigned n) {
107  detail::emitNBSP(n, [&](Token t) { addToken(t); });
108  }

110  //===- Breaks -----------------------------------------------------------===//
111 
112  /// Add a 'neverbreak' break. Always 'fits'.
113  void neverbreak() { add<BreakToken>(0, 0, true); }
114 
115  /// Add a newline (break too wide to fit, always breaks).
116  void newline() { add<BreakToken>(PrettyPrinter::kInfinity); }
117 
118  /// Add breakable spaces.
119  void spaces(uint32_t n) { add<BreakToken>(n); }
120 
121  /// Add a breakable space.
122  void space() { spaces(1); }
123 
124  /// Add a break that is zero-wide if not broken.
125  void zerobreak() { add<BreakToken>(0); }

127  //===- Groups -----------------------------------------------------------===//
128 
129  /// Start a IndentStyle::Block group with specified offset.
130  void bbox(int32_t offset = 0, Breaks breaks = Breaks::Consistent) {
131  add<BeginToken>(offset, breaks, IndentStyle::Block);
132  }
133 
134  /// Start a consistent group with specified offset.
135  void cbox(int32_t offset = 0, IndentStyle style = IndentStyle::Visual) {
136  add<BeginToken>(offset, Breaks::Consistent, style);
137  }
138 
139  /// Start an inconsistent group with specified offset.
140  void ibox(int32_t offset = 0, IndentStyle style = IndentStyle::Visual) {
141  add<BeginToken>(offset, Breaks::Inconsistent, style);
142  }
143 
144  /// Start a group that cannot break, including nested groups.
145  /// Use sparingly.
146  void neverbox() { add<BeginToken>(0, Breaks::Never); }
147 
148  /// End a group.
149  void end() { add<EndToken>(); }

203 /// Wrap a PrettyPrinter with TokenBuilder features as well as operator<<'s.
204 /// String behavior:
205 /// Strings streamed as `const char *` are assumed to have external storage,
206 /// and StringRef's are saved until no longer needed.
207 /// Use PPExtString() and PPSaveString() wrappers to specify/override behavior.
208 template <typename PPTy = PrettyPrinter>
209 class TokenStream : public TokenBuilder<PPTy> {
210  using Base = TokenBuilder<PPTy>;
211  TokenStringSaver &saver;
212 
213 public:
214  /// Create a TokenStream using the specified PrettyPrinter and StringSaver
215  /// storage. Strings are saved in `saver`, which is generally the listener in
216  /// the PrettyPrinter, but may not be (e.g., using BufferingPP).
217  TokenStream(PPTy &pp, TokenStringSaver &saver) : Base(pp), saver(saver) {}
218 
219  /// Add a string literal (external storage).
220  TokenStream &operator<<(const char *s) {
221  Base::literal(s);
222  return *this;
223  }
224  /// Add a string token (saved to storage).
225  TokenStream &operator<<(StringRef s) {
226  Base::template add<StringToken>(saver.save(s));
227  return *this;
228  }
229 
230  /// String has external storage.
231  TokenStream &operator<<(const PPExtString &str) {
232  Base::literal(str.str);
233  return *this;
234  }
235 
236  /// String must be saved.
237  TokenStream &operator<<(const PPSaveString &str) {
238  Base::template add<StringToken>(saver.save(str.str));
239  return *this;
240  }
241 
242  /// Convenience for inline streaming of builder methods.
243  TokenStream &operator<<(PP s) {
244  switch (s) {
245  case PP::bbox2:
246  Base::bbox(2);
247  break;
248  case PP::cbox0:
249  Base::cbox(0);
250  break;
251  case PP::cbox2:
252  Base::cbox(2);
253  break;
254  case PP::end:
255  Base::end();
256  break;
257  case PP::eof:
258  Base::eof();
259  break;
260  case PP::ibox0:
261  Base::ibox(0);
262  break;
263  case PP::ibox2:
264  Base::ibox(2);
265  break;
266  case PP::nbsp:
267  Base::nbsp();
268  break;
269  case PP::neverbox:
270  Base::neverbox();
271  break;
272  case PP::neverbreak:
273  Base::neverbreak();
274  break;
275  case PP::newline:
276  Base::newline();
277  break;
278  case PP::space:
279  Base::space();
280  break;
281  case PP::zerobreak:
282  Base::zerobreak();
283  break;
284  }
285  return *this;
286  }

294  /// General-purpose "format this" helper, for types not supported by
295  /// operator<< yet.
296  template <typename T>
297  TokenStream &addAsString(T &&t) {
298  invokeWithStringOS([&](auto &os) { os << std::forward<T>(t); });
299  return *this;
300  }
301 
302  /// Helper to invoke code with a llvm::raw_ostream argument for compatibility.
303  /// All data is gathered into a single string token.
304  template <typename Callable, unsigned BufferLen = 128>
305  auto invokeWithStringOS(Callable &&c) {
306  SmallString<BufferLen> ss;
307  llvm::raw_svector_ostream ssos(ss);
308  auto flush = llvm::make_scope_exit([&]() {
309  if (!ss.empty())
310  *this << ss;
311  });
312  return std::invoke(std::forward<Callable>(c), ssos);
313  }
314 
315  /// Write escaped versions of the string, saved in storage.
316  TokenStream &writeEscaped(StringRef str, bool useHexEscapes = false) {
317  return writeQuotedEscaped(str, useHexEscapes, "", "");
318  }
319  TokenStream &writeQuotedEscaped(StringRef str, bool useHexEscapes = false,
320  StringRef left = "\"",
321  StringRef right = "\"") {
322  // Add as a single StringToken.
323  invokeWithStringOS([&](auto &os) {
324  os << left;
325  os.write_escaped(str, useHexEscapes);
326  os << right;
327  });
328  return *this;
329  }
330 
331  /// Open a box, invoke the lambda, and close it after.
332  template <typename T, typename Callable>
333  auto scopedBox(T &&t, Callable &&c, Token close = EndToken()) {
334  *this << std::forward<T>(t);
335  auto done = llvm::make_scope_exit([&]() { *this << close; });
336  return std::invoke(std::forward<Callable>(c));
337  }
338 };
