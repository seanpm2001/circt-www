122 /// Base class for the signaling standard of a particular port. Abstracts the
123 /// details of a particular signaling standard from the port layout. Subclasses
124 /// keep around port mapping information to use when updating instances.
125 class SignalingStandard {
126 public:
127  SignalingStandard(ChannelRewriter &rewriter, PortInfo origPort)
128  : rewriter(rewriter), body(rewriter.getBody()), origPort(origPort) {}
129  virtual ~SignalingStandard() = default;
130 
131  // Lower the specified (possibly high-level ESI) port into a wire-level
132  // signaling protocol. The two virtual methods 'build*Signals' should be
133  // overridden by subclasses. They should use the 'create*' methods in
134  // 'ChannelRewriter' to create the necessary ports.
135  void lowerPort() {
136  if (origPort.direction == PortDirection::OUTPUT)
137  buildOutputSignals();
138  else
139  buildInputSignals();
140  }
141 
142  /// Update an instance port to the new port information. Also adds the proper
143  /// ESI ops to map the channel to the wire signaling standard. These get
144  /// lowered away in a later pass.
145  virtual void mapInputSignals(OpBuilder &b, Operation *inst, Value instValue,
146  SmallVectorImpl<Value> &newOperands,
147  ArrayRef<Backedge> newResults) = 0;
148  virtual void mapOutputSignals(OpBuilder &b, Operation *inst, Value instValue,
149  SmallVectorImpl<Value> &newOperands,
150  ArrayRef<Backedge> newResults) = 0;
151 
152 protected:
153  virtual void buildInputSignals() = 0;
154  virtual void buildOutputSignals() = 0;
155 
156  ChannelRewriter &rewriter;
157  Block *body;
158  PortInfo origPort;
159 
160  hw::HWMutableModuleLike getModule() { return rewriter.getModule(); }
161  MLIRContext *getContext() { return getModule()->getContext(); }
162 
163  SmallVector<PortInfo> dataPorts;
164 
165  /// Data ports aren't (generally) treated differently by different signaling
166  /// standards. Since we want to support "flattened" structs as well as not
167  /// (which is orthogonal to the control signals), put all of the data port
168  /// transform logic in these general methods.
169  Value buildInputDataPorts();
170  void buildOutputDataPorts(Value data);
171  void mapInputDataPorts(OpBuilder &b, Value unwrappedData,
172  SmallVectorImpl<Value> &newOperands);
173  Value mapOutputDataPorts(OpBuilder &b, ArrayRef<Backedge> newResults);

176 // Build the input data ports. If it's a channel of a struct (and we've been
177 // asked to flatten it), flatten it into a port for each wire. If there's a
178 // body, re-construct the struct.
179 Value SignalingStandard::buildInputDataPorts() {
180  auto chanTy = origPort.type.dyn_cast<ChannelType>();
181  Type dataPortType = chanTy ? chanTy.getInner() : origPort.type;
182  hw::StructType origStruct =
183  chanTy ? dataPortType.dyn_cast<hw::StructType>() : hw::StructType();
184 
185  if (!rewriter.shouldFlattenStructs() || !origStruct) {
186  dataPorts.push_back({});
187  return rewriter.createNewInput(origPort, "", dataPortType, dataPorts[0]);
188  }
189 
190  ArrayRef<hw::StructType::FieldInfo> elements = origStruct.getElements();
191  dataPorts.append(elements.size(), {});
192  SmallVector<Value, 16> elementValues;
193  for (auto [idx, element] : llvm::enumerate(elements))
194  elementValues.push_back(rewriter.createNewInput(
195  origPort, "_" + element.name.getValue(), element.type, dataPorts[idx]));
196  if (!body)
197  return {};
198  return OpBuilder::atBlockBegin(body).create<hw::StructCreateOp>(
199  origPort.loc, origStruct, elementValues);
200 }

202 // Map a data value into the new operands for an instance. If the original type
203 // was a channel of a struct (and we've been asked to flatten it), break it up
204 // into per-field values.
205 void SignalingStandard::mapInputDataPorts(OpBuilder &b, Value unwrappedData,
206  SmallVectorImpl<Value> &newOperands) {
207  auto chanTy = origPort.type.dyn_cast<ChannelType>();
208  Type dataPortType = chanTy ? chanTy.getInner() : origPort.type;
209  hw::StructType origStruct =
210  chanTy ? dataPortType.dyn_cast<hw::StructType>() : hw::StructType();
211 
212  if (!rewriter.shouldFlattenStructs() || !origStruct) {
213  newOperands[dataPorts[0].argNum] = unwrappedData;
214  } else {
215  auto explode = b.create<hw::StructExplodeOp>(origPort.loc, unwrappedData);
216  assert(explode->getNumResults() == dataPorts.size());
217  for (auto [dataPort, fieldValue] :
218  llvm::zip(dataPorts, explode.getResults()))
219  newOperands[dataPort.argNum] = fieldValue;
220  }
221 }

223 // Build the data ports for outputs. If the original type was a channel of a
224 // struct (and we've been asked to flatten it), explode the struct to create
225 // individual ports.
226 void SignalingStandard::buildOutputDataPorts(Value data) {
227  auto chanTy = origPort.type.dyn_cast<ChannelType>();
228  Type dataPortType = chanTy ? chanTy.getInner() : origPort.type;
229  hw::StructType origStruct =
230  chanTy ? dataPortType.dyn_cast<hw::StructType>() : hw::StructType();
231 
232  if (!rewriter.shouldFlattenStructs() || !origStruct) {
233  dataPorts.push_back({});
234  rewriter.createNewOutput(origPort, "", dataPortType, data, dataPorts[0]);
235  } else {
236  ArrayRef<hw::StructType::FieldInfo> elements = origStruct.getElements();
237  dataPorts.append(elements.size(), {});
238 
239  Operation *explode = nullptr;
240  if (body)
241  explode = OpBuilder::atBlockTerminator(body).create<hw::StructExplodeOp>(
242  origPort.loc, data);
243 
244  for (size_t idx = 0, e = elements.size(); idx < e; ++idx) {
245  auto field = elements[idx];
246  Value fieldValue = explode ? explode->getResult(idx) : Value();
247  rewriter.createNewOutput(origPort, "_" + field.name.getValue(),
248  field.type, fieldValue, dataPorts[idx]);
249  }
250  }
251 }

253 // Map the data ports coming off an instance back into the original ports. If
254 // the original type was a channel of a struct (and we've been asked to flatten
255 // it), construct the original struct from the new ports.
256 Value SignalingStandard::mapOutputDataPorts(OpBuilder &b,
257  ArrayRef<Backedge> newResults) {
258  auto chanTy = origPort.type.dyn_cast<ChannelType>();
259  Type dataPortType = chanTy ? chanTy.getInner() : origPort.type;
260  hw::StructType origStruct =
261  chanTy ? dataPortType.dyn_cast<hw::StructType>() : hw::StructType();
262 
263  if (!rewriter.shouldFlattenStructs() || !origStruct)
264  return newResults[dataPorts[0].argNum];
265 
266  SmallVector<Value, 16> fieldValues;
267  for (auto portInfo : dataPorts)
268  fieldValues.push_back(newResults[portInfo.argNum]);
269  return b.create<hw::StructCreateOp>(origPort.loc, origStruct, fieldValues);
270 }

272 /// We consider non-ESI ports to be ad-hoc signaling or 'raw wires'. (Which
273 /// counts as a signaling protocol if one squints pretty hard). We mostly do
274 /// this since it allows us a more consistent internal API.
275 class RawWires : public SignalingStandard {
276 public:
277  using SignalingStandard::SignalingStandard;
278 
279  void mapInputSignals(OpBuilder &b, Operation *inst, Value instValue,
280  SmallVectorImpl<Value> &newOperands,
281  ArrayRef<Backedge> newResults) override {
282  mapInputDataPorts(b, instValue, newOperands);
283  }
284  void mapOutputSignals(OpBuilder &b, Operation *inst, Value instValue,
285  SmallVectorImpl<Value> &newOperands,
286  ArrayRef<Backedge> newResults) override {
287  instValue.replaceAllUsesWith(mapOutputDataPorts(b, newResults));
288  }
289 
290 private:
291  void buildInputSignals() override {
292  Value newValue = buildInputDataPorts();
293  if (body)
294  body->getArgument(origPort.argNum).replaceAllUsesWith(newValue);
295  }
296 
297  void buildOutputSignals() override {
298  Value output;
299  if (body)
300  output = body->getTerminator()->getOperand(origPort.argNum);
301  buildOutputDataPorts(output);
302  }
303 };

385 LogicalResult ChannelRewriter::rewriteChannelsOnModule() {
386  // Build ops in the module.
387  ModulePortInfo ports = mod.getPorts();
388 
389  // Determine and create a `SignalingStandard` for said port.
390  auto createLowering = [&](PortInfo port) -> LogicalResult {
391  auto &loweredPorts = port.direction == PortDirection::OUTPUT
392  ? loweredOutputs
393  : loweredInputs;
394 
395  auto chanTy = port.type.dyn_cast<ChannelType>();
396  if (!chanTy) {
397  loweredPorts.emplace_back(new RawWires(*this, port));
398  } else {
399  // Mark this as a module which needs port lowering.
400  foundEsiPorts = true;
401 
402  // Determine which ESI signaling standard is specified.
403  ChannelSignaling signaling = chanTy.getSignaling();
404  if (signaling == ChannelSignaling::ValidReady) {
405  loweredPorts.emplace_back(new ValidReady(*this, port));
406  } else if (signaling == ChannelSignaling::FIFO0) {
407  loweredPorts.emplace_back(new FIFO(*this, port));
408  } else {
409  auto error =
410  mod.emitOpError("encountered unknown signaling standard on port '")
411  << stringifyEnum(signaling) << "'";
412  error.attachNote(port.loc);
413  return error;
414  }
415  }
416  return success();
417  };
418 
419  // Find the ESI ports and decide the signaling standard.
420  for (PortInfo port : ports.inputs)
421  if (failed(createLowering(port)))
422  return failure();
423  for (PortInfo port : ports.outputs)
424  if (failed(createLowering(port)))
425  return failure();
426 
427  // Bail early if we didn't find any.
428  if (!foundEsiPorts) {
429  // Memory optimization.
430  loweredInputs.clear();
431  loweredOutputs.clear();
432  return success();
433  }
434 
435  // Lower the ESI ports -- this mutates the body directly and builds the port
436  // lists.
437  for (auto &lowering : loweredInputs)
438  lowering->lowerPort();
439  for (auto &lowering : loweredOutputs)
440  lowering->lowerPort();
441 
442  // Set up vectors to erase _all_ the ports. It's easier to rebuild everything
443  // (including the non-ESI ports) than reason about interleaving the newly
444  // lowered ESI ports with the non-ESI ports. Also, the 'modifyPorts' method
445  // ends up rebuilding the port lists anyway, so this isn't nearly as expensive
446  // as it may seem.
447  SmallVector<unsigned> inputsToErase;
448  for (size_t i = 0, e = mod.getNumInputs(); i < e; ++i)
449  inputsToErase.push_back(i);
450  SmallVector<unsigned> outputsToErase;
451  for (size_t i = 0, e = mod.getNumOutputs(); i < e; ++i)
452  outputsToErase.push_back(i);
453 
454  mod.modifyPorts(newInputs, newOutputs, inputsToErase, outputsToErase);
455 
456  if (!body)
457  return success();
458 
459  // We should only erase the original arguments. New ones were appended with
460  // the `createInput` method call.
461  body->eraseArguments([&ports](BlockArgument arg) {
462  return arg.getArgNumber() < ports.inputs.size();
463  });
464  // Set the new operands, overwriting the old ones.
465  body->getTerminator()->setOperands(newOutputValues);
466 
467  // Memory optimization -- we don't need these anymore.
468  newInputs.clear();
469  newOutputs.clear();
470  newOutputValues.clear();
471  return success();
472 }

474 void ValidReady::buildInputSignals() {
475  Type i1 = IntegerType::get(getContext(), 1, IntegerType::Signless);
476 
477  // When we find one, add a data and valid signal to the new args.
478  Value data = buildInputDataPorts();
479  Value valid =
480  rewriter.createNewInput(origPort, rewriter.validSuffix, i1, validPort);
481 
482  Value ready;
483  if (body) {
484  ImplicitLocOpBuilder b(origPort.loc, body, body->begin());
485  // Build the ESI wrap operation to translate the lowered signals to what
486  // they were. (A later pass takes care of eliminating the ESI ops.)
487  auto wrap = b.create<WrapValidReadyOp>(data, valid);
488  ready = wrap.getReady();
489  // Replace uses of the old ESI port argument with the new one from the
490  // wrap.
491  body->getArgument(origPort.argNum).replaceAllUsesWith(wrap.getChanOutput());
492  }
493 
494  rewriter.createNewOutput(origPort, rewriter.readySuffix, i1, ready,
495  readyPort);
496 }

511  Value ready =

604  newResults[emptyPort.argNum]);
605  inst->getResult(origPort.argNum).replaceAllUsesWith(wrap.getChanOutput());
606  newOperands[rdenPort.argNum] = wrap.getRden();
607 }

609 /// Update an instance of an updated module by adding `esi.[un]wrap.vr`
610 /// ops around the instance. Lowering or folding away `[un]wrap` ops is
611 /// another pass.
612 void ChannelRewriter::updateInstance(InstanceOp inst) {
613  if (!foundEsiPorts)
614  return;
615 
616  ImplicitLocOpBuilder b(inst.getLoc(), inst);
617  BackedgeBuilder beb(b, inst.getLoc());
618  ModulePortInfo ports = mod.getPorts();
619 
620  // Create backedges for the future instance results so the signal mappers can
621  // use the future results as values.
622  SmallVector<Backedge> newResults;
623  for (PortInfo outputPort : ports.outputs)
624  newResults.push_back(beb.get(outputPort.type));
625 
626  // Map the operands.
627  SmallVector<Value> newOperands(ports.inputs.size(), {});
628  for (size_t oldOpIdx = 0, e = inst.getNumOperands(); oldOpIdx < e; ++oldOpIdx)
629  loweredInputs[oldOpIdx]->mapInputSignals(
630  b, inst, inst->getOperand(oldOpIdx), newOperands, newResults);
631 
632  // Map the results.
633  for (size_t oldResIdx = 0, e = inst.getNumResults(); oldResIdx < e;
634  ++oldResIdx)
635  loweredOutputs[oldResIdx]->mapOutputSignals(
636  b, inst, inst->getResult(oldResIdx), newOperands, newResults);
637 
638  // Clone the instance. We cannot just modifiy the existing one since the
639  // result types might have changed types and number of them.
640  assert(llvm::none_of(newOperands, [](Value v) { return !v; }));
641  b.setInsertionPointAfter(inst);
642  auto newInst =
643  b.create<InstanceOp>(mod, inst.getInstanceNameAttr(), newOperands,
644  inst.getParameters(), inst.getInnerSymAttr());
645  newInst->setDialectAttrs(inst->getDialectAttrs());
646 
647  // Assign the backedges to the new results.
648  for (auto [idx, be] : llvm::enumerate(newResults))
649  be.setValue(newInst.getResult(idx));
650 
651  // Erase the old instance.
652  inst.erase();
653 }

670 /// Iterate through the `hw.module[.extern]`s and lower their ports.
671 void ESIPortsPass::runOnOperation() {
672  ModuleOp top = getOperation();
673  ESIHWBuilder b(top);
674  build = &b;
675 
676  // Find all externmodules and try to modify them. Remember the modified
677  // ones.
678  DenseMap<SymbolRefAttr, HWModuleExternOp> externModsMutated;
679  for (auto mod : top.getOps<HWModuleExternOp>())
680  if (mod->hasAttrOfType<UnitAttr>(extModBundleSignalsAttrName) &&
681  updateFunc(mod))
682  externModsMutated[FlatSymbolRefAttr::get(mod)] = mod;
683 
684  // Find all instances and update them.
685  top.walk([&externModsMutated, this](InstanceOp inst) {
686  auto mapIter = externModsMutated.find(inst.getModuleNameAttr());
687  if (mapIter != externModsMutated.end())
688  updateInstance(mapIter->second, inst);
689  });
690 
691  // Find all modules and try to modify them to have wires with valid/ready
692  // semantics. Remember the modified ones.
693  DenseMap<SymbolRefAttr, ChannelRewriter> modsMutated;
694  for (auto mod : top.getOps<HWMutableModuleLike>()) {
695  auto modSym = FlatSymbolRefAttr::get(mod);
696  if (externModsMutated.find(modSym) != externModsMutated.end())
697  continue;
698  auto [entry, emplaced] = modsMutated.try_emplace(modSym, mod);
699  if (!emplaced) {
700  auto error = mod.emitOpError("Detected duplicate symbol on module: ")
701  << modSym;
702  error.attachNote(entry->second.getModule().getLoc());
703  signalPassFailure();
704  continue;
705  }
706  if (failed(entry->second.rewriteChannelsOnModule()))
707  signalPassFailure();
708  }
709 
710  // Find all instances and update them.
711  top.walk([&modsMutated](InstanceOp inst) {
712  auto mapIter = modsMutated.find(inst.getModuleNameAttr());
713  if (mapIter != modsMutated.end())
714  mapIter->second.updateInstance(inst);
715  });
716 
717  build = nullptr;
718 }

720 /// Convert all input and output ChannelTypes into SV Interfaces. For inputs,
721 /// just switch the type to `ModportType`. For outputs, append a `ModportType`
722 /// to the inputs and remove the output channel from the results. Returns true
723 /// if 'mod' was updated. Delay updating the instances to amortize the IR walk
724 /// over all the module updates.
725 bool ESIPortsPass::updateFunc(HWModuleExternOp mod) {
726  auto *ctxt = &getContext();
727 
728  bool updated = false;
729 
730  SmallVector<Attribute> newArgNames, newArgLocs, newResultNames, newResultLocs;
731 
732  // Reconstruct the list of operand types, changing the type whenever an ESI
733  // port is found.
734  SmallVector<Type, 16> newArgTypes;
735  size_t nextArgNo = 0;
736  for (auto argTy : mod.getArgumentTypes()) {
737  auto chanTy = argTy.dyn_cast<ChannelType>();
738  newArgNames.push_back(getModuleArgumentNameAttr(mod, nextArgNo));
739  newArgLocs.push_back(getModuleArgumentLocAttr(mod, nextArgNo));
740  nextArgNo++;
741 
742  if (!chanTy) {
743  newArgTypes.push_back(argTy);
744  continue;
745  }
746 
747  // When we find one, construct an interface, and add the 'source' modport
748  // to the type list.
749  auto iface = build->getOrConstructInterface(chanTy);
750  newArgTypes.push_back(iface.getModportType(ESIHWBuilder::sourceStr));
751  updated = true;
752  }
753 
754  // Iterate through the results and append to one of the two below lists. The
755  // first for non-ESI-ports. The second, ports which have been re-located to
756  // an operand.
757  SmallVector<Type, 8> newResultTypes;
758  SmallVector<DictionaryAttr, 4> newResultAttrs;
759  auto funcType = mod.getFunctionType();
760  for (size_t resNum = 0, numRes = mod.getNumResults(); resNum < numRes;
761  ++resNum) {
762  Type resTy = funcType.getResult(resNum);
763  auto chanTy = resTy.dyn_cast<ChannelType>();
764  auto resNameAttr = getModuleResultNameAttr(mod, resNum);
765  auto resLocAttr = getModuleResultLocAttr(mod, resNum);
766  if (!chanTy) {
767  newResultTypes.push_back(resTy);
768  newResultNames.push_back(resNameAttr);
769  newResultLocs.push_back(resLocAttr);
770  continue;
771  }
772 
773  // When we find one, construct an interface, and add the 'sink' modport to
774  // the type list.
775  sv::InterfaceOp iface = build->getOrConstructInterface(chanTy);
776  sv::ModportType sinkPort = iface.getModportType(ESIHWBuilder::sinkStr);
777  newArgTypes.push_back(sinkPort);
778  newArgNames.push_back(resNameAttr);
779  newArgLocs.push_back(resLocAttr);
780  updated = true;
781  }
782 
783  mod->removeAttr(extModBundleSignalsAttrName);
784  if (!updated)
785  return false;
786 
787  // Set the new types.
788  auto newFuncType = FunctionType::get(ctxt, newArgTypes, newResultTypes);
789  mod.setType(newFuncType);
790  setModuleArgumentNames(mod, newArgNames);
791  setModuleArgumentLocs(mod, newArgLocs);
792  setModuleResultNames(mod, newResultNames);
793  setModuleResultLocs(mod, newResultLocs);
794  return true;
795 }

813 /// Create a reasonable name for a SV interface instance.
814 static std::string &constructInstanceName(Value operand, sv::InterfaceOp iface,
815  std::string &name) {
816  llvm::raw_string_ostream s(name);
817  // Drop the "IValidReady_" part of the interface name.
818  s << llvm::toLower(iface.getSymName()[12]) << iface.getSymName().substr(13);
819 
820  // Indicate to where the source is connected.
821  if (operand.hasOneUse()) {
822  Operation *dstOp = *operand.getUsers().begin();
823  if (auto instOp = dyn_cast<InstanceOp>(dstOp))
824  s << "To" << llvm::toUpper(instOp.getInstanceName()[0])
825  << instOp.getInstanceName().substr(1);
826  else if (auto dstName = dstOp->getAttrOfType<StringAttr>("name"))
827  s << "To" << dstName.getValue();
828  }
829 
830  // Indicate to where the sink is connected.
831  StringRef operName = getOperandName(operand);
832  if (!operName.empty())
833  s << "From" << llvm::toUpper(operName[0]) << operName.substr(1);
834  return s.str();
835 }

837 /// Update an instance of an updated module by adding `esi.(un)wrap.iface`
838 /// around the instance. Create a new instance at the end from the lists built
839 /// up before.
840 void ESIPortsPass::updateInstance(HWModuleExternOp mod, InstanceOp inst) {
841  using namespace circt::sv;
842  circt::ImplicitLocOpBuilder instBuilder(inst.getLoc(), inst);
843  FunctionType funcTy = mod.getFunctionType();
844 
845  // op counter for error reporting purposes.
846  size_t opNum = 0;
847  // List of new operands.
848  SmallVector<Value, 16> newOperands;
849 
850  // Fill the new operand list with old plain operands and mutated ones.
851  std::string nameStringBuffer; // raw_string_ostream uses std::string.
852  for (auto op : inst.getOperands()) {
853  auto instChanTy = op.getType().dyn_cast<ChannelType>();
854  if (!instChanTy) {
855  newOperands.push_back(op);
856  ++opNum;
857  continue;
858  }
859 
860  // Get the interface from the cache, and make sure it's the same one as
861  // being used in the module.
862  auto iface = build->getOrConstructInterface(instChanTy);
863  if (iface.getModportType(ESIHWBuilder::sourceStr) !=
864  funcTy.getInput(opNum)) {
865  inst.emitOpError("ESI ChannelType (operand #")
866  << opNum << ") doesn't match module!";
867  ++opNum;
868  newOperands.push_back(op);
869  continue;
870  }
871  ++opNum;
872 
873  // Build a gasket by instantiating an interface, connecting one end to an
874  // `esi.unwrap.iface` and the other end to the instance.
875  auto ifaceInst =
876  instBuilder.create<InterfaceInstanceOp>(iface.getInterfaceType());
877  nameStringBuffer.clear();
878  ifaceInst->setAttr(
879  "name",
880  StringAttr::get(mod.getContext(),
881  constructInstanceName(op, iface, nameStringBuffer)));
882  GetModportOp sinkModport =
883  instBuilder.create<GetModportOp>(ifaceInst, ESIHWBuilder::sinkStr);
884  instBuilder.create<UnwrapSVInterfaceOp>(op, sinkModport);
885  GetModportOp sourceModport =
886  instBuilder.create<GetModportOp>(ifaceInst, ESIHWBuilder::sourceStr);
887  // Finally, add the correct modport to the list of operands.
888  newOperands.push_back(sourceModport);
889  }
890 
891  // Go through the results and get both a list of the plain old values being
892  // produced and their types.
893  SmallVector<Value, 8> newResults;
894  SmallVector<Type, 8> newResultTypes;
895  for (size_t resNum = 0, numRes = inst.getNumResults(); resNum < numRes;
896  ++resNum) {
897  Value res = inst.getResult(resNum);
898  auto instChanTy = res.getType().dyn_cast<ChannelType>();
899  if (!instChanTy) {
900  newResults.push_back(res);
901  newResultTypes.push_back(res.getType());
902  continue;
903  }
904 
905  // Get the interface from the cache, and make sure it's the same one as
906  // being used in the module.
907  auto iface = build->getOrConstructInterface(instChanTy);
908  if (iface.getModportType(ESIHWBuilder::sinkStr) != funcTy.getInput(opNum)) {
909  inst.emitOpError("ESI ChannelType (result #")
910  << resNum << ", operand #" << opNum << ") doesn't match module!";
911  ++opNum;
912  newResults.push_back(res);
913  newResultTypes.push_back(res.getType());
914  continue;
915  }
916  ++opNum;
917 
918  // Build a gasket by instantiating an interface, connecting one end to an
919  // `esi.wrap.iface` and the other end to the instance. Append it to the
920  // operand list.
921  auto ifaceInst =
922  instBuilder.create<InterfaceInstanceOp>(iface.getInterfaceType());
923  nameStringBuffer.clear();
924  ifaceInst->setAttr(
925  "name",
926  StringAttr::get(mod.getContext(),
927  constructInstanceName(res, iface, nameStringBuffer)));
928  GetModportOp sourceModport =
929  instBuilder.create<GetModportOp>(ifaceInst, ESIHWBuilder::sourceStr);
930  auto newChannel =
931  instBuilder.create<WrapSVInterfaceOp>(res.getType(), sourceModport);
932  // Connect all the old users of the output channel with the newly
933  // wrapped replacement channel.
934  res.replaceAllUsesWith(newChannel);
935  GetModportOp sinkModport =
936  instBuilder.create<GetModportOp>(ifaceInst, ESIHWBuilder::sinkStr);
937  // And add the modport on the other side to the new operand list.
938  newOperands.push_back(sinkModport);
939  }
940 
941  // Create the new instance!
942  InstanceOp newInst = instBuilder.create<InstanceOp>(
943  mod, inst.getInstanceNameAttr(), newOperands, inst.getParameters(),
944  inst.getInnerSymAttr());
945 
946  // Go through the old list of non-ESI result values, and replace them with
947  // the new non-ESI results.
948  for (size_t resNum = 0, numRes = newResults.size(); resNum < numRes;
949  ++resNum) {
950  newResults[resNum].replaceAllUsesWith(newInst.getResult(resNum));
951  }
952  // Erase the old instance!
953  inst.erase();
954 }
