74 /// Lower CompRegOp to `sv.reg` and `sv.alwaysff`. Use a posedge clock and
75 /// synchronous reset.
76 template <typename OpTy>
77 class CompRegLower : public OpConversionPattern<OpTy> {
78 public:
79  CompRegLower(MLIRContext *context, bool lowerToAlwaysFF)
80  : OpConversionPattern<OpTy>(context), lowerToAlwaysFF(lowerToAlwaysFF) {}
81 
82  using OpConversionPattern<OpTy>::OpConversionPattern;
83  using OpAdaptor = typename OpConversionPattern<OpTy>::OpAdaptor;
84 
85  LogicalResult
86  matchAndRewrite(OpTy reg, OpAdaptor adaptor,
87  ConversionPatternRewriter &rewriter) const final {
88  Location loc = reg.getLoc();
89 
90  auto svReg = rewriter.create<sv::RegOp>(loc, reg.getResult().getType(),
91  reg.getNameAttr());
92  svReg->setDialectAttrs(reg->getDialectAttrs());
93 
94  // If the seq::CompRegOp has an inner_sym attribute, set this for the
95  // sv::RegOp inner_sym attribute.
96  if (reg.getSymName().has_value())
97  svReg.setInnerSymAttr(reg.getSymNameAttr());
98 
99  circt::sv::setSVAttributes(svReg, circt::sv::getSVAttributes(reg));
100 
101  auto regVal = rewriter.create<sv::ReadInOutOp>(loc, svReg);
102 
103  auto assignValue = [&] { createAssign(rewriter, svReg, reg); };
104  auto assignReset = [&] {
105  rewriter.create<sv::PAssignOp>(loc, svReg, adaptor.getResetValue());
106  };
107 
108  if (adaptor.getReset() && adaptor.getResetValue()) {
109  if (lowerToAlwaysFF) {
110  rewriter.create<sv::AlwaysFFOp>(
111  loc, sv::EventControl::AtPosEdge, adaptor.getClk(),
112  ResetType::SyncReset, sv::EventControl::AtPosEdge,
113  adaptor.getReset(), assignValue, assignReset);
114  } else {
115  rewriter.create<sv::AlwaysOp>(
116  loc, sv::EventControl::AtPosEdge, adaptor.getClk(), [&] {
117  rewriter.create<sv::IfOp>(loc, adaptor.getReset(), assignReset,
118  assignValue);
119  });
120  }
121  } else {
122  if (lowerToAlwaysFF) {
123  rewriter.create<sv::AlwaysFFOp>(loc, sv::EventControl::AtPosEdge,
124  adaptor.getClk(), assignValue);
125  } else {
126  rewriter.create<sv::AlwaysOp>(loc, sv::EventControl::AtPosEdge,
127  adaptor.getClk(), assignValue);
128  }
129  }
130 
131  rewriter.replaceOp(reg, {regVal});
132  return success();
133  }
134 
135 private:
136  bool lowerToAlwaysFF;
137 };

213 void FirRegLower::addToIfBlock(OpBuilder &builder, Value cond,
214  const std::function<void()> &trueSide,
215  const std::function<void()> &falseSide) {
216  auto op = ifCache.lookup({builder.getBlock(), cond});
217  // Always build both sides of the if, in case we want to use an empty else
218  // later. This way we don't have to build a new if and replace it.
219  if (!op) {
220  auto newIfOp =
221  builder.create<sv::IfOp>(cond.getLoc(), cond, trueSide, falseSide);
222  ifCache.insert({{builder.getBlock(), cond}, newIfOp});
223  } else {
224  OpBuilder::InsertionGuard guard(builder);
225  builder.setInsertionPointToEnd(op.getThenBlock());
226  trueSide();
227  builder.setInsertionPointToEnd(op.getElseBlock());
228  falseSide();
229  }

232 void FirRegLower::lower() {
233  // Find all registers to lower in the module.
234  auto regs = module.getOps<seq::FirRegOp>();
235  if (regs.empty())
236  return;
237 
238  // Lower the regs to SV regs.
239  SmallVector<RegLowerInfo> toInit;
240  for (auto reg : llvm::make_early_inc_range(regs))
241  toInit.push_back(lower(reg));
242 
243  // Compute total width of random space. Place non-chisel registers at the end
244  // of the space. The Random space is unique to the initial block, due to
245  // verilog thread rules, so we can drop trailing random calls if they are
246  // unused.
247  uint64_t maxBit = 0;
248  for (auto reg : toInit)
249  if (reg.randStart >= 0)
250  maxBit = std::max(maxBit, (uint64_t)reg.randStart + reg.width);
251 
252  // This is a map from async reset signals to register info.
253  llvm::MapVector<Value, SmallVector<RegLowerInfo>> asyncResets;
254  for (auto &reg : toInit) {
255  if (reg.randStart == -1) {
256  reg.randStart = maxBit;
257  maxBit += reg.width;
258  }
259  if (reg.asyncResetSignal)
260  asyncResets[reg.asyncResetSignal].emplace_back(reg);
261  }
262 
263  // Create an initial block at the end of the module where random
264  // initialisation will be inserted. Create two builders into the two
265  // `ifdef` ops where the registers will be placed.
266  //
267  // `ifndef SYNTHESIS
268  // `ifdef RANDOMIZE_REG_INIT
269  // ... regBuilder ...
270  // `endif
271  // initial
272  // `INIT_RANDOM_PROLOG_
273  // ... initBuilder ..
274  // `endif
275  if (toInit.empty() || (disableRegRandomization && asyncResets.empty()))
276  return;
277 
278  auto loc = module.getLoc();
279  MLIRContext *context = module.getContext();
280  auto randInitRef = sv::MacroIdentAttr::get(context, "RANDOMIZE_REG_INIT");
281 
282  auto builder =
283  ImplicitLocOpBuilder::atBlockTerminator(loc, module.getBodyBlock());
284 
285  builder.create<sv::IfDefOp>("ENABLE_INITIAL_REG_", [&] {
286  builder.create<sv::OrderedOutputOp>([&] {
287  builder.create<sv::IfDefOp>("FIRRTL_BEFORE_INITIAL", [&] {
288  builder.create<sv::VerbatimOp>("`FIRRTL_BEFORE_INITIAL");
289  });
290 
291  builder.create<sv::InitialOp>([&] {
292  if (!disableRegRandomization) {
293  builder.create<sv::IfDefProceduralOp>("INIT_RANDOM_PROLOG_", [&] {
294  builder.create<sv::VerbatimOp>("`INIT_RANDOM_PROLOG_");
295  });
296  builder.create<sv::IfDefProceduralOp>(randInitRef, [&] {
297  // Create randomization vector
298  SmallVector<Value> randValues;
299  auto numRandomCalls = (maxBit + 31) / 32;
300  auto logic = builder.create<sv::LogicOp>(
301  loc,
302  hw::UnpackedArrayType::get(builder.getIntegerType(32),
303  numRandomCalls),
304  "_RANDOM");
305  // Indvar's width must be equal to `ceil(log2(numRandomCalls +
306  // 1))` to avoid overflow.
307  auto inducionVariableWidth = llvm::Log2_64_Ceil(numRandomCalls + 1);
308  auto arrayIndexWith = llvm::Log2_64_Ceil(numRandomCalls);
309  auto lb =
310  getOrCreateConstant(loc, APInt::getZero(inducionVariableWidth));
311  auto ub = getOrCreateConstant(
312  loc, APInt(inducionVariableWidth, numRandomCalls));
313  auto step =
314  getOrCreateConstant(loc, APInt(inducionVariableWidth, 1));
315  auto forLoop = builder.create<sv::ForOp>(
316  loc, lb, ub, step, "i", [&](BlockArgument iter) {
317  auto rhs = builder.create<sv::MacroRefExprSEOp>(
318  loc, builder.getIntegerType(32), "RANDOM");
319  Value iterValue = iter;
320  if (!iter.getType().isInteger(arrayIndexWith))
321  iterValue = builder.create<comb::ExtractOp>(
322  loc, iterValue, 0, arrayIndexWith);
323  auto lhs = builder.create<sv::ArrayIndexInOutOp>(loc, logic,
324  iterValue);
325  builder.create<sv::BPAssignOp>(loc, lhs, rhs);
326  });
327  builder.setInsertionPointAfter(forLoop);
328  for (uint64_t x = 0; x < numRandomCalls; ++x) {
329  auto lhs = builder.create<sv::ArrayIndexInOutOp>(
330  loc, logic,
331  getOrCreateConstant(loc, APInt(arrayIndexWith, x)));
332  randValues.push_back(lhs.getResult());
333  }
334 
335  // Create initialisers for all registers.
336  for (auto &svReg : toInit)
337  initialize(builder, svReg, randValues);
338  });
339  }
340 
341  if (!asyncResets.empty()) {
342  // If the register is async reset, we need to insert extra
343  // initialization in post-randomization so that we can set the
344  // reset value to register if the reset signal is enabled.
345  for (auto &reset : asyncResets) {
346  // if (reset) begin
347  // ..
348  // end
349  builder.create<sv::IfOp>(reset.first, [&] {
350  for (auto &reg : reset.second)
351  builder.create<sv::BPAssignOp>(reg.reg.getLoc(), reg.reg,
352  reg.asyncResetValue);
353  });
354  }
355  }
356  });
357 
358  builder.create<sv::IfDefOp>("FIRRTL_AFTER_INITIAL", [&] {
359  builder.create<sv::VerbatimOp>("`FIRRTL_AFTER_INITIAL");
360  });
361  });
362  });
363 
364  module->removeAttr("firrtl.random_init_width");
365 }

367 // Return true if two arguments are equivalent, or if both of them are the same
368 // array indexing.
369 // NOLINTNEXTLINE(misc-no-recursion)
370 static bool areEquivalentValues(Value term, Value next) {
371  if (term == next)
372  return true;
373  // Check whether these values are equivalent array accesses with constant
374  // index. We have to check the equivalence recursively because they might not
375  // be CSEd.
376  if (auto t1 = term.getDefiningOp<hw::ArrayGetOp>())
377  if (auto t2 = next.getDefiningOp<hw::ArrayGetOp>())
378  if (auto c1 = t1.getIndex().getDefiningOp<hw::ConstantOp>())
379  if (auto c2 = t2.getIndex().getDefiningOp<hw::ConstantOp>())
380  return c1.getType() == c2.getType() &&
381  c1.getValue() == c2.getValue() &&
382  areEquivalentValues(t1.getInput(), t2.getInput());
383  // Otherwise, regard as different.
384  // TODO: Handle struct if necessary.
385  return false;
386 }

388 static llvm::SetVector<Value> extractConditions(Value value) {
389  auto andOp = value.getDefiningOp<comb::AndOp>();
390  // If the value is not AndOp with a bin flag, use it as a condition.
391  if (!andOp || !andOp.getTwoState()) {
392  llvm::SetVector<Value> ret;

408 // Return a tuple <cond, idx, val> if the array register update can be
409 // represented with a dynamic index assignment:
410 // if (cond)
411 // reg[idx] <= val;
412 //
413 std::optional<std::tuple<Value, Value, Value>>
414 FirRegLower::tryRestoringSubaccess(OpBuilder &builder, Value reg, Value term,
415  hw::ArrayCreateOp nextRegValue) {
416  Value trueVal;
417  SmallVector<Value> muxConditions;
418  // Compat fix for GCC12's libstdc++, cannot use
419  // llvm::enumerate(llvm::reverse(OperandRange)). See #4900.
420  SmallVector<Value> reverseOpValues(llvm::reverse(nextRegValue.getOperands()));
421  if (!llvm::all_of(llvm::enumerate(reverseOpValues), [&](auto idxAndValue) {
422  // Check that `nextRegValue[i]` is `cond_i ? val : reg[i]`.
423  auto [i, value] = idxAndValue;
424  auto mux = value.template getDefiningOp<comb::MuxOp>();
425  // Ensure that mux has binary flag.
426  if (!mux || !mux.getTwoState())
427  return false;
428  // The next value must be same.
429  if (trueVal && trueVal != mux.getTrueValue())
430  return false;
431  if (!trueVal)
432  trueVal = mux.getTrueValue();
433  muxConditions.push_back(mux.getCond());
434  // Check that ith element is an element of the register we are
435  // currently lowering.
436  auto arrayGet =
437  mux.getFalseValue().template getDefiningOp<hw::ArrayGetOp>();
438  if (!arrayGet)
439  return false;
440  return areEquivalentValues(arrayGet.getInput(), term) &&
441  getConstantValue(arrayGet.getIndex()) == i;
442  }))
443  return {};
444 
445  // Extract common expressions among mux conditions.
446  llvm::SetVector<Value> commonConditions =
447  extractConditions(muxConditions.front());
448  for (auto condition : ArrayRef(muxConditions).drop_front()) {
449  auto cond = extractConditions(condition);
450  commonConditions.remove_if([&](auto v) { return !cond.contains(v); });
451  }
452  Value indexValue;
453  for (auto [idx, condition] : llvm::enumerate(muxConditions)) {
454  llvm::SetVector<Value> extractedConditions = extractConditions(condition);
455  // Remove common conditions and check the remaining condition is only an
456  // index comparision.
457  extractedConditions.remove_if(
458  [&](auto v) { return commonConditions.contains(v); });
459  if (extractedConditions.size() != 1)
460  return {};
461 
462  auto indexCompare =
463  (*extractedConditions.begin()).getDefiningOp<comb::ICmpOp>();
464  if (!indexCompare || !indexCompare.getTwoState() ||
465  indexCompare.getPredicate() != comb::ICmpPredicate::eq)
466  return {};
467  // `IndexValue` must be same.
468  if (indexValue && indexValue != indexCompare.getLhs())
469  return {};
470  if (!indexValue)
471  indexValue = indexCompare.getLhs();
472  if (getConstantValue(indexCompare.getRhs()) != idx)
473  return {};
474  }
475 
476  OpBuilder::InsertionGuard guard(builder);
477  builder.setInsertionPointAfterValue(reg);
478  Value commonConditionValue;
479  if (commonConditions.empty())
480  commonConditionValue = getOrCreateConstant(reg.getLoc(), APInt(1, 1));
481  else
482  commonConditionValue = builder.createOrFold<comb::AndOp>(
483  reg.getLoc(), builder.getI1Type(), commonConditions.takeVector(), true);
484  return std::make_tuple(commonConditionValue, indexValue, trueVal);
485 }

487 void FirRegLower::createTree(OpBuilder &builder, Value reg, Value term,
488  Value next) {
489  // If term and next values are equivalent, we don't have to create an
490  // assignment.
491  if (areEquivalentValues(term, next))
492  return;
493  auto mux = next.getDefiningOp<comb::MuxOp>();
494  if (mux && mux.getTwoState()) {
495  addToIfBlock(
496  builder, mux.getCond(),
497  [&]() { createTree(builder, reg, term, mux.getTrueValue()); },
498  [&]() { createTree(builder, reg, term, mux.getFalseValue()); });
499  } else {
500  // If the next value is an array creation, split the value into
501  // invidial elements and construct trees recursively.
502  if (auto array = next.getDefiningOp<hw::ArrayCreateOp>()) {
503  // First, try restoring subaccess assignments.
504  if (auto matchResultOpt =
505  tryRestoringSubaccess(builder, reg, term, array)) {
506  Value cond, index, trueValue;
507  std::tie(cond, index, trueValue) = *matchResultOpt;
508  addToIfBlock(
509  builder, cond,
510  [&]() {
511  Value nextReg;
512  {
513  // Create an array index op just after `reg`.
514  OpBuilder::InsertionGuard guard(builder);
515  builder.setInsertionPointAfterValue(reg);
516  nextReg = builder.create<sv::ArrayIndexInOutOp>(reg.getLoc(),
517  reg, index);
518  }
519  auto termElement =
520  builder.create<hw::ArrayGetOp>(term.getLoc(), term, index);
521  createTree(builder, nextReg, termElement, trueValue);
522  termElement.erase();
523  },
524  []() {});
525  ++numSubaccessRestored;
526  return;
527  }
528  // Compat fix for GCC12's libstdc++, cannot use
529  // llvm::enumerate(llvm::reverse(OperandRange)). See #4900.
530  SmallVector<Value> reverseOpValues(llvm::reverse(array.getOperands()));
531  for (auto [idx, value] : llvm::enumerate(reverseOpValues)) {
532 
533  // Create an index constant.
534  auto idxVal = getOrCreateConstant(
535  array.getLoc(),
536  APInt(std::max(1u, llvm::Log2_64_Ceil(array.getOperands().size())),
537  idx));
538 
539  auto &index = arrayIndexCache[{reg, idx}];
540  if (!index) {
541  // Create an array index op just after `reg`.
542  OpBuilder::InsertionGuard guard(builder);
543  builder.setInsertionPointAfterValue(reg);
544  index =
545  builder.create<sv::ArrayIndexInOutOp>(reg.getLoc(), reg, idxVal);
546  }
547 
548  auto termElement =
549  builder.create<hw::ArrayGetOp>(term.getLoc(), term, idxVal);
550  createTree(builder, index, termElement, value);
551  // This value was used to check the equivalence of elements so useless
552  // anymore.
553  termElement.erase();
554  }
555  return;
556  }
557  builder.create<sv::PAssignOp>(term.getLoc(), reg, next);
558  }
559 }

561 FirRegLower::RegLowerInfo FirRegLower::lower(FirRegOp reg) {
562  Location loc = reg.getLoc();
563 
564  ImplicitLocOpBuilder builder(reg.getLoc(), reg);
565  RegLowerInfo svReg{nullptr, nullptr, nullptr, -1, 0};
566  svReg.reg = builder.create<sv::RegOp>(loc, reg.getType(), reg.getNameAttr());
567  svReg.width = hw::getBitWidth(reg.getResult().getType());
568 
569  if (auto attr = reg->getAttrOfType<IntegerAttr>("firrtl.random_init_start"))
570  svReg.randStart = attr.getUInt();
571 
572  // Don't move these over
573  reg->removeAttr("firrtl.random_init_start");
574 
575  // Move Attributes
576  svReg.reg->setDialectAttrs(reg->getDialectAttrs());
577 
578  if (auto innerSymAttr = reg.getInnerSymAttr())
579  svReg.reg.setInnerSymAttr(innerSymAttr);
580 
581  auto regVal = builder.create<sv::ReadInOutOp>(loc, svReg.reg);
582 
583  if (reg.hasReset()) {
584  addToAlwaysBlock(
585  module.getBodyBlock(), sv::EventControl::AtPosEdge, reg.getClk(),
586  [&](OpBuilder &b) {
587  // If this is an AsyncReset, ensure that we emit a self connect to
588  // avoid erroneously creating a latch construct.
589  if (reg.getIsAsync() && areEquivalentValues(reg, reg.getNext()))
590  b.create<sv::PAssignOp>(reg.getLoc(), svReg.reg, reg);
591  else
592  createTree(b, svReg.reg, reg, reg.getNext());
593  },
594  reg.getIsAsync() ? ResetType::AsyncReset : ResetType::SyncReset,
595  sv::EventControl::AtPosEdge, reg.getReset(),
596  [&](OpBuilder &builder) {

615 // Initialize registers by assigning each element recursively instead of
616 // initializing entire registers. This is necessary as a workaround for
617 // verilator which allocates many local variables for concat op.
618 void FirRegLower::initializeRegisterElements(Location loc, OpBuilder &builder,
619  Value reg, Value randomSource,
620  unsigned &pos) {
621  auto type = reg.getType().cast<sv::InOutType>().getElementType();
622  if (auto intTy = hw::type_dyn_cast<IntegerType>(type)) {
623  // Use randomSource[pos-1:pos-width] as a random value.
624  pos -= intTy.getWidth();
625  auto elem = builder.createOrFold<comb::ExtractOp>(loc, randomSource, pos,
626  intTy.getWidth());
627  builder.create<sv::BPAssignOp>(loc, reg, elem);
628  } else if (auto array = hw::type_dyn_cast<hw::ArrayType>(type)) {
629  for (unsigned i = 0, e = array.getSize(); i < e; ++i) {
630  auto index = getOrCreateConstant(loc, APInt(llvm::Log2_64_Ceil(e), i));
631  initializeRegisterElements(
632  loc, builder, builder.create<sv::ArrayIndexInOutOp>(loc, reg, index),
633  randomSource, pos);
634  }
635  } else if (auto structType = hw::type_dyn_cast<hw::StructType>(type)) {
636  for (auto e : structType.getElements())
637  initializeRegisterElements(
638  loc, builder,
639  builder.create<sv::StructFieldInOutOp>(loc, reg, e.name),
640  randomSource, pos);
641  } else {
642  assert(false && "unsupported type");
643  }
644 }

646 void FirRegLower::initialize(OpBuilder &builder, RegLowerInfo reg,
647  ArrayRef<Value> rands) {
648  auto loc = reg.reg.getLoc();
649  SmallVector<Value> nibbles;
650  if (reg.width == 0)
651  return;
652 
653  uint64_t width = reg.width;
654  uint64_t offset = reg.randStart;
655  while (width) {
656  auto index = offset / 32;
657  auto start = offset % 32;
658  auto nwidth = std::min(32 - start, width);
659  auto elemVal = builder.create<sv::ReadInOutOp>(loc, rands[index]);
660  auto elem =
661  builder.createOrFold<comb::ExtractOp>(loc, elemVal, start, nwidth);
662  nibbles.push_back(elem);
663  offset += nwidth;
664  width -= nwidth;
665  }
666  auto concat = builder.createOrFold<comb::ConcatOp>(loc, nibbles);
667  unsigned pos = reg.width;
668  // Initialize register elements.
669  initializeRegisterElements(loc, builder, reg.reg, concat, pos);
670 }

672 void FirRegLower::addToAlwaysBlock(Block *block, sv::EventControl clockEdge,
673  Value clock,
674  std::function<void(OpBuilder &)> body,
675  ::ResetType resetStyle,
676  sv::EventControl resetEdge, Value reset,
677  std::function<void(OpBuilder &)> resetBody) {
678  auto loc = clock.getLoc();
679  auto builder = ImplicitLocOpBuilder::atBlockTerminator(loc, block);
680  AlwaysKeyType key{builder.getBlock(), clockEdge, clock,
681  resetStyle, resetEdge, reset};
682 
683  sv::AlwaysOp alwaysOp;
684  sv::IfOp insideIfOp;
685  if (!emitSeparateAlwaysBlocks) {
686  std::tie(alwaysOp, insideIfOp) = alwaysBlocks[key];
687  }
688 
689  if (!alwaysOp) {
690  if (reset) {
691  assert(resetStyle != ::ResetType::NoReset);
692  // Here, we want to create the following structure with sv.always and
693  // sv.if. If `reset` is async, we need to add `reset` to a sensitivity
694  // list.
695  //
696  // sv.always @(clockEdge or reset) {
697  // sv.if (reset) {
698  // resetBody
699  // } else {
700  // body
701  // }
702  // }
703 
704  auto createIfOp = [&]() {
705  // It is weird but intended. Here we want to create an empty sv.if
706  // with an else block.
707  insideIfOp = builder.create<sv::IfOp>(
708  reset, []() {}, []() {});
709  };
710  if (resetStyle == ::ResetType::AsyncReset) {
711  sv::EventControl events[] = {clockEdge, resetEdge};
712  Value clocks[] = {clock, reset};
713 
714  alwaysOp = builder.create<sv::AlwaysOp>(events, clocks, [&]() {
715  if (resetEdge == sv::EventControl::AtNegEdge)
716  llvm_unreachable("negative edge for reset is not expected");
717  createIfOp();
718  });
719  } else {
720  alwaysOp = builder.create<sv::AlwaysOp>(clockEdge, clock, createIfOp);
721  }
722  } else {
723  assert(!resetBody);
724  alwaysOp = builder.create<sv::AlwaysOp>(clockEdge, clock);
725  insideIfOp = nullptr;
726  }
727  }
728 
729  if (reset) {
730  assert(insideIfOp && "reset body must be initialized before");
731  auto resetBuilder =
732  ImplicitLocOpBuilder::atBlockEnd(loc, insideIfOp.getThenBlock());
733  resetBody(resetBuilder);
734 
735  auto bodyBuilder =
736  ImplicitLocOpBuilder::atBlockEnd(loc, insideIfOp.getElseBlock());
737  body(bodyBuilder);
738  } else {
739  auto bodyBuilder =
740  ImplicitLocOpBuilder::atBlockEnd(loc, alwaysOp.getBodyBlock());
741  body(bodyBuilder);
742  }
743 
744  if (!emitSeparateAlwaysBlocks) {
745  alwaysBlocks[key] = {alwaysOp, insideIfOp};
746  }
747 }
