23 /// ExprVisitor is a visitor for FIRRTL expression nodes.
24 template <typename ConcreteType, typename ResultType = void,
25  typename... ExtraArgs>
26 class ExprVisitor {
27 public:
28  ResultType dispatchExprVisitor(Operation *op, ExtraArgs... args) {
29  auto *thisCast = static_cast<ConcreteType *>(this);
30  return TypeSwitch<Operation *, ResultType>(op)
31  // Basic Expressions
32  .template Case<
33  ConstantOp, SpecialConstantOp, AggregateConstantOp, InvalidValueOp,
34  SubfieldOp, SubindexOp, SubaccessOp, IsTagOp, SubtagOp,
35  BundleCreateOp, VectorCreateOp, FEnumCreateOp, MultibitMuxOp,
36  TagExtractOp, OpenSubfieldOp, OpenSubindexOp,
37  // Arithmetic and Logical Binary Primitives.
38  AddPrimOp, SubPrimOp, MulPrimOp, DivPrimOp, RemPrimOp, AndPrimOp,
39  OrPrimOp, XorPrimOp,
40  // Elementwise operations,
41  ElementwiseOrPrimOp, ElementwiseAndPrimOp, ElementwiseXorPrimOp,
42  // Comparisons.
43  LEQPrimOp, LTPrimOp, GEQPrimOp, GTPrimOp, EQPrimOp, NEQPrimOp,
44  // Misc Binary Primitives.
45  CatPrimOp, DShlPrimOp, DShlwPrimOp, DShrPrimOp,
46  // Unary operators.
47  AsSIntPrimOp, AsUIntPrimOp, AsAsyncResetPrimOp, AsClockPrimOp,
48  CvtPrimOp, NegPrimOp, NotPrimOp, AndRPrimOp, OrRPrimOp, XorRPrimOp,
49  // Intrinsic Expressions.
50  IsXIntrinsicOp, PlusArgsValueIntrinsicOp, PlusArgsTestIntrinsicOp,
51  SizeOfIntrinsicOp,
52  // Miscellaneous.
53  BitsPrimOp, HeadPrimOp, MuxPrimOp, PadPrimOp, ShlPrimOp, ShrPrimOp,
54  TailPrimOp, VerbatimExprOp, HWStructCastOp, BitCastOp, RefSendOp,
55  RefResolveOp, RefSubOp,
56  // Casts to deal with weird stuff
57  UninferredResetCastOp, UninferredWidthCastOp, ConstCastOp,
58  mlir::UnrealizedConversionCastOp>([&](auto expr) -> ResultType {
59  return thisCast->visitExpr(expr, args...);
60  })
61  .Default([&](auto expr) -> ResultType {
62  return thisCast->visitInvalidExpr(op, args...);
63  });
64  }
65 
66  /// This callback is invoked on any non-expression operations.
67  ResultType visitInvalidExpr(Operation *op, ExtraArgs... args) {
68  op->emitOpError("unknown FIRRTL expression");
69  abort();
70  }
71 
72  /// This callback is invoked on any expression operations that are not handled
73  /// by the concrete visitor.
74  ResultType visitUnhandledExpr(Operation *op, ExtraArgs... args) {
75  return ResultType();
76  }
77 
78  /// This fallback is invoked on any unary expr that isn't explicitly handled.
79  /// The default implementation delegates to the unhandled expression fallback.
80  ResultType visitUnaryExpr(Operation *op, ExtraArgs... args) {
81  return static_cast<ConcreteType *>(this)->visitUnhandledExpr(op, args...);
82  }
83 
84  /// This fallback is invoked on any binary expr that isn't explicitly handled.
85  /// The default implementation delegates to the unhandled expression fallback.
86  ResultType visitBinaryExpr(Operation *op, ExtraArgs... args) {
87  return static_cast<ConcreteType *>(this)->visitUnhandledExpr(op, args...);
88  }
89 
90 #define HANDLE(OPTYPE, OPKIND) \
91  ResultType visitExpr(OPTYPE op, ExtraArgs... args) { \
92  return static_cast<ConcreteType *>(this)->visit##OPKIND##Expr(op, \
93  args...); \
94  }
95 
96  // Basic expressions.
97  HANDLE(ConstantOp, Unhandled);
98  HANDLE(SpecialConstantOp, Unhandled);
99  HANDLE(AggregateConstantOp, Unhandled);
100  HANDLE(BundleCreateOp, Unhandled);
101  HANDLE(VectorCreateOp, Unhandled);
102  HANDLE(FEnumCreateOp, Unhandled);
103  HANDLE(SubfieldOp, Unhandled);
104  HANDLE(SubindexOp, Unhandled);
105  HANDLE(SubaccessOp, Unhandled);
106  HANDLE(IsTagOp, Unhandled);
107  HANDLE(SubtagOp, Unhandled);
108  HANDLE(TagExtractOp, Unhandled);
109  HANDLE(MultibitMuxOp, Unhandled);
110  HANDLE(OpenSubfieldOp, Unhandled);
111  HANDLE(OpenSubindexOp, Unhandled);
112 
113  // Arithmetic and Logical Binary Primitives.
114  HANDLE(AddPrimOp, Binary);
115  HANDLE(SubPrimOp, Binary);
116  HANDLE(MulPrimOp, Binary);
117  HANDLE(DivPrimOp, Binary);
118  HANDLE(RemPrimOp, Binary);
119  HANDLE(AndPrimOp, Binary);
120  HANDLE(OrPrimOp, Binary);
121  HANDLE(XorPrimOp, Binary);
122 
123  // Comparisons.
124  HANDLE(LEQPrimOp, Binary);
125  HANDLE(LTPrimOp, Binary);
126  HANDLE(GEQPrimOp, Binary);
127  HANDLE(GTPrimOp, Binary);
128  HANDLE(EQPrimOp, Binary);
129  HANDLE(NEQPrimOp, Binary);
130 
131  // Misc Binary Primitives.
132  HANDLE(CatPrimOp, Binary);
133  HANDLE(DShlPrimOp, Binary);
134  HANDLE(DShlwPrimOp, Binary);
135  HANDLE(DShrPrimOp, Binary);
136 
137  // Unary operators.
138  HANDLE(AsSIntPrimOp, Unary);
139  HANDLE(AsUIntPrimOp, Unary);
140  HANDLE(AsAsyncResetPrimOp, Unary);
141  HANDLE(AsClockPrimOp, Unary);
142  HANDLE(CvtPrimOp, Unary);
143  HANDLE(NegPrimOp, Unary);
144  HANDLE(NotPrimOp, Unary);
145  HANDLE(AndRPrimOp, Unary);
146  HANDLE(OrRPrimOp, Unary);
147  HANDLE(XorRPrimOp, Unary);
148 
149  HANDLE(ElementwiseOrPrimOp, Unhandled);
150  HANDLE(ElementwiseAndPrimOp, Unhandled);
151  HANDLE(ElementwiseXorPrimOp, Unhandled);
152 
153  // Intrinsic Expr.
154  HANDLE(IsXIntrinsicOp, Unhandled);
155  HANDLE(PlusArgsValueIntrinsicOp, Unhandled);
156  HANDLE(PlusArgsTestIntrinsicOp, Unhandled);
157  HANDLE(SizeOfIntrinsicOp, Unhandled);
158 
159  // Miscellaneous.
160  HANDLE(BitsPrimOp, Unhandled);
161  HANDLE(HeadPrimOp, Unhandled);
162  HANDLE(InvalidValueOp, Unhandled);
163  HANDLE(MuxPrimOp, Unhandled);
164  HANDLE(PadPrimOp, Unhandled);
165  HANDLE(ShlPrimOp, Unhandled);
166  HANDLE(ShrPrimOp, Unhandled);
167  HANDLE(TailPrimOp, Unhandled);
168  HANDLE(VerbatimExprOp, Unhandled);
169  HANDLE(RefSendOp, Unhandled);
170  HANDLE(RefResolveOp, Unhandled);
171  HANDLE(RefSubOp, Unhandled);
172 
173  // Conversions.
174  HANDLE(HWStructCastOp, Unhandled);
175  HANDLE(UninferredResetCastOp, Unhandled);
176  HANDLE(UninferredWidthCastOp, Unhandled);
177  HANDLE(ConstCastOp, Unhandled);
178  HANDLE(mlir::UnrealizedConversionCastOp, Unhandled);
179  HANDLE(BitCastOp, Unhandled);
180 #undef HANDLE
181 };

183 /// ExprVisitor is a visitor for FIRRTL statement nodes.
184 template <typename ConcreteType, typename ResultType = void,
185  typename... ExtraArgs>
186 class StmtVisitor {
187 public:
188  ResultType dispatchStmtVisitor(Operation *op, ExtraArgs... args) {
189  auto *thisCast = static_cast<ConcreteType *>(this);
190  return TypeSwitch<Operation *, ResultType>(op)
191  .template Case<AttachOp, ConnectOp, StrictConnectOp, RefDefineOp,
192  ForceOp, PrintFOp, SkipOp, StopOp, WhenOp, AssertOp,
193  AssumeOp, CoverOp, ProbeOp, RefForceOp,
194  RefForceInitialOp, RefReleaseOp, RefReleaseInitialOp>(
195  [&](auto opNode) -> ResultType {
196  return thisCast->visitStmt(opNode, args...);
197  })
198  .Default([&](auto expr) -> ResultType {
199  return thisCast->visitInvalidStmt(op, args...);
200  });
201  }
202 
203  /// This callback is invoked on any non-Stmt operations.
204  ResultType visitInvalidStmt(Operation *op, ExtraArgs... args) {
205  op->emitOpError("unknown firrtl stmt");
206  abort();
207  }
208 
209  /// This callback is invoked on any Stmt operations that are not handled
210  /// by the concrete visitor.
211  ResultType visitUnhandledStmt(Operation *op, ExtraArgs... args) {
212  return ResultType();
213  }
214 
215 #define HANDLE(OPTYPE) \
216  ResultType visitStmt(OPTYPE op, ExtraArgs... args) { \
217  return static_cast<ConcreteType *>(this)->visitUnhandledStmt(op, args...); \
218  }
219 
220  HANDLE(AttachOp);
221  HANDLE(ConnectOp);
222  HANDLE(StrictConnectOp);
223  HANDLE(RefDefineOp);
224  HANDLE(ForceOp);
225  HANDLE(PrintFOp);
226  HANDLE(SkipOp);
227  HANDLE(StopOp);
228  HANDLE(WhenOp);
229  HANDLE(AssertOp);
230  HANDLE(AssumeOp);
231  HANDLE(CoverOp);
232  HANDLE(ProbeOp);
233  HANDLE(RefForceOp);
234  HANDLE(RefForceInitialOp);
235  HANDLE(RefReleaseOp);
236  HANDLE(RefReleaseInitialOp);
237 
238 #undef HANDLE
239 };

241 /// ExprVisitor is a visitor for FIRRTL declaration nodes.
242 template <typename ConcreteType, typename ResultType = void,
243  typename... ExtraArgs>
244 class DeclVisitor {
245 public:
246  ResultType dispatchDeclVisitor(Operation *op, ExtraArgs... args) {
247  auto *thisCast = static_cast<ConcreteType *>(this);
248  return TypeSwitch<Operation *, ResultType>(op)
249  .template Case<InstanceOp, MemOp, NodeOp, RegOp, RegResetOp, WireOp,
250  VerbatimWireOp>([&](auto opNode) -> ResultType {
251  return thisCast->visitDecl(opNode, args...);
252  })
253  .Default([&](auto expr) -> ResultType {
254  return thisCast->visitInvalidDecl(op, args...);
255  });
256  }
257 
258  /// This callback is invoked on any non-Decl operations.
259  ResultType visitInvalidDecl(Operation *op, ExtraArgs... args) {
260  op->emitOpError("unknown firrtl decl");
261  abort();
262  }
263 
264  /// This callback is invoked on any Decl operations that are not handled
265  /// by the concrete visitor.
266  ResultType visitUnhandledDecl(Operation *op, ExtraArgs... args) {
267  return ResultType();
268  }
269 
270 #define HANDLE(OPTYPE) \
271  ResultType visitDecl(OPTYPE op, ExtraArgs... args) { \
272  return static_cast<ConcreteType *>(this)->visitUnhandledDecl(op, args...); \
273  }
274 
275  HANDLE(InstanceOp);
276  HANDLE(MemOp);
277  HANDLE(NodeOp);
278  HANDLE(RegOp);
279  HANDLE(RegResetOp);
280  HANDLE(WireOp);
281  HANDLE(VerbatimWireOp);
282 #undef HANDLE
283 };

285 /// FIRRTLVisitor allows you to visit all of the expr/stmt/decls with one class
286 /// declaration.
287 ///
288 /// Clients call dispatchVisitor to invoke the dispatch, and may implement
289 /// visitInvalidOp() to get notified about non-FIRRTL dialect nodes and
290 /// visitUnhandledOp() to get notified about FIRRTL dialect ops that are not
291 /// handled specifically.
292 template <typename ConcreteType, typename ResultType = void,
293  typename... ExtraArgs>
294 class FIRRTLVisitor
295  : public ExprVisitor<ConcreteType, ResultType, ExtraArgs...>,
296  public StmtVisitor<ConcreteType, ResultType, ExtraArgs...>,
297  public DeclVisitor<ConcreteType, ResultType, ExtraArgs...> {
298 public:
299  /// This is the main entrypoint for the FIRRTLVisitor.
300  ResultType dispatchVisitor(Operation *op, ExtraArgs... args) {
301  return this->dispatchExprVisitor(op, args...);
302  }
303 
304  // Chain from each visitor onto the next one.
305  ResultType visitInvalidExpr(Operation *op, ExtraArgs... args) {
306  return this->dispatchStmtVisitor(op, args...);
307  }
308  ResultType visitInvalidStmt(Operation *op, ExtraArgs... args) {
309  return this->dispatchDeclVisitor(op, args...);
310  }
311  ResultType visitInvalidDecl(Operation *op, ExtraArgs... args) {
312  return static_cast<ConcreteType *>(this)->visitInvalidOp(op, args...);
313  }
314 
315  // Default to chaining visitUnhandledXXX to visitUnhandledOp.
316  ResultType visitUnhandledExpr(Operation *op, ExtraArgs... args) {
317  return static_cast<ConcreteType *>(this)->visitUnhandledOp(op, args...);
318  }
319  ResultType visitUnhandledStmt(Operation *op, ExtraArgs... args) {
320  return static_cast<ConcreteType *>(this)->visitUnhandledOp(op, args...);
321  }
322  ResultType visitUnhandledDecl(Operation *op, ExtraArgs... args) {
323  return static_cast<ConcreteType *>(this)->visitUnhandledOp(op, args...);
324  }
325 
326  /// visitInvalidOp is an override point for non-FIRRTL dialect operations.
327  ResultType visitInvalidOp(Operation *op, ExtraArgs... args) {
328  return ResultType();
329  }
330 
331  /// visitUnhandledOp is an override point for FIRRTL dialect ops that the
332  /// concrete visitor didn't bother to implement.
333  ResultType visitUnhandledOp(Operation *op, ExtraArgs... args) {
334  return ResultType();
335  }
336 };
