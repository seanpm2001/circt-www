42 // Uses the capnp dynamic interface in conjunction with ESI type reflection to
43 // de-serialize a capnp message into an ESI value.
44 template <typename TESIType>
45 TESIType fromCapnp(DynamicValue::Reader value) {
46  TESIType ret;
47  if (value.getType() != DynamicValue::STRUCT)
48  throw std::runtime_error(
49  "Expected a struct value (all ESI types should be capnp structs)");
50 
51  auto structValue = value.as<DynamicStruct>();
52  for_each(refl::reflect(ret).members, [&](auto member) {
53  auto fieldValue = structValue.get(member.name);
54  using ValueType = typename decltype(member)::value_type;
55  auto castValue = fieldValue.template as<ValueType>();
56  member(ret) = castValue;
57  });
58 
59  return ret;
60 }

62 // A function like above, but which uses Capnproto dynamic message building
63 // together with the refl-cpp reflection to constuct the Capnp message.
64 template <typename TESIType>
65 void toCapnp(TESIType &value, DynamicValue::Builder &message) {
66  if (message.getType() != DynamicValue::STRUCT)
67  throw std::runtime_error(
68  "Expected a struct value (all ESI types should be capnp structs)");
69 
70  auto structMessage = message.as<DynamicStruct>();
71  for_each(refl::reflect(value).members, [&](auto member) {
72  auto &fieldValue = member(value);
73  structMessage.set(member.name, fieldValue);
74  });
75 }

120  // Returns a list of all available endpoints.
121  const std::vector<detail::EsiDpiInterfaceDesc> &list() {
122  if (endpoints.has_value())
123  return *endpoints;
124 
125  // Query the DPI server for a list of available endpoints.
126  auto listReq = dpiClient->listRequest();
127  auto ifaces = listReq.send().wait(ezClient->getWaitScope()).getIfaces();
128  endpoints = std::vector<detail::EsiDpiInterfaceDesc>();
129  for (auto iface : ifaces) {
130  detail::EsiDpiInterfaceDesc desc;
131  desc.endpointID = iface.getEndpointID().cStr();
132  desc.sendTypeID = iface.getSendTypeID();
133  desc.recvTypeID = iface.getRecvTypeID();
134  endpoints->push_back(desc);
135  }
136 
137  // print out the endpoints
138  for (auto ep : *endpoints) {
139  std::cout << "Endpoint: " << ep.endpointID << std::endl;
140  std::cout << " Send Type: " << ep.sendTypeID << std::endl;
141  std::cout << " Recv Type: " << ep.recvTypeID << std::endl;
142  }
143 
144  return *endpoints;

147  template <typename CnPWriteType, typename CnPReadType>
148  auto getPort(const std::vector<std::string> &clientPath) {
149  // Join client path into a single string with '.' as a separator.
150  std::string clientPathStr;
151  for (auto &path : clientPath) {
152  if (!clientPathStr.empty())
153  clientPathStr += '_';
154  clientPathStr += path;
155  }
156 
157  // Everything is nested under "TOP.top"
158  clientPathStr = "TOP.top." + clientPathStr;
159 
160  auto openReq = dpiClient->openRequest<CnPWriteType, CnPReadType>();
161 
162  // Scan through the available endpoints to find the requested one.
163  bool found = false;
164  for (auto &ep : list()) {
165  auto epid = ep.endpointID;
166  if (epid == clientPathStr) {
167  auto iface = openReq.getIface();
168  iface.setEndpointID(epid);
169  iface.setSendTypeID(ep.sendTypeID);
170  iface.setRecvTypeID(ep.recvTypeID);
171  found = true;
172  break;
173  }
174  }
175 
176  if (!found)
177  throw std::runtime_error("Could not find endpoint: " + clientPathStr);
178 
179  // Open the endpoint.
180  auto openResp = openReq.send().wait(ezClient->getWaitScope());
181  return openResp.getIface();
182  }

185  // The cosim backend only supports cosim connectivity implementations
186  return implType == "cosim";
187  }

206  ReadType operator()(WriteType arg) {
207  auto req = port->sendRequest();
208  MallocMessageBuilder mmb;
209  auto dynBuilder = DynamicValue::Builder(req.getMsg());
210  toCapnp<WriteType>(arg, dynBuilder);
211  req.send().wait(this->backend->getWaitScope());
212  std::optional<capnp::Response<typename EsiDpiEndpoint<

215  do {

217  recvReq.setBlock(false);
218 
219  resp = recvReq.send().wait(this->backend->getWaitScope());
220  } while (!resp->getHasData());
