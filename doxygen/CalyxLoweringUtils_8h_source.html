161 public:
162  /// Register 'scheduleable' as being generated through lowering 'block'.
163  ///
164  /// TODO(mortbopet): Add a post-insertion check to ensure that the use-def
165  /// ordering invariant holds for the groups. When the control schedule is
166  /// generated, scheduleables within a block are emitted sequentially based on
167  /// the order that this function was called during conversion.

183  /// scheduleables registered themselves with the block.

206  /// Register reg as being the idx'th iter_args register for 'op'.
207  void addLoopIterReg(Loop op, calyx::RegisterOp reg, unsigned idx) {
208  assert(loopIterRegs[op.getOperation()].count(idx) == 0 &&
209  "A register was already registered for the given loop iter_arg "
210  "index");
211  assert(idx < op.getBodyArgs().size());
212  loopIterRegs[op.getOperation()][idx] = reg;
213  }
214 
215  /// Return a mapping of block argument indices to block argument.
216  calyx::RegisterOp getLoopIterReg(Loop op, unsigned idx) {
217  auto iterRegs = getLoopIterRegs(op);
218  auto it = iterRegs.find(idx);
219  assert(it != iterRegs.end() &&
220  "No iter arg register set for the provided index");
221  return it->second;
222  }
223 
224  /// Return a mapping of block argument indices to block argument.
225  const DenseMap<unsigned, calyx::RegisterOp> &getLoopIterRegs(Loop op) {
226  return loopIterRegs[op.getOperation()];
227  }
228 
229  /// Registers grp to be the loop latch group of `op`.
230  void setLoopLatchGroup(Loop op, calyx::GroupOp group) {
231  Operation *operation = op.getOperation();
232  assert(loopLatchGroups.count(operation) == 0 &&
233  "A latch group was already set for this loopOp");
234  loopLatchGroups[operation] = group;
235  }
236 
237  /// Retrieve the loop latch group registered for `op`.
238  calyx::GroupOp getLoopLatchGroup(Loop op) {
239  auto it = loopLatchGroups.find(op.getOperation());
240  assert(it != loopLatchGroups.end() &&
241  "No loop latch group was set for this loopOp");
242  return it->second;
243  }
244 
245  /// Creates a new group that assigns the 'ops' values to the iter arg
246  /// registers of the loop operation.
247  calyx::GroupOp buildLoopIterArgAssignments(OpBuilder &builder, Loop op,
248  calyx::ComponentOp componentOp,
249  Twine uniqueSuffix,
250  MutableArrayRef<OpOperand> ops) {
251  /// Pass iteration arguments through registers. This follows closely
252  /// to what is done for branch ops.
253  std::string groupName = "assign_" + uniqueSuffix.str();
254  auto groupOp = calyx::createGroup<calyx::GroupOp>(builder, componentOp,
255  op.getLoc(), groupName);
256  /// Create register assignment for each iter_arg. a calyx::GroupDone signal
257  /// is created for each register. These will be &'ed together in
258  /// MultipleGroupDonePattern.
259  for (OpOperand &arg : ops) {
260  auto reg = getLoopIterReg(op, arg.getOperandNumber());
261  buildAssignmentsForRegisterWrite(builder, groupOp, componentOp, reg,
262  arg.get());
263  }
264  return groupOp;
265  }
266 
267 private:
268  /// A mapping from loop ops to iteration argument registers.
269  DenseMap<Operation *, DenseMap<unsigned, calyx::RegisterOp>> loopIterRegs;
270 
271  /// A loop latch group is a group that should be sequentially executed when
272  /// finishing a loop body. The execution of this group will write the
273  /// yield'ed loop body values to the iteration argument registers.
274  DenseMap<Operation *, calyx::GroupOp> loopLatchGroups;
275 };

462 /// Base class for partial lowering passes. A partial lowering pass
463 /// modifies the root operation in place, but does not replace the root
464 /// operation.
465 /// The RewritePatternType template parameter allows for using both
466 /// OpRewritePattern (default) or OpInterfaceRewritePattern.
467 template <class OpType,
468  template <class> class RewritePatternType = OpRewritePattern>
469 class PartialLoweringPattern : public RewritePatternType<OpType> {
470 public:
471  using RewritePatternType<OpType>::RewritePatternType;
472  PartialLoweringPattern(MLIRContext *ctx, LogicalResult &resRef,
473  PatternApplicationState &patternState)
474  : RewritePatternType<OpType>(ctx), partialPatternRes(resRef),
475  patternState(patternState) {}
476 
477  LogicalResult matchAndRewrite(OpType op,
478  PatternRewriter &rewriter) const override {
479  // If this pattern has been applied to this op, it should now fail to match.
480  if (patternState[this].contains(op))
481  return failure();
482 
483  // Do the actual rewrite, marking this op as updated. Because the op is
484  // marked as updated, the pattern driver will re-enqueue the op again.
485  rewriter.updateRootInPlace(
486  op, [&] { partialPatternRes = partiallyLower(op, rewriter); });
487 
488  // Mark that this pattern has been applied to this op.
489  patternState[this].insert(op);
490 
491  return partialPatternRes;
492  }
493 
494  // Hook for subclasses to lower the op using the rewriter.
495  //
496  // Note that this call is wrapped in `updateRootInPlace`, so any direct IR
497  // mutations that are legal to apply during a root update of op are allowed.
498  //
499  // Also note that this means the op will be re-enqueued to the greedy
500  // rewriter's worklist. A safeguard is in place to prevent patterns from
501  // running multiple times, but if the op is erased or otherwise becomes dead
502  // after the call to `partiallyLower`, there will likely be use-after-free
503  // violations. If you will erase the op, override `matchAndRewrite` directly.
504  virtual LogicalResult partiallyLower(OpType op,
505  PatternRewriter &rewriter) const = 0;
506 
507 private:
508  LogicalResult &partialPatternRes;
509  PatternApplicationState &patternState;
510 };
