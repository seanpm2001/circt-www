49 static ParseResult namedPort(StringRef name, FModuleLike mod, unsigned n,
50  StringRef portName) {
51  auto ports = mod.getPorts();
52  if (n >= ports.size()) {
53  mod.emitError(name) << " missing port " << n;
54  return failure();
55  }
56  if (!ports[n].getName().equals(portName)) {
57  mod.emitError(name) << " port " << n << " named '" << ports[n].getName()
58  << "' instead of '" << portName << "'";
59  return failure();
60  }
61  return success();
62 }

101 static ParseResult namedParam(StringRef name, FModuleLike mod,
102  StringRef paramName) {
103  for (auto a : mod.getParameters()) {
104  auto param = a.cast<ParamDeclAttr>();
105  if (param.getName().getValue().equals(paramName)) {
106  if (param.getValue().isa<StringAttr>())
107  return success();
108 
109  mod.emitError(name) << " test has parameter '" << param.getName()
110  << "' which should be a string but is not";
111  return failure();
112  }
113  }
114  mod.emitError(name) << " is missing parameter " << paramName;
115  return failure();
116 }

126 static bool lowerCirctSizeof(InstancePathCache &instancePathCache,
127  FModuleLike mod) {
128  auto ports = mod.getPorts();
129  if (hasNPorts("circt.sizeof", mod, 2) ||
130  namedPort("circt.sizeof", mod, 0, "i") ||
131  namedPort("circt.sizeof", mod, 1, "size") ||
132  sizedPort<UIntType>("circt.sizeof", mod, 1, 32) ||
133  hasNParam("circt.sizeof", mod, 0))
134  return false;
135 
136  for (auto *use : lookupInstNode(instancePathCache, mod)->uses()) {
137  auto inst = cast<InstanceOp>(use->getInstance().getOperation());
138  ImplicitLocOpBuilder builder(inst.getLoc(), inst);
139  auto inputWire = builder.create<WireOp>(ports[0].type).getResult();
140  inst.getResult(0).replaceAllUsesWith(inputWire);
141  auto size = builder.create<SizeOfIntrinsicOp>(inputWire);
142  inst.getResult(1).replaceAllUsesWith(size);
143  inst.erase();
144  }
145  return true;
146 }

214 static bool lowerCirctClockGate(InstancePathCache &instancePathCache,
215  FModuleLike mod) {
216  if (hasNPorts("circt.clock_gate", mod, 3) ||
217  namedPort("circt.clock_gate", mod, 0, "in") ||
218  namedPort("circt.clock_gate", mod, 1, "en") ||
219  namedPort("circt.clock_gate", mod, 2, "out") ||
220  typedPort<ClockType>("circt.clock_gate", mod, 0) ||
221  sizedPort<UIntType>("circt.clock_gate", mod, 1, 1) ||
222  typedPort<ClockType>("circt.clock_gate", mod, 2) ||
223  hasNParam("circt.clock_gate", mod, 0))
224  return false;
225 
226  for (auto *use : lookupInstNode(instancePathCache, mod)->uses()) {
227  auto inst = cast<InstanceOp>(use->getInstance().getOperation());
228  ImplicitLocOpBuilder builder(inst.getLoc(), inst);
229  auto in = builder.create<WireOp>(inst.getResult(0).getType()).getResult();
230  auto en = builder.create<WireOp>(inst.getResult(1).getType()).getResult();
231  inst.getResult(0).replaceAllUsesWith(in);
232  inst.getResult(1).replaceAllUsesWith(en);
233  auto out = builder.create<ClockGateIntrinsicOp>(in, en, Value{});
234  inst.getResult(2).replaceAllUsesWith(out);
235  inst.erase();
236  }
237  return true;
238 }

254 // This is the main entrypoint for the lowering pass.
255 void LowerIntrinsicsPass::runOnOperation() {
256  size_t numFailures = 0;
257  size_t numConverted = 0;
258  InstancePathCache instancePathCache(getAnalysis<InstanceGraph>());
259  for (auto &op : llvm::make_early_inc_range(getOperation().getOps())) {
260  if (!isa<FExtModuleOp, FIntModuleOp>(op))
261  continue;
262  StringAttr intname;
263  if (isa<FExtModuleOp>(op)) {
264  auto anno = AnnotationSet(&op).getAnnotation("circt.Intrinsic");
265  if (!anno)
266  continue;
267  intname = anno.getMember<StringAttr>("intrinsic");
268  if (!intname) {
269  op.emitError("intrinsic annotation with no intrinsic name");
270  ++numFailures;
271  continue;
272  }
273  } else {
274  intname = cast<FIntModuleOp>(op).getIntrinsicAttr();
275  if (!intname) {
276  op.emitError("intrinsic module with no intrinsic name");
277  ++numFailures;
278  continue;
279  }
280  }
281 
282  bool found = false;
283  for (const auto &intrinsic : intrinsics) {
284  if (intname.getValue().equals(intrinsic.first)) {
285  found = true;
286  if (intrinsic.second(instancePathCache, cast<FModuleLike>(op))) {
287  ++numConverted;
288  op.erase();
289  } else {
290  ++numFailures;
291  }
292  break;
293  }
294  }
295  if (!found) {
296  op.emitError("unknown intrinsic: '") << intname.getValue() << "'";
297  ++numFailures;
298  }
299  }
300  if (numFailures)
301  signalPassFailure();
302  if (!numConverted)
303  markAllAnalysesPreserved();
304 }
