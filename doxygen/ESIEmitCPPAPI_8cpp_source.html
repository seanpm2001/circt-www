68 LogicalResult CPPAPI::gatherTypes() {
69  auto storeType = [&](mlir::Type type) -> LogicalResult {
70  auto dirType = esi::innerType(type);
71  auto dirTypeSchemaIt = types.find(dirType);
72  if (dirTypeSchemaIt == types.end()) {
73  CPPType dirTypeSchema(dirType);
74  if (!dirTypeSchema.isSupported())
75  return emitError(module.getLoc())
76  << "Type " << dirType << " not supported.";
77  dirTypeSchemaIt = types.insert({dirType, dirTypeSchema}).first;
78  }
79  return success();
80  };
81 
82  for (auto serviceDeclOp : module.getOps<ServiceDeclOpInterface>()) {
83  llvm::SmallVector<ServicePortInfo> ports;
84  serviceDeclOp.getPortList(ports);
85  for (auto portInfo : ports) {
86  if (portInfo.toClientType)
87  if (failed(storeType(portInfo.toClientType)))
88  return failure();
89  if (portInfo.toServerType)
90  if (failed(storeType(portInfo.toServerType)))
91  return failure();
92  }
93  }
94  return success();
95 }

97 LogicalResult CPPAPI::emit() {
98  // Walk and collect the type data.
99  if (failed(gatherTypes()))
100  return failure();
101 
102  os << "#pragma once\n\n";
103 
104  os << "// The ESI C++ API relies on the refl-cpp library for type "
105  "introspection. This must be provided by the user.\n";
106  os << "// See https://github.com/veselink1/refl-cpp \n";
107  os << "#include \"refl.hpp\"\n\n";
108 
109  os << "#include <cstdint>\n";
110  os << "#include \"esi/backends/capnp.h\"\n";
111  os << "\n// Include the generated Cap'nProto schema header. This must "
112  "defined by the build system.\n";
113  os << "#include ESI_COSIM_CAPNP_H\n";
114  os << "\n\n";
115 
116  os << "namespace esi {\n";
117  os << "namespace runtime {\n\n";
118 
119  if (failed(emitTypes()) || failed(emitServiceDeclarations()) ||
120  failed(emitDesignModules()))
121  return failure();
122 
123  os << "} // namespace runtime\n";
124  os << "} // namespace esi\n\n";
125 
126  os << "// ESI dynamic reflection support\n";
127  if (failed(emitGlobalNamespace()))
128  return failure();
129 
130  return success();

133 LogicalResult CPPAPI::emitServiceDeclarations() {
134  // Locate all of the service declarations which are needed by the
135  // services in the service hierarchy.
136  for (auto serviceDeclOp : module.getOps<ServiceDeclOpInterface>()) {
137  auto cppService = CPPService(serviceDeclOp, types);
138  if (failed(cppService.write(os)))
139  return failure();
140  cppServices.push_back(cppService);
141  }
142 
143  return success();
144 }

155 LogicalResult CPPAPI::emitTypes() {
156  os << "class ESITypes {\n";
157  os << "public:\n";
158  os.indent();
159 
160  // Iterate through the various types and emit their CPP APIs.
161  for (const auto &cppType : types) {
162  if (failed(cppType.second.write(os)))
163  // If we fail during an emission, dump out early since the output may
164  // be corrupted.
165  return failure();
166  }
167 
168  os.unindent();
169  os << "};\n\n";
170 
171  return success();
172 }

174 LogicalResult CPPAPI::emitDesignModules() {
175  // Get a list of metadata ops which originated in modules (path is empty).
176  SmallVector<
177  std::pair<hw::HWModuleLike, SmallVector<ServiceHierarchyMetadataOp, 0>>>
178  modsWithLocalServices;
179  for (auto hwmod : module.getOps<hw::HWModuleLike>()) {
180  SmallVector<ServiceHierarchyMetadataOp, 0> metadataOps;
181  hwmod.walk([&metadataOps](ServiceHierarchyMetadataOp md) {
182  if (md.getServerNamePath().empty() && md.getImplType() == "cosim")
183  metadataOps.push_back(md);
184  });
185  if (!metadataOps.empty())
186  modsWithLocalServices.push_back(std::make_pair(hwmod, metadataOps));
187  }
188 
189  SmallVector<CPPDesignModule> designMods;
190  for (auto &mod : modsWithLocalServices)
191  designMods.push_back(CPPDesignModule(mod.first, mod.second, cppServices));
192 
193  // Write modules
194  for (auto &designMod : designMods) {
195  if (failed(designMod.write(os)))
196  return failure();
197  os << "\n";
198  }
199  return success();
200 }
