30 LogicalResult PipelineOp::verify() {
31  bool anyInputIsAChannel = llvm::any_of(getInputs(), [](Value operand) {
32  return operand.getType().isa<esi::ChannelType>();
33  });
34  bool anyOutputIsAChannel = llvm::any_of(
35  getResultTypes(), [](Type type) { return type.isa<esi::ChannelType>(); });
36 
37  if ((anyInputIsAChannel || anyOutputIsAChannel) && !isLatencyInsensitive()) {
38  return emitOpError("if any port of this pipeline is an ESI channel, all "
39  "ports must be ESI channels.");
40  }
41 
42  if (getBodyBlock()->getNumArguments() != getInputs().size())
43  return emitOpError("expected ")
44  << getInputs().size()
45  << " arguments in the pipeline body block, got "
46  << getBodyBlock()->getNumArguments() << ".";
47 
48  for (size_t i = 0; i < getInputs().size(); i++) {
49  Type expectedInArg = getInputs()[i].getType();
50  Type bodyArg = getBodyBlock()->getArgument(i).getType();
51 
52  if (isLatencyInsensitive())
53  expectedInArg = expectedInArg.cast<esi::ChannelType>().getInner();
54 
55  if (expectedInArg != bodyArg)
56  return emitOpError("expected body block argument ")
57  << i << " to have type " << expectedInArg << ", got " << bodyArg
58  << ".";
59  }
60 
61  // Check mixing of `pipeline.stage` and `pipeline.stage.register` ops.
62  // This verifier thus ensures a proper phase ordering between stage ops
63  // and their materialized stage register op counterparts.
64  bool hasStageOps = !getOps<PipelineStageOp>().empty();
65  bool hasStageRegOps = !getOps<PipelineStageRegisterOp>().empty();
66 
67  if (hasStageOps && hasStageRegOps)
68  return emitOpError("mixing `pipeline.stage` and `pipeline.stage.register` "
69  "ops is illegal.");
70 
71  return success();
72 }

74 bool PipelineOp::isLatencyInsensitive() {
75  bool allInputsAreChannels = llvm::all_of(getInputs(), [](Value operand) {
76  return operand.getType().isa<esi::ChannelType>();
77  });
78  bool allOutputsAreChannels = llvm::all_of(
79  getResultTypes(), [](Type type) { return type.isa<esi::ChannelType>(); });
80  return allInputsAreChannels && allOutputsAreChannels;
81 }

87 LogicalResult ReturnOp::verify() {
88  PipelineOp parent = getOperation()->getParentOfType<PipelineOp>();
89  if (getOutputs().size() != parent.getResults().size())
90  return emitOpError("expected ")
91  << parent.getResults().size() << " return values, got "
92  << getOutputs().size() << ".";
93 
94  bool isLatencyInsensitive = parent.isLatencyInsensitive();
95  for (size_t i = 0; i < parent.getResults().size(); i++) {
96  Type expectedType = parent.getResultTypes()[i];
97  Type actualType = getOperandTypes()[i];
98  if (isLatencyInsensitive)
99  expectedType = expectedType.cast<esi::ChannelType>().getInner();
100  if (expectedType != actualType)
101  return emitOpError("expected argument ")
102  << i << " to have type " << expectedType << ", got " << actualType
103  << ".";
104  }
105 
106  return success();
107 }
