45 // Replaces an output op with a new output with flattened (exploded) structs.
46 struct OutputOpConversion : public OpConversionPattern<hw::OutputOp> {
47  OutputOpConversion(TypeConverter &typeConverter, MLIRContext *context,
48  DenseSet<Operation *> *opVisited)
49  : OpConversionPattern(typeConverter, context), opVisited(opVisited) {}
50 
51  LogicalResult
52  matchAndRewrite(hw::OutputOp op, OpAdaptor adaptor,
53  ConversionPatternRewriter &rewriter) const override {
54  llvm::SmallVector<Value> convOperands;
55 
56  // Flatten the operands.
57  for (auto operand : adaptor.getOperands()) {
58  if (auto structType = getStructType(operand.getType())) {
59  auto explodedStruct = rewriter.create<hw::StructExplodeOp>(
60  op.getLoc(), getInnerTypes(structType), operand);
61  llvm::copy(explodedStruct.getResults(),
62  std::back_inserter(convOperands));
63  } else
64  convOperands.push_back(operand);
65  }
66 
67  // And replace.
68  rewriter.replaceOpWithNewOp<hw::OutputOp>(op, convOperands);
69  opVisited->insert(op->getParentOp());
70  return success();
71  }
72  mutable DenseSet<Operation *> *opVisited;
73 };

130 template <typename... TOp>
131 static void addSignatureConversion(DenseMap<Operation *, IOInfo> &ioMap,
132  ConversionTarget &target,
133  RewritePatternSet &patterns,
134  FlattenIOTypeConverter &typeConverter) {
135  (mlir::populateFunctionOpInterfaceTypeConversionPattern<TOp>(patterns,
136  typeConverter),
137  ...);
138 
139  // Legality is defined by a module having been processed once. This is due to
140  // that a pattern cannot be applied multiple times (a 'pattern was already
141  // applied' error - a case that would occur for nested structs). Additionally,
142  // if a pattern could be applied multiple times, this would complicate
143  // updating arg/res names.
144 
145  // Instead, we define legality as when a module has had a modification to its
146  // top-level i/o. This ensures that only a single level of structs are
147  // processed during signature conversion, which then allows us to use the
148  // signature conversion in a recursive manner.
149  target.addDynamicallyLegalOp<TOp...>([&](FunctionOpInterface moduleLikeOp) {
150  if (isLegalFuncLikeOp(moduleLikeOp))
151  return true;
152 
153  // This op is involved in conversion. Check if the signature has changed.
154  auto ioInfoIt = ioMap.find(moduleLikeOp);
155  if (ioInfoIt == ioMap.end()) {
156  // Op wasn't primed in the map. Do the safe thing, assume
157  // that it's not considered in this pass, and mark it as legal
158  return true;
159  }
160  auto ioInfo = ioInfoIt->second;
161 
162  auto compareTypes = [&](TypeRange oldTypes, TypeRange newTypes) {
163  return llvm::any_of(llvm::zip(oldTypes, newTypes), [&](auto typePair) {
164  auto oldType = std::get<0>(typePair);
165  auto newType = std::get<1>(typePair);
166  return oldType != newType;
167  });
168  };
169  if (compareTypes(moduleLikeOp.getResultTypes(), ioInfo.resTypes) ||
170  compareTypes(moduleLikeOp.getArgumentTypes(), ioInfo.argTypes))
171  return true;
172 
173  // We're pre-conversion for an op that was primed in the map - it will
174  // always be illegal since it has to-be-converted struct types at its I/O.
175  return false;
176  });
177 }

193 static void updateNameAttribute(FunctionOpInterface op, StringRef attrName,
194  DenseMap<unsigned, hw::StructType> &structMap) {
195  llvm::SmallVector<Attribute> newNames;
196  auto oldNames = op.getOperation()
197  ->getAttrOfType<ArrayAttr>(attrName)
198  .getAsValueRange<StringAttr>();
199  for (auto [i, oldName] : llvm::enumerate(oldNames)) {
200  // Was this arg/res index a struct?
201  auto it = structMap.find(i);
202  if (it == structMap.end()) {
203  // No, keep old name.
204  newNames.push_back(StringAttr::get(op.getContext(), oldName));
205  continue;
206  }
207 
208  // Yes - create new names from the struct fields and the old name at the
209  // index.
210  auto structType = it->second;
211  for (auto field : structType.getElements())
212  newNames.push_back(
213  StringAttr::get(op.getContext(), oldName + "." + field.name.str()));
214  }
215  op.getOperation()->setAttr(attrName,
216  ArrayAttr::get(op.getContext(), newNames));
217 }

219 static void updateLocAttribute(FunctionOpInterface op, StringRef attrName,
220  DenseMap<unsigned, hw::StructType> &structMap) {
221  llvm::SmallVector<Attribute> newLocs;
222  auto oldLocs = op.getOperation()->getAttrOfType<ArrayAttr>(attrName);
223  if (!oldLocs)
224  return;
225  for (auto [i, oldLoc] : llvm::enumerate(oldLocs)) {
226  // Was this arg/res index a struct?
227  auto it = structMap.find(i);
228  if (it == structMap.end()) {
229  // No, keep old name.
230  newLocs.push_back(oldLoc);
231  continue;
232  }
233 
234  auto structType = it->second;
235  for (size_t i = 0, e = structType.getElements().size(); i < e; ++i)
236  newLocs.push_back(oldLoc);
237  }
238  op.getOperation()->setAttr(attrName,
239  ArrayAttr::get(op.getContext(), newLocs));
240 }

242 /// The conversion framework seems to throw away block argument locations. We

256 template <typename T>
257 static DenseMap<Operation *, IOInfo> populateIOInfoMap(mlir::ModuleOp module) {
258  DenseMap<Operation *, IOInfo> ioInfoMap;
259  for (auto op : module.getOps<T>()) {
260  IOInfo ioInfo;
261  ioInfo.argTypes = op.getArgumentTypes();
262  ioInfo.resTypes = op.getResultTypes();
263  for (auto [i, arg] : llvm::enumerate(ioInfo.argTypes)) {
264  if (auto structType = getStructType(arg))
265  ioInfo.argStructs[i] = structType;
266  }
267  for (auto [i, res] : llvm::enumerate(ioInfo.resTypes)) {
268  if (auto structType = getStructType(res))
269  ioInfo.resStructs[i] = structType;
270  }
271  ioInfoMap[op] = ioInfo;
272  }
273  return ioInfoMap;
274 }

276 template <typename T>
277 static LogicalResult flattenOpsOfType(ModuleOp module, bool recursive) {
278  auto *ctx = module.getContext();
279  FlattenIOTypeConverter typeConverter;
280 
281  // Recursively (in case of nested structs) lower the module. We do this one
282  // conversion at a time to allow for updating the arg/res names of the
283  // module in between flattening each level of structs.
284  while (hasUnconvertedOps<T>(module)) {
285  ConversionTarget target(*ctx);
286  RewritePatternSet patterns(ctx);
287  target.addLegalDialect<hw::HWDialect>();
288 
289  // Record any struct types at the module signature. This will be used
290  // post-conversion to update the argument and result names.
291  auto ioInfoMap = populateIOInfoMap<T>(module);
292 
293  // Signature conversion and legalization patterns.
294  addSignatureConversion<T>(ioInfoMap, target, patterns, typeConverter);
295 
296  // Argument conversion for output ops. Similarly to the signature
297  // conversion, legality is based on the op having been visited once, due to
298  // the possibility of nested structs.
299  DenseSet<Operation *> opVisited;
300  patterns.add<OutputOpConversion>(typeConverter, ctx, &opVisited);
301  // patterns.add<StructExplodeOpConversion>(typeConverter, ctx);
302  target.addDynamicallyLegalOp<hw::OutputOp>(
303  [&](auto op) { return opVisited.contains(op->getParentOp()); });
304 
305  if (failed(applyPartialConversion(module, target, std::move(patterns))))
306  return failure();
307 
308  // Update the arg/res names of the module.
309  for (auto op : module.getOps<T>()) {
310  auto ioInfo = ioInfoMap[op];
311  updateNameAttribute(op, "argNames", ioInfo.argStructs);
312  updateNameAttribute(op, "resultNames", ioInfo.resStructs);
313  updateLocAttribute(op, "argLocs", ioInfo.argStructs);
314  updateLocAttribute(op, "resultLocs", ioInfo.resStructs);
315  updateBlockLocations(op, "argLocs", ioInfo.argStructs);
316  }
317 
318  // Break if we've only lowering a single level of structs.
319  if (!recursive)
320  break;
321  }
322  return success();
323 }
