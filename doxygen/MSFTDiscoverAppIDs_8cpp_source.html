43 void DiscoverAppIDsPass::runOnOperation() {
44  ModuleOp topMod = getOperation();
45  topLevelSyms.addDefinitions(topMod);
46  if (failed(verifyInstances(topMod))) {
47  signalPassFailure();
48  return;
49  }
50 
51  // Sort modules in partial order be use. Enables single-pass processing.
52  SmallVector<MSFTModuleOp> sortedMods;
53  getAndSortModules(topMod, sortedMods);
54 
55  for (MSFTModuleOp mod : sortedMods)
56  processMod(mod);
57 }

59 /// Find the AppIDs in a given module.
60 void DiscoverAppIDsPass::processMod(MSFTModuleOp mod) {
61  SmallDenseMap<StringAttr, uint64_t> appBaseCounts;
62  SmallPtrSet<StringAttr, 32> localAppIDBases;
63  SmallDenseMap<AppIDAttr, Operation *> localAppIDs;
64 
65  mod.walk([&](Operation *op) {
66  // If an operation has an "appid" dialect attribute, it is considered a
67  // "local" appid.
68  if (auto appid = op->getAttrOfType<AppIDAttr>("msft.appid")) {
69  if (localAppIDs.find(appid) != localAppIDs.end()) {
70  op->emitOpError("Found multiple identical AppIDs in same module")
71  .attachNote(localAppIDs[appid]->getLoc())
72  << "first AppID located here";
73  signalPassFailure();
74  } else {
75  localAppIDs[appid] = op;
76  }
77  localAppIDBases.insert(appid.getName());
78  }
79 
80  // Instance ops should expose their module's AppIDs recursively. Track the
81  // number of instances which contain a base name.
82  if (auto inst = dyn_cast<InstanceOp>(op)) {
83  auto targetMod = dyn_cast<MSFTModuleOp>(
84  topLevelSyms.getDefinition(inst.getModuleNameAttr()));
85  if (targetMod && targetMod.getChildAppIDBases())
86  for (auto base :
87  targetMod.getChildAppIDBasesAttr().getAsRange<StringAttr>())
88  appBaseCounts[base] += 1;
89  }
90  });
91 
92  // Collect the list of AppID base names with which to annotate 'mod'.
93  SmallVector<Attribute, 32> finalModBases;
94  for (auto baseCount : appBaseCounts) {
95  // If multiple instances expose the same base name, don't expose them
96  // through this module. If any of the instances expose basenames which are
97  // exposed locally, also don't expose them up.
98  if (baseCount.getSecond() == 1 &&
99  !localAppIDBases.contains(baseCount.getFirst()))
100  finalModBases.push_back(baseCount.getFirst());
101  }
102 
103  // Add all of the local base names.
104  for (StringAttr lclBase : localAppIDBases)
105  finalModBases.push_back(lclBase);
106 
107  if (finalModBases.empty())
108  return;
109  ArrayAttr childrenBases = ArrayAttr::get(mod.getContext(), finalModBases);
110  mod.setChildAppIDBasesAttr(childrenBases);
111 }
