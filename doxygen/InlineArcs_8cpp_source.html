53 void InlineArcsPass::runOnOperation() {
54  auto module = getOperation();
55 
56  // Store the call hierarcy manually since we need to keep it updated and the
57  // provided datastructures do not seem to support this
58  DenseMap<DefineOp, DenseSet<DefineOp>> arcsCalledInBody;
59  DenseMap<DefineOp, DenseSet<mlir::CallOpInterface>> usersPerArc;
60 
61  SymbolTableCollection symbolTable;
62  ArcInliner inliner(&getContext());
63 
64  // Compute Arc call hierarchy
65  module->walk([&](Operation *op) {
66  if (auto defOp = dyn_cast<DefineOp>(op)) {
67  arcsCalledInBody.insert({defOp, {}});
68  usersPerArc.insert({defOp, {}});
69  }
70 
71  if (auto callOp = dyn_cast<mlir::CallOpInterface>(op)) {
72  if (auto defOp =
73  dyn_cast<DefineOp>(callOp.resolveCallable(&symbolTable))) {
74  usersPerArc[defOp].insert(callOp);
75  if (auto parentOp = op->getParentOfType<DefineOp>())
76  arcsCalledInBody[parentOp].insert(defOp);
77  }
78  }
79  });
80 
81  // Remove all unused arcs
82  SmallVector<DefineOp> removeList(module.getOps<DefineOp>());
83  while (!removeList.empty()) {
84  auto defOp = removeList.pop_back_val();
85  if (!defOp)
86  continue;
87 
88  if (usersPerArc[defOp].empty()) {
89  usersPerArc.erase(defOp);
90  for (auto callee : arcsCalledInBody[defOp]) {
91  defOp->walk([&](mlir::CallOpInterface call) {
92  usersPerArc[callee].erase(call);
93  });
94  if (usersPerArc[callee].empty())
95  removeList.push_back(callee);
96  }
97  arcsCalledInBody.erase(defOp);
98  defOp.erase();
99  ++numRemovedArcs;
100  }
101  }
102 
103  // It's a bit ugly that we have an intermediate datastructure here, but some
104  // attempts in using a datastructure with deterministic iteration over
105  // arcsCalledInBody have shown worse performance.
106  SmallVector<DefineOp> arcsWithoutFurtherCalls;
107  for (const auto &[defOp, list] : arcsCalledInBody)
108  if (defOp && list.empty())
109  arcsWithoutFurtherCalls.push_back(defOp);
110 
111  // We need to sort here because we iterated over a hashmap above.
112  std::sort(arcsWithoutFurtherCalls.begin(), arcsWithoutFurtherCalls.end());
113  SetVector<DefineOp> worklist(arcsWithoutFurtherCalls.begin(),
114  arcsWithoutFurtherCalls.end());
115 
116  while (!worklist.empty()) {
117  auto defOp = worklist.pop_back_val();
118 
119  // TODO: copying here and sorting is not ideal, we should use another
120  // datastructure than a DenseSet. We need to unique the initial set of
121  // elements, deterministic iteration and fast element deletion, so maybe a
122  // linked list?
123  SmallVector<mlir::CallOpInterface> users(usersPerArc[defOp].begin(),
124  usersPerArc[defOp].end());
125  std::sort(users.begin(), users.end());
126 
127  // Update the worklist
128  for (auto caller : users) {
129  if (auto parentOp = caller->getParentOfType<DefineOp>()) {
130  arcsCalledInBody[parentOp].erase(defOp);
131  if (arcsCalledInBody[parentOp].empty())
132  worklist.insert(parentOp);
133  }
134  }
135 
136  // Check if we should inline the arc.
137  if (!shouldInline(defOp, users))
138  continue;
139 
140  LLVM_DEBUG(llvm::dbgs() << "Inlining " << defOp.getSymName() << "\n");
141 
142  // Inline all uses of the arc. Currently we inline all of them but in the
143  // future we may decide per use site whether to inline or not.
144  unsigned numUsersLeft = users.size();
145  for (auto user : users) {
146  if (!user->getParentOfType<DefineOp>() && intoArcsOnly)
147  continue;
148 
149  // Recursive calls of arcs are not allowed, but since we cannot verify
150  // that in the op verifier, we need to check check here just to be sure
151  if (defOp == user->getParentOfType<DefineOp>())
152  continue;
153 
154  if (succeeded(mlir::inlineCall(inliner, user, defOp, &defOp.getBody(),
155  numUsersLeft > 1))) {
156  usersPerArc[defOp].erase(user);
157  user.erase();
158  --numUsersLeft;
159  ++numInlinedArcs;
160  }
161 
162  if (numUsersLeft == 0) {
163  arcsCalledInBody.erase(defOp);
164  usersPerArc.erase(defOp);
165  defOp.erase();
166  ++numRemovedArcs;
167  }
168  }
169  }
170 }

172 bool InlineArcsPass::shouldInline(DefineOp defOp,
173  ArrayRef<mlir::CallOpInterface> users) {
174  // Count the number of non-trivial ops in the arc. If there are only a few,
175  // inline the arc.
176  unsigned numNonTrivialOps = 0;
177  defOp.getBodyBlock().walk([&](Operation *op) {
178  if (!op->hasTrait<OpTrait::ConstantLike>() && !isa<OutputOp>(op))
179  ++numNonTrivialOps;
180  });
181  if (numNonTrivialOps <= maxNonTrivialOpsInBody) {
182  ++numTrivialArcs;
183  return true;
184  }
185  LLVM_DEBUG(llvm::dbgs() << "Arc " << defOp.getSymName() << " has "
186  << numNonTrivialOps << " non-trivial ops\n");
187 
188  // Check if the arc is only ever used once.
189  if (users.size() == 1) {
190  ++numSingleUseArcs;
191  return true;
192  }
193 
194  return false;
195 }
