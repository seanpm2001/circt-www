35 
36  llvm::SmallVector<OpAsmParser::UnresolvedOperand, 4> operands;
37  if (parser.parseOperandList(operands, /*requiredOperandCount=*/3,
38  /*delimiter=*/OpAsmParser::Delimiter::None))
39  return failure();
40 
41  Type innerOutputType;
42  if (parser.parseOptionalAttrDict(result.attributes) || parser.parseColon() ||
43  parser.parseType(innerOutputType))
44  return failure();
45  auto outputType =
46  ChannelType::get(parser.getBuilder().getContext(), innerOutputType);
47  result.addTypes({outputType});
48 
49  auto i1 = IntegerType::get(result.getContext(), 1);
50  if (parser.resolveOperands(operands, {i1, i1, outputType}, inputOperandsLoc,
51  result.operands))
52  return failure();
53  return success();
54 }

244 static Type getEsiDataType(circt::sv::InterfaceOp iface) {
245  using namespace circt::sv;
246  if (!iface.lookupSymbol<InterfaceSignalOp>("valid"))
247  return Type();

256 /// Verify that the modport type of 'modportArg' points to an interface which
257 /// looks like an ESI interface and the inner data from said interface matches
258 /// the chan type's inner data type.
259 static LogicalResult verifySVInterface(Operation *op,
260  circt::sv::ModportType modportType,
261  ChannelType chanType) {
262  auto modport =
263  SymbolTable::lookupNearestSymbolFrom<circt::sv::InterfaceModportOp>(
264  op, modportType.getModport());
265  if (!modport)
266  return op->emitError("Could not find modport ")
267  << modportType.getModport() << " in symbol table.";
268  auto iface = cast<circt::sv::InterfaceOp>(modport->getParentOp());
269  Type esiDataType = getEsiDataType(iface);
270  if (!esiDataType)
271  return op->emitOpError("Interface is not a valid ESI interface.");
272  if (esiDataType != chanType.getInner())
273  return op->emitOpError("Operation specifies ")
274  << chanType << " but type inside doesn't match interface data type "
275  << esiDataType << ".";
276  return success();
277 }

388 /// Validate a connection request against a service decl by comparing against
389 /// the port list.
390 template <class OpType>
391 LogicalResult validateRequest(ServiceDeclOpInterface svc, OpType req) {
392  ServicePortInfo portDecl;
393  SmallVector<ServicePortInfo> ports;
394  svc.getPortList(ports);
395  for (ServicePortInfo portFromList : ports)
396  if (portFromList.name == req.getServicePort().getName()) {
397  portDecl = portFromList;
398  break;
399  }
400  if (!portDecl.name)
401  return req.emitOpError("Could not locate port ")
402  << req.getServicePort().getName();
403 
404  auto *ctxt = req.getContext();
405  auto anyChannelType = ChannelType::get(ctxt, AnyType::get(ctxt));
406  auto [toServerType, toClientType] = getToServerToClientTypes(req);
407 
408  // TODO: Because `inout` requests get broken in two pretty early on, we can't
409  // tell if a to_client/to_server request was initially part of an inout
410  // request, so we can't check that an inout port is only accessed by an inout
411  // request. Consider a different way to do this.
412 
413  // Check the input port type.
414  if (!isa<RequestToClientConnectionOp>(req) &&
415  portDecl.toServerType != toServerType &&
416  portDecl.toServerType != anyChannelType)
417  return req.emitOpError("Request to_server type does not match port type ")
418  << portDecl.toServerType;
419 
420  // Check the output port type.
421  if (!isa<RequestToServerConnectionOp>(req) &&
422  portDecl.toClientType != toClientType &&
423  portDecl.toClientType != anyChannelType)
424  return req.emitOpError("Request to_client type does not match port type ")
425  << portDecl.toClientType;
426  return success();
427 }

467 void ServiceImplementReqOp::gatherPairedReqs(
468  llvm::SmallVectorImpl<std::pair<RequestToServerConnectionOp,
469  RequestToClientConnectionOp>> &reqPairs) {
470 
471  // Build a mapping of client path names to requests.
472  DenseMap<std::pair<hw::InnerRefAttr, ArrayAttr>,
473  SmallVector<RequestToServerConnectionOp, 0>>
474  clientNameToServer;
475  DenseMap<std::pair<hw::InnerRefAttr, ArrayAttr>,
476  SmallVector<RequestToClientConnectionOp, 0>>
477  clientNameToClient;
478  for (auto &op : getOps())
479  if (auto req = dyn_cast<RequestToClientConnectionOp>(op))
480  clientNameToClient[std::make_pair(req.getServicePort(),
481  req.getClientNamePathAttr())]
482  .push_back(req);
483  else if (auto req = dyn_cast<RequestToServerConnectionOp>(op))
484  clientNameToServer[std::make_pair(req.getServicePort(),
485  req.getClientNamePathAttr())]
486  .push_back(req);
487 
488  // Find all of the pairs and emit them.
489  DenseSet<Operation *> emittedOps;
490  for (auto op : getOps<RequestToServerConnectionOp>()) {
491  std::pair<hw::InnerRefAttr, ArrayAttr> clientName =
492  std::make_pair(op.getServicePort(), op.getClientNamePathAttr());
493  const SmallVector<RequestToServerConnectionOp, 0> &ops =
494  clientNameToServer[clientName];
495 
496  // Only emit a pair if there's one toServer and one toClient request for a
497  // given client name path.
498  if (ops.size() == 1) {
499  auto toClientF = clientNameToClient.find(clientName);
500  if (toClientF != clientNameToClient.end() &&
501  toClientF->second.size() == 1) {
502  reqPairs.push_back(
503  std::make_pair(ops.front(), toClientF->second.front()));
504  emittedOps.insert(ops.front());
505  emittedOps.insert(toClientF->second.front());
506  continue;
507  }
508  }
509  }
510 
511  // Emit partial pairs for all the remaining requests.
512  for (auto &op : getOps()) {
513  if (emittedOps.contains(&op))
514  continue;
515  if (auto req = dyn_cast<RequestToClientConnectionOp>(op))
516  reqPairs.push_back(std::make_pair(nullptr, req));
517  else if (auto req = dyn_cast<RequestToServerConnectionOp>(op))
518  reqPairs.push_back(std::make_pair(req, nullptr));
519  }
520 }

526 LogicalResult ESIPureModuleOp::verify() {
527  ESIDialect *esiDialect = getContext()->getLoadedDialect<ESIDialect>();
528  Block &body = getBody().front();
529  auto channelOrOutput = [](Value v) {
530  if (v.getType().isa<ChannelType>())
531  return true;
532  if (v.getUsers().empty())
533  return false;
534  return llvm::all_of(v.getUsers(), [](Operation *op) {
535  return isa<ESIPureModuleOutputOp>(op);
536  });
537  };
538 
539  DenseMap<StringAttr, std::tuple<hw::PortDirection, Type, Operation *>> ports;
540  for (Operation &op : body.getOperations()) {
541  if (hw::HWInstanceLike inst = dyn_cast<hw::HWInstanceLike>(op)) {
542  if (llvm::any_of(op.getOperands(), [](Value v) {
543  return !(v.getType().isa<ChannelType>() ||
544  isa<ESIPureModuleInputOp>(v.getDefiningOp()));
545  }))
546  return inst.emitOpError(
547  "instances in ESI pure modules can only contain channel ports or "
548  "ports driven by 'input' ops");
549  if (!llvm::all_of(op.getResults(), channelOrOutput))
550  return inst.emitOpError(
551  "instances in ESI pure modules can only contain channel ports or "
552  "drive only 'outputs'");
553  } else {
554  // Pure modules can only contain instance ops and ESI ops.
555  if (op.getDialect() != esiDialect)
556  return op.emitOpError("operation not allowed in ESI pure modules");
557  }
558 
559  // Check for port validity.
560  if (auto port = dyn_cast<ESIPureModuleInputOp>(op)) {
561  auto existing = ports.find(port.getNameAttr());
562  Type portType = port.getResult().getType();
563  if (existing != ports.end()) {
564  auto [dir, type, op] = existing->getSecond();
565  if (dir != hw::PortDirection::INPUT || type != portType)
566  return (port.emitOpError("port '")
567  << port.getName() << "' previously declared as type " << type)
568  .attachNote(op->getLoc());
569  }
570  ports[port.getNameAttr()] = std::make_tuple(
571  hw::PortDirection::INPUT, portType, port.getOperation());
572  } else if (auto port = dyn_cast<ESIPureModuleOutputOp>(op)) {
573  auto existing = ports.find(port.getNameAttr());
574  if (existing != ports.end())
575  return (port.emitOpError("port '")
576  << port.getName() << "' previously declared")
577  .attachNote(std::get<2>(existing->getSecond())->getLoc());
578  ports[port.getNameAttr()] =
579  std::make_tuple(hw::PortDirection::INPUT, port.getValue().getType(),
580  port.getOperation());
581  }
582  }
583  return success();
584 }
