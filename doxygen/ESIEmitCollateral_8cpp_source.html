35 static llvm::json::Value toJSON(Type type) {
36  // TODO: This is far from complete. Build out as necessary.
37  using llvm::json::Array;
38  using llvm::json::Object;
39  using llvm::json::Value;
40 
41  StringRef dialect = type.getDialect().getNamespace();
42  std::string m;
43  Object o = TypeSwitch<Type, Object>(type)
44  .Case([&](ChannelType t) {
45  m = "channel";
46  return Object({{"inner", toJSON(t.getInner())}});
47  })
48  .Case([&](AnyType t) {
49  m = "any";
50  return Object();
51  })
52  .Case([&](StructType t) {
53  m = "struct";
54  Array fields;
55  for (auto field : t.getElements())
56  fields.push_back(Object({{"name", field.name.getValue()},
57  {"type", toJSON(field.type)}}));
58  return Object({{"fields", Value(std::move(fields))}});
59  })
60  .Default([&](Type t) {
61  llvm::raw_string_ostream(m) << t;
62  return Object();
63  });
64  o["dialect"] = dialect;
65  if (m.length())
66  o["mnemonic"] = m;
67  return o;
68 }

70 // Serialize an attribute to a JSON value.
71 static llvm::json::Value toJSON(Attribute attr) {
72  // TODO: This is far from complete. Build out as necessary.
73  using llvm::json::Value;
74  return TypeSwitch<Attribute, Value>(attr)
75  .Case([&](StringAttr a) { return a.getValue(); })
76  .Case([&](IntegerAttr a) { return a.getValue().getLimitedValue(); })
77  .Case([&](TypeAttr a) {
78  Type t = a.getValue();
79  llvm::json::Object typeMD;
80  typeMD["type_desc"] = toJSON(t);
81 
82  std::string buf;
83  llvm::raw_string_ostream(buf) << t;
84  typeMD["mlir_name"] = buf;
85 
86  if (auto chanType = t.dyn_cast<ChannelType>()) {
87  Type inner = chanType.getInner();
88  typeMD["hw_bitwidth"] = hw::getBitWidth(inner);
89  ESIAPIType cosimSchema(inner);
90  typeMD["capnp_type_id"] = cosimSchema.typeID();
91  typeMD["capnp_name"] = cosimSchema.name().str();
92  } else {
93  typeMD["hw_bitwidth"] = hw::getBitWidth(t);
94  }
95  return typeMD;
96  })
97  .Case([&](ArrayAttr a) {
98  return llvm::json::Array(
99  llvm::map_range(a, [](Attribute a) { return toJSON(a); }));
100  })
101  .Case([&](DictionaryAttr a) {
102  llvm::json::Object dict;
103  for (const auto &entry : a.getValue())
104  dict[entry.getName().getValue()] = toJSON(entry.getValue());
105  return dict;
106  })
107  .Case([&](InnerRefAttr ref) {
108  llvm::json::Object dict;
109  dict["outer_sym"] = ref.getModule().getValue();
110  dict["inner"] = ref.getName().getValue();
111  return dict;
112  })
113  .Default([&](Attribute a) {
114  std::string buff;
115  llvm::raw_string_ostream(buff) << a;
116  return buff;
117  });
118 }

131 void ESIEmitCollateralPass::emitServiceJSON() {
132  ModuleOp mod = getOperation();
133  auto *ctxt = &getContext();
134  SymbolCache topSyms;
135  topSyms.addDefinitions(mod);
136 
137  // Check for invalid top names.
138  for (StringRef topModName : tops)
139  if (topSyms.getDefinition(FlatSymbolRefAttr::get(ctxt, topModName)) ==
140  nullptr) {
141  mod.emitError("Could not find module named '") << topModName << "'\n";
142  signalPassFailure();
143  return;
144  }
145 
146  std::string jsonStrBuffer;
147  llvm::raw_string_ostream os(jsonStrBuffer);
148  llvm::json::OStream j(os, 2);
149 
150  // Emit the list of ports of a service declaration.
151  auto emitPorts = [&](ServiceDeclOpInterface decl) {
152  SmallVector<ServicePortInfo> ports;
153  decl.getPortList(ports);
154  for (ServicePortInfo port : ports) {
155  j.object([&] {
156  j.attribute("name", port.name.getValue());
157  if (port.toClientType)
158  j.attribute("to-client-type", toJSON(port.toClientType));
159  if (port.toServerType)
160  j.attribute("to-server-type", toJSON(port.toServerType));
161  });
162  }
163  };
164 
165  j.object([&] {
166  // Emit a list of the service declarations in a design.
167  j.attributeArray("declarations", [&] {
168  for (auto *op : llvm::make_pointer_range(mod.getOps())) {
169  if (auto decl = dyn_cast<ServiceDeclOpInterface>(op)) {
170  j.object([&] {
171  j.attribute("name", SymbolTable::getSymbolName(op).getValue());
172  j.attributeArray("ports", [&] { emitPorts(decl); });
173  });
174  }
175  }
176  });
177 
178  j.attributeArray("top_levels", [&] {
179  for (auto topModName : tops) {
180  j.object([&] {
181  auto sym = FlatSymbolRefAttr::get(ctxt, topModName);
182  Operation *hwMod = topSyms.getDefinition(sym);
183  j.attribute("module", toJSON(sym));
184  j.attributeArray("services", [&] {
185  hwMod->walk([&](ServiceHierarchyMetadataOp md) {
186  j.object([&] {
187  j.attribute("service", md.getServiceSymbol());
188  j.attribute("instance_path",
189  toJSON(md.getServerNamePathAttr()));
190  });
191  });
192  });
193  });
194  }
195  });
196 
197  // Get a list of metadata ops which originated in modules (path is empty).
198  SmallVector<
199  std::pair<hw::HWModuleLike, SmallVector<ServiceHierarchyMetadataOp, 0>>>
200  modsWithLocalServices;
201  for (auto hwmod : mod.getOps<hw::HWModuleLike>()) {
202  SmallVector<ServiceHierarchyMetadataOp, 0> metadataOps;
203  hwmod.walk([&metadataOps](ServiceHierarchyMetadataOp md) {
204  if (md.getServerNamePath().empty())
205  metadataOps.push_back(md);
206  });
207  if (!metadataOps.empty())
208  modsWithLocalServices.push_back(std::make_pair(hwmod, metadataOps));
209  }
210 
211  // Then output metadata for those modules exclusively.
212  j.attributeArray("modules", [&] {
213  for (auto &nameMdPair : modsWithLocalServices) {
214  hw::HWModuleLike hwmod = nameMdPair.first;
215  auto &mdOps = nameMdPair.second;
216  j.object([&] {
217  j.attribute("symbol", hwmod.getModuleName());
218  j.attributeArray("services", [&] {
219  for (ServiceHierarchyMetadataOp metadata : mdOps) {
220  j.object([&] {
221  j.attribute("service", metadata.getServiceSymbol());
222  j.attribute("impl_type", metadata.getImplType());
223  if (metadata.getImplDetailsAttr())
224  j.attribute("impl_details",
225  toJSON(metadata.getImplDetailsAttr()));
226  j.attributeArray("clients", [&] {
227  for (auto client : metadata.getClients())
228  j.value(toJSON(client));
229  });
230  });
231  }
232  });
233  });
234  }
235  });
236  });
237 
238  j.flush();
239  OpBuilder b = OpBuilder::atBlockEnd(mod.getBody());
240  auto verbatim = b.create<sv::VerbatimOp>(b.getUnknownLoc(),
241  StringAttr::get(ctxt, os.str()));
242  auto outputFileAttr = OutputFileAttr::getFromFilename(ctxt, "services.json");
243  verbatim->setAttr("output_file", outputFileAttr);
244 }

246 void ESIEmitCollateralPass::runOnOperation() {
247  ModuleOp mod = getOperation();
248  auto *ctxt = &getContext();
249 
250  emitServiceJSON();
251 
252  // Check for cosim endpoints in the design. If the design doesn't have any
253  // we don't need a schema.
254  WalkResult cosimWalk =
255  mod.walk([](CosimEndpointOp _) { return WalkResult::interrupt(); });
256  if (!cosimWalk.wasInterrupted())
257  return;
258 
259  // Generate the schema
260  std::string schemaStrBuffer;
261  llvm::raw_string_ostream os(schemaStrBuffer);
262  if (failed(exportCosimSchema(mod, os))) {
263  signalPassFailure();
264  return;
265  }
266 
267  // And stuff if in a verbatim op with a filename, optionally.
268  OpBuilder b = OpBuilder::atBlockEnd(mod.getBody());
269  auto verbatim = b.create<sv::VerbatimOp>(b.getUnknownLoc(),
270  StringAttr::get(ctxt, os.str()));
271  if (!schemaFile.empty()) {
272  auto outputFileAttr = OutputFileAttr::getFromFilename(ctxt, schemaFile);
273  verbatim->setAttr("output_file", outputFileAttr);
274  }
275 }
