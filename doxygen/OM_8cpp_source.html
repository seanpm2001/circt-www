52 /// Use the Evaluator to Instantiate an Object from its class name and actual
53 /// parameters.
54 OMObject omEvaluatorInstantiate(OMEvaluator evaluator, MlirAttribute className,
55  intptr_t nActualParams,
56  MlirAttribute const *actualParams) {
57  // Unwrap the Evaluator.
58  Evaluator *cppEvaluator = unwrap(evaluator);
59 
60  // Unwrap the className, which the client must supply as a StringAttr.
61  StringAttr cppClassName = unwrap(className).cast<StringAttr>();
62 
63  // Unwrap the actual parameters, which the client must supply as Attributes.
64  SmallVector<Attribute> actualParamsTmp;
65  SmallVector<ObjectValue> cppActualParams(
66  unwrapList(nActualParams, actualParams, actualParamsTmp));
67 
68  // Invoke the Evaluator to instantiate the Object.
69  FailureOr<std::shared_ptr<Object>> result =
70  cppEvaluator->instantiate(cppClassName, cppActualParams);
71 
72  // If instantiation failed, return a null Object. A Diagnostic will be emitted
73  // in this case.
74  if (failed(result))
75  return OMObject();
76 
77  // Wrap and return a *new* shared pointer to the Object, to ensure the
78  // reference count is kept up to date.
79  return wrap(new std::shared_ptr<Object>(result.value()));
80 }

103 /// Get a field from an Object, which must contain a field of that name.
104 OMObjectValue omEvaluatorObjectGetField(OMObject object, MlirAttribute name) {
105  // Unwrap the Object and get the field of the name, which the client must
106  // supply as a StringAttr.
107  FailureOr<ObjectValue> result =
108  (*unwrap(object))->getField(unwrap(name).cast<StringAttr>());
109 
110  // If getField failed, return a null ObjectValue. A Diagnostic will be emitted
111  // in this case.
112  if (failed(result))
113  return OMObjectValue();
114 
115  // If the field is an Object, return an ObjectValue with the Object set.
116  if (auto *object = std::get_if<std::shared_ptr<Object>>(&result.value()))
117  return OMObjectValue{MlirAttribute(), wrap(object)};
118 
119  // If the field is an Attribute, return an ObjectValue with the Primitive set.
120  if (auto *primitive = std::get_if<Attribute>(&result.value()))
121  return OMObjectValue{wrap(*primitive), OMObject()};
122 
123  // This case should never be hit, but return a null ObjectValue that is
124  // neither an Object nor a Primitive.
125  return OMObjectValue();
126 }
