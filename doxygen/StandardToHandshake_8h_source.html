32 /// Strategy class to control the behavior of SSA maximization. The class
33 /// exposes overridable filter functions to dynamically select which blocks,
34 /// block arguments, operations, and operation results should be put into
35 /// maximal SSA form. All filter functions should return true whenever the
36 /// entity they operate on should be considered for SSA maximization. By
37 /// default, all filter functions always return true.
38 class SSAMaximizationStrategy {
39 public:
40  /// Determines whether a block should have the values it defines (i.e., block
41  /// arguments and operation results within the block) SSA maximized.
42  virtual bool maximizeBlock(Block *block);
43  /// Determines whether a block argument should be SSA maximized.
44  virtual bool maximizeArgument(BlockArgument arg);
45  /// Determines whether an operation should have its results SSA maximized.
46  virtual bool maximizeOp(Operation *op);
47  /// Determines whether an operation's result should be SSA maximized.
48  virtual bool maximizeResult(OpResult res);
49 
50  virtual ~SSAMaximizationStrategy() = default;
51 };

53 /// Converts a single value within a function into maximal SSA form. This
54 /// removes any implicit dataflow of this specific value within the enclosing

128  template <typename TTerm>
129  LogicalResult setControlOnlyPath(ConversionPatternRewriter &rewriter) {
130  // Creates start and end points of the control-only path
131 
132  // Add start point of the control-only path to the entry block's arguments
133  Block *entryBlock = &r.front();
134  startCtrl = entryBlock->addArgument(rewriter.getNoneType(),
135  rewriter.getUnknownLoc());
136  setBlockEntryControl(entryBlock, startCtrl);
137 
138  // Replace original return ops with new returns with additional control
139  // input
140  for (auto retOp : llvm::make_early_inc_range(r.getOps<TTerm>())) {
141  rewriter.setInsertionPoint(retOp);
142  SmallVector<Value, 8> operands(retOp->getOperands());
143  operands.push_back(startCtrl);
144  rewriter.replaceOpWithNewOp<handshake::ReturnOp>(retOp, operands);
145  }
146 
147  // Store the number of block arguments in each block
148  DenseMap<Block *, unsigned> numArgsPerBlock;
149  for (auto &block : r.getBlocks())
150  numArgsPerBlock[&block] = block.getNumArguments();
151 
152  // Apply SSA maximization on the newly added entry block argument to
153  // propagate it explicitly between the start-point of the control-only
154  // network and the function's terminators
155  if (failed(maximizeSSA(startCtrl, rewriter)))
156  return failure();
157 
158  // Identify all block arguments belonging to the control-only network
159  // (needed to later insert control merges after those values). These are the
160  // last arguments to blocks that got a new argument during SSA maximization
161  for (auto &[block, numArgs] : numArgsPerBlock)
162  if (block->getNumArguments() != numArgs)
163  setBlockEntryControl(block, block->getArguments().back());
164 
165  return success();
166  }

217 template <typename T, typename... TArgs, typename... TArgs2>
218 LogicalResult runPartialLowering(
219  T &instance,
220  LogicalResult (T::*memberFunc)(ConversionPatternRewriter &, TArgs2...),
221  TArgs &...args) {
222  return partiallyLowerRegion(
223  [&](Region &, ConversionPatternRewriter &rewriter) -> LogicalResult {
224  return (instance.*memberFunc)(rewriter, args...);
225  },
226  instance.getContext(), instance.getRegion());

229 // Helper to check the validity of the dataflow conversion
230 // Driver that applies the partial lowerings expressed in HandshakeLowering to
231 // the region encapsulated in it. The region is assumed to have a terminator of
232 // type TTerm. See HandshakeLowering for the different lowering steps.
233 template <typename TTerm>
234 LogicalResult lowerRegion(HandshakeLowering &hl, bool sourceConstants,
235  bool disableTaskPipelining) {
236  // Perform initial dataflow conversion. This process allows for the use of
237  // non-deterministic merge-like operations.
238  HandshakeLowering::MemRefToMemoryAccessOp memOps;
239 
240  if (failed(
241  runPartialLowering(hl, &HandshakeLowering::replaceMemoryOps, memOps)))
242  return failure();
243  if (failed(runPartialLowering(hl,
244  &HandshakeLowering::setControlOnlyPath<TTerm>)))
245  return failure();
246  if (failed(runPartialLowering(hl, &HandshakeLowering::addMergeOps)))
247  return failure();
248  if (failed(runPartialLowering(hl, &HandshakeLowering::replaceCallOps)))
249  return failure();
250  if (failed(runPartialLowering(hl, &HandshakeLowering::addBranchOps)))
251  return failure();
252 
253  // The following passes modifies the dataflow graph to being safe for task
254  // pipelining. In doing so, non-deterministic merge structures are replaced
255  // for deterministic structures.
256  if (!disableTaskPipelining) {
257  if (failed(
258  runPartialLowering(hl, &HandshakeLowering::loopNetworkRewriting)))
259  return failure();
260  if (failed(
261  runPartialLowering(hl, &HandshakeLowering::feedForwardRewriting)))
262  return failure();
263  }
264 
265  if (failed(runPartialLowering(
266  hl, &HandshakeLowering::connectConstantsToControl, sourceConstants)))
267  return failure();
268 
269  bool lsq = false;
270  if (failed(runPartialLowering(hl, &HandshakeLowering::connectToMemory, memOps,
271  lsq)))
272  return failure();
273 
274  return success();
275 }
