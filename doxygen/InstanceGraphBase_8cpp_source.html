15 void InstanceRecord::erase() {
16  // Update the prev node to point to the next node.
17  if (prevUse)
18  prevUse->nextUse = nextUse;
19  else
20  target->firstUse = nextUse;
21  // Update the next node to point to the prev node.
22  if (nextUse)
23  nextUse->prevUse = prevUse;
24  getParent()->instances.erase(this);
25 }

53 InstanceGraphBase::InstanceGraphBase(Operation *parent) : parent(parent) {
54  parent->walk([&](HWModuleLike module) {
55  auto name = module.getModuleNameAttr();
56  auto *currentNode = getOrAddNode(name);
57  currentNode->module = module;
58 
59  // Find all instance operations in the module body.
60  module.walk([&](HWInstanceLike instanceOp) {
61  // Add an edge to indicate that this module instantiates the target.
62  auto *targetNode = getOrAddNode(instanceOp.getReferencedModuleNameAttr());
63  currentNode->addInstance(instanceOp, targetNode);
64  });
65  });
66 }

103 void InstanceGraphBase::replaceInstance(HWInstanceLike inst,
104  HWInstanceLike newInst) {
105  assert(inst.getReferencedModuleName() == newInst.getReferencedModuleName() &&
106  "Both instances must be targeting the same module");
107 
108  // Find the instance record of this instance.
109  auto *node = lookup(inst.getReferencedModuleNameAttr());
110  auto it = llvm::find_if(node->uses(), [&](InstanceRecord *record) {
111  return record->getInstance() == inst;
112  });
113  assert(it != node->usesEnd() && "Instance of module not recorded in graph");
114 
115  // We can just replace the instance op in the InstanceRecord without updating
116  // any instance lists.
117  (*it)->instance = newInst;
118 }

120 bool InstanceGraphBase::isAncestor(HWModuleLike child, HWModuleLike parent) {
121  DenseSet<InstanceGraphNode *> seen;
122  SmallVector<InstanceGraphNode *> worklist;
123  auto *cn = lookup(child);
124  worklist.push_back(cn);
125  seen.insert(cn);
126  while (!worklist.empty()) {
127  auto *node = worklist.back();
128  worklist.pop_back();
129  if (node->getModule() == parent)
130  return true;
131  for (auto *use : node->uses()) {
132  auto *mod = use->getParent();
133  if (!seen.count(mod)) {
134  seen.insert(mod);
135  worklist.push_back(mod);
136  }
137  }
138  }
139  return false;
140 }

142 FailureOr<llvm::ArrayRef<InstanceGraphNode *>>
143 InstanceGraphBase::getInferredTopLevelNodes() {
144  if (!inferredTopLevelNodes.empty())
145  return {inferredTopLevelNodes};
146 
147  /// Topologically sort the instance graph.
148  llvm::SetVector<InstanceGraphNode *> visited, marked;
149  llvm::SetVector<InstanceGraphNode *> candidateTopLevels(this->begin(),
150  this->end());
151  SmallVector<InstanceGraphNode *> cycleTrace;
152 
153  // Recursion function; returns true if a cycle was detected.
154  std::function<bool(InstanceGraphNode *, SmallVector<InstanceGraphNode *>)>
155  cycleUtil =
156  [&](InstanceGraphNode *node, SmallVector<InstanceGraphNode *> trace) {
157  if (visited.contains(node))
158  return false;
159  trace.push_back(node);
160  if (marked.contains(node)) {
161  // Cycle detected.
162  cycleTrace = trace;
163  return true;
164  }
165  marked.insert(node);
166  for (auto use : *node) {
167  InstanceGraphNode *targetModule = use->getTarget();
168  candidateTopLevels.remove(targetModule);
169  if (cycleUtil(targetModule, trace))
170  return true; // Cycle detected.
171  }
172  marked.remove(node);
173  visited.insert(node);
174  return false;
175  };
176 
177  bool cyclic = false;
178  for (auto moduleIt : *this) {
179  if (visited.contains(moduleIt))
180  continue;
181 
182  cyclic |= cycleUtil(moduleIt, {});
183  if (cyclic)
184  break;
185  }
186 
187  if (cyclic) {
188  auto err = getParent()->emitOpError();
189  err << "cannot deduce top level module - cycle "
190  "detected in instance graph (";
191  llvm::interleave(
192  cycleTrace, err,
193  [&](auto node) { err << node->getModule().getModuleName(); }, "->");
194  err << ").";
195  return err;
196  }
197  assert(!candidateTopLevels.empty() &&
198  "if non-cyclic, there should be at least 1 candidate top level");
199 
200  inferredTopLevelNodes = llvm::SmallVector<InstanceGraphNode *>(
201  candidateTopLevels.begin(), candidateTopLevels.end());
202  return {inferredTopLevelNodes};
203 }

205 ArrayRef<InstancePath> InstancePathCache::getAbsolutePaths(HWModuleLike op) {
206  InstanceGraphNode *node = instanceGraph[op];
207 
208  // If we have reached the circuit root, we're done.
209  if (node == instanceGraph.getTopLevelNode()) {
210  static InstancePath empty{};
211  return empty; // array with single empty path
212  }
213 
214  // Fast path: hit the cache.
215  auto cached = absolutePathsCache.find(op);
216  if (cached != absolutePathsCache.end())
217  return cached->second;
218 
219  // For each instance, collect the instance paths to its parent and append the
220  // instance itself to each.
221  SmallVector<InstancePath, 8> extendedPaths;
222  for (auto *inst : node->uses()) {
223  if (auto module = inst->getParent()->getModule()) {
224  auto instPaths = getAbsolutePaths(module);
225  extendedPaths.reserve(instPaths.size());
226  for (auto path : instPaths) {
227  extendedPaths.push_back(
228  appendInstance(path, cast<HWInstanceLike>(*inst->getInstance())));
229  }
230  }
231  }
232 
233  // Move the list of paths into the bump allocator for later quick retrieval.
234  ArrayRef<InstancePath> pathList;
235  if (!extendedPaths.empty()) {
236  auto *paths = allocator.Allocate<InstancePath>(extendedPaths.size());
237  std::copy(extendedPaths.begin(), extendedPaths.end(), paths);
238  pathList = ArrayRef<InstancePath>(paths, extendedPaths.size());
239  }
240  absolutePathsCache.insert({op, pathList});
241  return pathList;
242 }

253 void InstancePathCache::replaceInstance(HWInstanceLike oldOp,
254  HWInstanceLike newOp) {
255 
256  instanceGraph.replaceInstance(oldOp, newOp);
257 
258  // Iterate over all the paths, and search for the old HWInstanceLike. If
259  // found, then replace it with the new HWInstanceLike, and create a new copy
260  // of the paths and update the cache.
261  auto instanceExists = [&](const ArrayRef<InstancePath> &paths) -> bool {
262  return llvm::any_of(
263  paths, [&](InstancePath p) { return llvm::is_contained(p, oldOp); });
264  };
265 
266  for (auto &iter : absolutePathsCache) {
267  if (!instanceExists(iter.getSecond()))
268  continue;
269  SmallVector<InstancePath, 8> updatedPaths;
270  for (auto path : iter.getSecond()) {
271  const auto *iter = llvm::find(path, oldOp);
272  if (iter == path.end()) {
273  // path does not contain the oldOp, just copy it as is.
274  updatedPaths.push_back(path);
275  continue;
276  }
277  auto *newPath = allocator.Allocate<HWInstanceLike>(path.size());
278  llvm::copy(path, newPath);
279  newPath[iter - path.begin()] = newOp;
280  updatedPaths.push_back(InstancePath(newPath, path.size()));
281  }
282  // Move the list of paths into the bump allocator for later quick
283  // retrieval.
284  auto *paths = allocator.Allocate<InstancePath>(updatedPaths.size());
285  llvm::copy(updatedPaths, paths);
286  iter.getSecond() = ArrayRef<InstancePath>(paths, updatedPaths.size());
287  }
288 }
