220  firrtl::PreserveValues::PreserveMode getPreserveMode() const {
221  if (!buildMode.getNumOccurrences())
222  return preserveMode;
223  switch (buildMode) {
224  case BuildModeDebug:
225  return firrtl::PreserveValues::Named;
226  case BuildModeRelease:
227  return firrtl::PreserveValues::None;
228  }
229  llvm_unreachable("unknown build mode");
230  }
