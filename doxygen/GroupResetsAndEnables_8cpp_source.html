29 struct ResetGroupingPattern : public OpRewritePattern<ClockTreeOp> {
30  using OpRewritePattern::OpRewritePattern;
31  LogicalResult matchAndRewrite(ClockTreeOp clockTreeOp,
32  PatternRewriter &rewriter) const override {
33  // Group similar resets into single IfOps
34  // Create a list of reset values and map from them to the states they reset
35  llvm::MapVector<mlir::Value, SmallVector<scf::IfOp>> resetMap;
36 
37  for (auto ifOp : clockTreeOp.getBody().getOps<scf::IfOp>())
38  if (ifOp.getResults().empty())
39  resetMap[ifOp.getCondition()].push_back(ifOp);
40 
41  // TODO: Check that conflicting memory effects aren't being reordered
42 
43  // Combine IfOps
44  bool changed = false;
45  for (auto &[cond, oldOps] : resetMap) {
46  if (oldOps.size() <= 1)
47  continue;
48  scf::IfOp lastIfOp = oldOps.pop_back_val();
49  for (auto thisOp : oldOps) {
50  // Inline the before and after region inside the original If
51  rewriter.eraseOp(thisOp.thenBlock()->getTerminator());
52  rewriter.inlineBlockBefore(thisOp.thenBlock(),
53  &lastIfOp.thenBlock()->front());
54  // Check we're not inlining an empty block
55  if (auto *elseBlock = thisOp.elseBlock()) {
56  rewriter.eraseOp(elseBlock->getTerminator());
57  if (auto *lastElseBlock = lastIfOp.elseBlock()) {
58  rewriter.inlineBlockBefore(elseBlock,
59  &lastIfOp.elseBlock()->front());
60  } else {
61  lastElseBlock = rewriter.createBlock(&lastIfOp.getElseRegion());
62  rewriter.setInsertionPointToEnd(lastElseBlock);
63  auto yieldOp = rewriter.create<scf::YieldOp>(
64  lastElseBlock->getParentOp()->getLoc());
65  rewriter.inlineBlockBefore(thisOp.elseBlock(), yieldOp);
66  }
67  }
68  rewriter.eraseOp(thisOp);
69  changed = true;
70  }
71  }
72  return success(changed);
73  }
74 };

76 struct EnableGroupingPattern : public OpRewritePattern<ClockTreeOp> {
77  using OpRewritePattern::OpRewritePattern;
78  LogicalResult matchAndRewrite(ClockTreeOp clockTreeOp,
79  PatternRewriter &rewriter) const override {
80  // Amass regions that we want to group enables in
81  SmallVector<Region *> groupingRegions;
82  groupingRegions.push_back(&clockTreeOp.getBody());
83  for (auto ifOp : clockTreeOp.getBody().getOps<scf::IfOp>()) {
84  groupingRegions.push_back(&ifOp.getThenRegion());
85  groupingRegions.push_back(&ifOp.getElseRegion());
86  }
87 
88  bool changed = false;
89  for (auto *region : groupingRegions) {
90  llvm::MapVector<mlir::Value, SmallVector<StateWriteOp>> enableMap;
91  for (auto writeOp : region->getOps<StateWriteOp>()) {
92  if (writeOp.getCondition())
93  enableMap[writeOp.getCondition()].push_back(writeOp);
94  }
95  for (auto &[enable, writeOps] : enableMap) {
96  // Only group if multiple writes share an enable
97  if (writeOps.size() <= 1)
98  continue;
99  if (region->getParentOp()->hasTrait<OpTrait::NoTerminator>())
100  rewriter.setInsertionPointToEnd(&region->back());
101  else
102  rewriter.setInsertionPoint(region->back().getTerminator());
103  scf::IfOp ifOp =
104  rewriter.create<scf::IfOp>(writeOps[0].getLoc(), enable, false);
105  for (auto writeOp : writeOps) {
106  rewriter.updateRootInPlace(writeOp, [&]() {
107  writeOp->moveBefore(ifOp.thenBlock()->getTerminator());
108  writeOp.getConditionMutable().erase(0);
109  });
110  }
111  changed = true;
112  }
113  }
114  return success(changed);
115  }
116 };

138 LogicalResult GroupResetsAndEnablesPass::runOnModel(ModelOp modelOp) {
