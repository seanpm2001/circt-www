42 // Copied nearly exactly from hwops.cpp.
43 // TODO: Unify code once a `ModuleLike` op interface exists.
44 static void buildModule(OpBuilder &builder, OperationState &result,
45  StringAttr name, const hw::ModulePortInfo &ports) {
46  using namespace mlir::function_interface_impl;
47  LocationAttr unknownLoc = builder.getUnknownLoc();
48 
49  // Add an attribute for the name.
50  result.addAttribute(SymbolTable::getSymbolAttrName(), name);
51 
52  SmallVector<Attribute> argNames, resultNames;
53  SmallVector<Type, 4> argTypes, resultTypes;
54  SmallVector<Attribute> argAttrs, resultAttrs;
55  SmallVector<Attribute> argLocs, resultLocs;
56  auto exportPortIdent = StringAttr::get(builder.getContext(), "hw.exportPort");
57 
58  for (auto elt : ports.inputs) {
59  if (elt.direction == hw::PortDirection::INOUT &&
60  !elt.type.isa<hw::InOutType>())
61  elt.type = hw::InOutType::get(elt.type);
62  argTypes.push_back(elt.type);
63  argNames.push_back(elt.name);
64  argLocs.push_back(elt.loc ? elt.loc : unknownLoc);
65  Attribute attr;
66  if (elt.sym && !elt.sym.empty())
67  attr = builder.getDictionaryAttr({{exportPortIdent, elt.sym}});
68  else
69  attr = builder.getDictionaryAttr({});
70  argAttrs.push_back(attr);
71  }
72 
73  for (auto elt : ports.outputs) {
74  resultTypes.push_back(elt.type);
75  resultNames.push_back(elt.name);
76  resultLocs.push_back(elt.loc ? elt.loc : unknownLoc);
77  Attribute attr;
78  if (elt.sym && !elt.sym.empty())
79  attr = builder.getDictionaryAttr({{exportPortIdent, elt.sym}});
80  else
81  attr = builder.getDictionaryAttr({});
82  resultAttrs.push_back(attr);
83  }
84 
85  // Record the argument and result types as an attribute.
86  auto type = builder.getFunctionType(argTypes, resultTypes);
87  result.addAttribute(MSFTModuleOp::getFunctionTypeAttrName(result.name),
88  TypeAttr::get(type));
89  result.addAttribute("argNames", builder.getArrayAttr(argNames));
90  result.addAttribute("resultNames", builder.getArrayAttr(resultNames));
91  result.addAttribute("argLocs", builder.getArrayAttr(argLocs));
92  result.addAttribute("resultLocs", builder.getArrayAttr(resultLocs));
93  result.addAttribute("parameters", builder.getDictionaryAttr({}));
94  result.addAttribute(MSFTModuleOp::getArgAttrsAttrName(result.name),
95  builder.getArrayAttr(argAttrs));
96  result.addAttribute(MSFTModuleOp::getResAttrsAttrName(result.name),
97  builder.getArrayAttr(resultAttrs));
98  result.addRegion();
99 }

105 static ParseResult parsePhysLoc(OpAsmParser &p, PhysLocationAttr &attr) {
106  llvm::SMLoc loc = p.getCurrentLocation();
107  StringRef devTypeStr;
108  uint64_t x, y, num;
109 
110  if (p.parseKeyword(&devTypeStr) || p.parseKeyword("x") || p.parseColon() ||
111  p.parseInteger(x) || p.parseKeyword("y") || p.parseColon() ||
112  p.parseInteger(y) || p.parseKeyword("n") || p.parseColon() ||
113  p.parseInteger(num))
114  return failure();
115 
116  std::optional<PrimitiveType> devType = symbolizePrimitiveType(devTypeStr);
117  if (!devType) {
118  p.emitError(loc, "Unknown device type '" + devTypeStr + "'");
119  return failure();
120  }
121  PrimitiveTypeAttr devTypeAttr =
122  PrimitiveTypeAttr::get(p.getContext(), *devType);
123  attr = PhysLocationAttr::get(p.getContext(), devTypeAttr, x, y, num);
124  return success();
125 }

133  LocationVectorAttr &locs) {
134  SmallVector<PhysLocationAttr, 32> locArr;
135  TypeAttr type;
136  if (p.parseAttribute(type) || p.parseLSquare() ||

177 /// Parse an parameter list if present. Same format as HW dialect.
178 /// module-parameter-list ::= `<` parameter-decl (`,` parameter-decl)* `>`
179 /// parameter-decl ::= identifier `:` type
180 /// parameter-decl ::= identifier `:` type `=` attribute
181 ///
182 static ParseResult parseParameterList(OpAsmParser &parser,
183  SmallVector<Attribute> &parameters) {
184 
185  return parser.parseCommaSeparatedList(
186  OpAsmParser::Delimiter::OptionalLessGreater, [&]() {
187  std::string name;
188  Type type;
189  Attribute value;
190 
191  if (parser.parseKeywordOrString(&name) || parser.parseColonType(type))
192  return failure();
193 
194  // Parse the default value if present.
195  if (succeeded(parser.parseOptionalEqual())) {
196  if (parser.parseAttribute(value, type))
197  return failure();
198  }
199 
200  auto &builder = parser.getBuilder();
201  parameters.push_back(hw::ParamDeclAttr::get(
202  builder.getContext(), builder.getStringAttr(name), type, value));
203  return success();
204  });
205 }

220 static void printParameterList(OpAsmPrinter &p, Operation *op,
221  ArrayAttr parameters) {
222  if (!parameters || parameters.empty())
223  return;

225  p << '<';

231  p.printAttributeWithoutType(value);

237 static ParseResult parseModuleLikeOp(OpAsmParser &parser,
238  OperationState &result,
239  bool withParameters = false) {
240  using namespace mlir::function_interface_impl;
241  auto loc = parser.getCurrentLocation();
242 
243  // Parse the name as a symbol.
244  StringAttr nameAttr;
245  if (parser.parseSymbolName(nameAttr, SymbolTable::getSymbolAttrName(),
246  result.attributes))
247  return failure();
248 
249  if (withParameters) {
250  // Parse the parameters
251  DictionaryAttr paramsAttr;
252  if (parser.parseAttribute(paramsAttr))
253  return failure();
254  result.addAttribute("parameters", paramsAttr);
255  }
256 
257  // Parse the function signature.
258  bool isVariadic = false;
259  SmallVector<OpAsmParser::Argument, 4> entryArgs;
260  SmallVector<Attribute> argNames;
261  SmallVector<Attribute> argLocs;
262  SmallVector<Attribute> resultNames;
263  SmallVector<DictionaryAttr, 4> resultAttrs;
264  SmallVector<Attribute> resultLocs;
265  TypeAttr functionType;
266  if (hw::module_like_impl::parseModuleFunctionSignature(
267  parser, isVariadic, entryArgs, argNames, argLocs, resultNames,
268  resultAttrs, resultLocs, functionType))
269  return failure();
270 
271  // If function attributes are present, parse them.
272  if (parser.parseOptionalAttrDictWithKeyword(result.attributes))
273  return failure();
274 
275  if (hasAttribute("argNames", result.attributes) ||
276  hasAttribute("resultNames", result.attributes)) {
277  parser.emitError(
278  loc, "explicit argNames and resultNames attributes not allowed");
279  return failure();
280  }
281 
282  auto *context = result.getContext();
283  result.addAttribute("argNames", ArrayAttr::get(context, argNames));
284  result.addAttribute("argLocs", ArrayAttr::get(context, argLocs));
285  result.addAttribute("resultNames", ArrayAttr::get(context, resultNames));
286  result.addAttribute("resultLocs", ArrayAttr::get(context, resultLocs));
287  result.addAttribute(MSFTModuleOp::getFunctionTypeAttrName(result.name),
288  functionType);
289 
290  // Add the attributes to the module arguments.
291  addArgAndResultAttrs(parser.getBuilder(), result, entryArgs, resultAttrs,
292  MSFTModuleOp::getArgAttrsAttrName(result.name),
293  MSFTModuleOp::getResAttrsAttrName(result.name));
294 
295  // Parse the optional module body.
296  auto regionSuccess =
297  parser.parseOptionalRegion(*result.addRegion(), entryArgs);
298  if (regionSuccess.has_value() && failed(*regionSuccess))
299  return failure();
300 
301  return success();
302 }

304 template <typename ModuleTy>
305 static void printModuleLikeOp(mlir::FunctionOpInterface moduleLike,
306  OpAsmPrinter &p, Attribute parameters = nullptr) {
307  using namespace mlir::function_interface_impl;
308 
309  auto argTypes = moduleLike.getArgumentTypes();
310  auto resultTypes = moduleLike.getResultTypes();
311 
312  // Print the operation and the function name.
313  p << ' ';
314  p.printSymbolName(SymbolTable::getSymbolName(moduleLike).getValue());
315 
316  if (parameters) {
317  // Print the parameterization.
318  p << ' ';
319  p.printAttribute(parameters);
320  }
321 
322  p << ' ';
323  bool needArgNamesAttr = false;
324  hw::module_like_impl::printModuleSignature(p, moduleLike, argTypes,
325  /*isVariadic=*/false, resultTypes,
326  needArgNamesAttr);
327 
328  SmallVector<StringRef, 3> omittedAttrs;
329  if (!needArgNamesAttr)
330  omittedAttrs.push_back("argNames");
331  omittedAttrs.push_back("argLocs");
332  omittedAttrs.push_back("resultNames");
333  omittedAttrs.push_back("resultLocs");
334  omittedAttrs.push_back("parameters");
335  omittedAttrs.push_back(
336  ModuleTy::getFunctionTypeAttrName(moduleLike->getName()));
337  omittedAttrs.push_back(ModuleTy::getArgAttrsAttrName(moduleLike->getName()));
338  omittedAttrs.push_back(ModuleTy::getResAttrsAttrName(moduleLike->getName()));
339 
340  printFunctionAttributes(p, moduleLike, omittedAttrs);
341 
342  // Print the body if this is not an external function.
343  Region &mbody = moduleLike.getFunctionBody();
344  if (!mbody.empty()) {
345  p << ' ';
346  p.printRegion(mbody, /*printEntryBlockArgs=*/false,
347  /*printBlockTerminators=*/true);
348  }
349 }

423 LogicalResult
424 InstanceOp::verifySignatureMatch(const hw::ModulePortInfo &ports) {
425  if (ports.inputs.size() != getNumOperands())
426  return emitOpError("wrong number of inputs (expected ")
427  << ports.inputs.size() << ")";
428  if (ports.outputs.size() != getNumResults())
429  return emitOpError("wrong number of outputs (expected ")
430  << ports.outputs.size() << ")";
431  for (auto port : ports.inputs)
432  if (getOperand(port.argNum).getType() != port.type)
433  return emitOpError("in input port ")
434  << port.name << ", expected type " << port.type << " got "
435  << getOperand(port.argNum).getType();
436  for (auto port : ports.outputs)
437  if (getResult(port.argNum).getType() != port.type)
438  return emitOpError("in output port ")
439  << port.name << ", expected type " << port.type << " got "
440  << getResult(port.argNum).getType();
441 
442  return success();
443 }

456 /// Return an encapsulated set of information about input and output ports of
457 /// the specified module or instance. The input ports always come before the
458 /// output ports in the list.
459 /// TODO: This should really be shared with the HW dialect instead of cloned.
460 /// Consider adding a `HasModulePorts` op interface to facilitate.
461 hw::ModulePortInfo MSFTModuleOp::getPorts() {
462  SmallVector<hw::PortInfo> inputs, outputs;
463  auto argNames = this->getArgNames();
464  auto argTypes = getArgumentTypes();
465  auto argLocs = getArgLocs();
466 
467  for (unsigned i = 0, e = argTypes.size(); i < e; ++i) {
468  bool isInOut = false;
469  auto argName = argNames[i].cast<StringAttr>();
470  auto direction =
471  isInOut ? hw::PortDirection::INOUT : hw::PortDirection::INPUT;
472  auto type = argTypes[i];
473  if (auto inout = type.dyn_cast<hw::InOutType>()) {
474  isInOut = true;
475  type = inout.getElementType();
476  }
477  auto argLoc = argLocs[i].cast<LocationAttr>();
478  inputs.push_back({argName, direction, type, i, {}, argLoc});
479  }
480 
481  auto resultNames = this->getResultNames();
482  auto resultTypes = getResultTypes();
483  auto resultLocs = getResultLocs();
484  for (unsigned i = 0, e = resultTypes.size(); i < e; ++i) {
485  outputs.push_back({resultNames[i].cast<StringAttr>(),
486  hw::PortDirection::OUTPUT,
487  resultTypes[i],
488  i,
489  {},
490  resultLocs[i].cast<LocationAttr>()});
491  }
492  return hw::ModulePortInfo(inputs, outputs);
493 }

495 SmallVector<BlockArgument>
496 MSFTModuleOp::addPorts(ArrayRef<std::pair<StringAttr, Type>> inputs,
497  ArrayRef<std::pair<StringAttr, Value>> outputs) {
498  auto *ctxt = getContext();
499  Block *body = getBodyBlock();
500 
501  // Append new inputs.
502  SmallVector<Type, 32> modifiedArgs(getArgumentTypes().begin(),
503  getArgumentTypes().end());
504  SmallVector<Attribute> modifiedArgNames(
505  getArgNames().getAsRange<Attribute>());
506  SmallVector<Attribute> modifiedArgLocs(getArgLocs().getAsRange<Attribute>());
507  SmallVector<BlockArgument> newBlockArgs;
508  Location unknownLoc = UnknownLoc::get(ctxt);
509  for (auto ttPair : inputs) {
510  modifiedArgNames.push_back(ttPair.first);
511  modifiedArgs.push_back(ttPair.second);
512  modifiedArgLocs.push_back(unknownLoc);
513  newBlockArgs.push_back(
514  body->addArgument(ttPair.second, Builder(ctxt).getUnknownLoc()));
515  }
516  setArgNamesAttr(ArrayAttr::get(ctxt, modifiedArgNames));
517  setArgLocsAttr(ArrayAttr::get(ctxt, modifiedArgLocs));
518 
519  // Append new outputs.
520  SmallVector<Type, 32> modifiedResults(getResultTypes().begin(),
521  getResultTypes().end());
522  SmallVector<Attribute> modifiedResultNames(
523  getResultNames().getAsRange<Attribute>());
524  SmallVector<Attribute> modifiedResultLocs(
525  getResultLocs().getAsRange<Attribute>());
526  Operation *terminator = body->getTerminator();
527  SmallVector<Value, 32> modifiedOutputs(terminator->getOperands());
528  for (auto tvPair : outputs) {
529  modifiedResultNames.push_back(tvPair.first);
530  modifiedResults.push_back(tvPair.second.getType());
531  modifiedResultLocs.push_back(unknownLoc);
532  modifiedOutputs.push_back(tvPair.second);
533  }
534  setResultNamesAttr(ArrayAttr::get(ctxt, modifiedResultNames));
535  setResultLocsAttr(ArrayAttr::get(ctxt, modifiedResultLocs));
536  terminator->setOperands(modifiedOutputs);
537 
538  // Finalize and return.
539  setType(FunctionType::get(ctxt, modifiedArgs, modifiedResults));
540  return newBlockArgs;
541 }

543 // Remove the ports at the specified indexes.
544 SmallVector<unsigned> MSFTModuleOp::removePorts(llvm::BitVector inputs,
545  llvm::BitVector outputs) {
546  MLIRContext *ctxt = getContext();
547  FunctionType ftype = getFunctionType();
548  Block *body = getBodyBlock();
549  Operation *terminator = body->getTerminator();
550 
551  SmallVector<Type, 4> newInputTypes;
552  SmallVector<Attribute, 4> newArgNames;
553  SmallVector<Attribute, 4> newArgLocs;
554  unsigned originalNumArgs = ftype.getNumInputs();
555  ArrayRef<Attribute> origArgNames = getArgNamesAttr().getValue();
556  ArrayRef<Attribute> origArgLocs = getArgLocsAttr().getValue();
557  assert(origArgNames.size() == originalNumArgs);
558  for (size_t i = 0; i < originalNumArgs; ++i) {
559  if (!inputs.test(i)) {
560  newInputTypes.emplace_back(ftype.getInput(i));
561  newArgNames.emplace_back(origArgNames[i]);
562  newArgLocs.emplace_back(origArgLocs[i]);
563  }
564  }
565 
566  SmallVector<Type, 4> newResultTypes;
567  SmallVector<Attribute, 4> newResultNames;
568  SmallVector<Attribute, 4> newResultLocs;
569  unsigned originalNumResults = getNumResults();
570  ArrayRef<Attribute> origResNames = getResultNamesAttr().getValue();
571  ArrayRef<Attribute> origResLocs = getResultLocsAttr().getValue();
572  assert(origResNames.size() == originalNumResults);
573  for (size_t i = 0; i < originalNumResults; ++i) {
574  if (!outputs.test(i)) {
575  newResultTypes.emplace_back(ftype.getResult(i));
576  newResultNames.emplace_back(origResNames[i]);
577  newResultLocs.emplace_back(origResLocs[i]);
578  }
579  }
580 
581  setType(FunctionType::get(ctxt, newInputTypes, newResultTypes));
582  setResultNamesAttr(ArrayAttr::get(ctxt, newResultNames));
583  setResultLocsAttr(ArrayAttr::get(ctxt, newResultLocs));
584  setArgNamesAttr(ArrayAttr::get(ctxt, newArgNames));
585  setArgLocsAttr(ArrayAttr::get(ctxt, newArgLocs));
586 
587  // Build new operand list for output op. Construct an output mapping to
588  // return as a side-effect.
589  unsigned numResults = ftype.getNumResults();
590  SmallVector<Value> newOutputValues;
591  SmallVector<unsigned> newToOldResultMap;
592 
593  for (unsigned i = 0; i < numResults; ++i) {
594  if (!outputs.test(i)) {
595  newOutputValues.push_back(terminator->getOperand(i));
596  newToOldResultMap.push_back(i);
597  }
598  }
599  terminator->setOperands(newOutputValues);
600 
601  // Erase the arguments after setting the new output op operands since the
602  // arguments might be used by output op.
603  for (unsigned argNum = 0, e = body->getArguments().size(); argNum < e;
604  ++argNum)
605  if (inputs.test(argNum))
606  body->getArgument(argNum).dropAllUses();
607  body->eraseArguments(inputs);
608 
609  return newToOldResultMap;
610 }

687 /// Check parameter specified by `value` to see if it is valid within the
688 /// scope of the specified module `module`. If not, emit an error at the
689 /// location of `usingOp` and return failure, otherwise return success. If
690 /// `usingOp` is null, then no diagnostic is generated. Same format as HW
691 /// dialect.
692 ///
693 /// If `disallowParamRefs` is true, then parameter references are not allowed.
694 static LogicalResult checkParameterInContext(Attribute value, Operation *module,
695  Operation *usingOp,
696  bool disallowParamRefs) {
697  // Literals are always ok. Their types are already known to match
698  // expectations.
699  if (value.isa<IntegerAttr>() || value.isa<FloatAttr>() ||
700  value.isa<StringAttr>() || value.isa<hw::ParamVerbatimAttr>())
701  return success();
702 
703  // Check both subexpressions of an expression.
704  if (auto expr = value.dyn_cast<hw::ParamExprAttr>()) {
705  for (auto op : expr.getOperands())
706  if (failed(
707  checkParameterInContext(op, module, usingOp, disallowParamRefs)))
708  return failure();
709  return success();
710  }
711 
712  // Parameter references need more analysis to make sure they are valid
713  // within this module.
714  if (auto parameterRef = value.dyn_cast<hw::ParamDeclRefAttr>()) {
715  auto nameAttr = parameterRef.getName();
716 
717  // Don't allow references to parameters from the default values of a
718  // parameter list.
719  if (disallowParamRefs) {
720  if (usingOp)
721  usingOp->emitOpError("parameter ")
722  << nameAttr << " cannot be used as a default value for a parameter";
723  return failure();
724  }
725 
726  // Find the corresponding attribute in the module.
727  for (auto param : module->getAttrOfType<ArrayAttr>("parameters")) {
728  auto paramAttr = param.cast<hw::ParamDeclAttr>();
729  if (paramAttr.getName() != nameAttr)
730  continue;
731 
732  // If the types match then the reference is ok.
733  if (paramAttr.getType() == parameterRef.getType())
734  return success();
735 
736  if (usingOp) {
737  auto diag = usingOp->emitOpError("parameter ")
738  << nameAttr << " used with type " << parameterRef.getType()
739  << "; should have type " << paramAttr.getType();
740  diag.attachNote(module->getLoc()) << "module declared here";
741  }
742  return failure();
743  }
744 
745  if (usingOp) {
746  auto diag = usingOp->emitOpError("use of unknown parameter ") << nameAttr;
747  diag.attachNote(module->getLoc()) << "module declared here";
748  }
749  return failure();
750  }
751 
752  if (usingOp)
753  usingOp->emitOpError("invalid parameter value ") << value;
754  return failure();
755 }

757 ParseResult MSFTModuleExternOp::parse(OpAsmParser &parser,
758  OperationState &result) {
759  using namespace mlir::function_interface_impl;
760  auto loc = parser.getCurrentLocation();
761 
762  // Parse the name as a symbol.
763  StringAttr nameAttr;
764  if (parser.parseSymbolName(nameAttr, SymbolTable::getSymbolAttrName(),
765  result.attributes))
766  return failure();
767 
768  // Parse the parameters.
769  SmallVector<Attribute> parameters;
770  if (parseParameterList(parser, parameters))
771  return failure();
772 
773  // Parse the function signature.
774  bool isVariadic = false;
775  SmallVector<OpAsmParser::Argument, 4> entryArgs;
776  SmallVector<Attribute> argNames;
777  SmallVector<Attribute> argLocs;
778  SmallVector<Attribute> resultNames;
779  SmallVector<DictionaryAttr> resultAttrs;
780  SmallVector<Attribute> resultLocs;
781  TypeAttr functionType;
782  if (failed(hw::module_like_impl::parseModuleFunctionSignature(
783  parser, isVariadic, entryArgs, argNames, argLocs, resultNames,
784  resultAttrs, resultLocs, functionType)))
785  return failure();
786 
787  // Parse the attribute dict.
788  if (failed(parser.parseOptionalAttrDictWithKeyword(result.attributes)))
789  return failure();
790 
791  if (hasAttribute("resultNames", result.attributes) ||
792  hasAttribute("parameters", result.attributes)) {
793  parser.emitError(
794  loc, "explicit `resultNames` / `parameters` attributes not allowed");
795  return failure();
796  }
797 
798  auto *context = result.getContext();
799 
800  // An explicit `argNames` attribute overrides the MLIR names. This is how
801  // we represent port names that aren't valid MLIR identifiers. Result and
802  // parameter names are printed quoted when they aren't valid identifiers, so
803  // they don't need this affordance.
804  if (!hasAttribute("argNames", result.attributes))
805  result.addAttribute("argNames", ArrayAttr::get(context, argNames));
806  result.addAttribute("argLocs", ArrayAttr::get(context, argLocs));
807  result.addAttribute("resultNames", ArrayAttr::get(context, resultNames));
808  result.addAttribute("resultLocs", ArrayAttr::get(context, resultLocs));
809  result.addAttribute("parameters", ArrayAttr::get(context, parameters));
810  result.addAttribute(MSFTModuleExternOp::getFunctionTypeAttrName(result.name),
811  functionType);
812 
813  // Add the attributes to the function arguments.
814  addArgAndResultAttrs(parser.getBuilder(), result, entryArgs, resultAttrs,
815  MSFTModuleExternOp::getArgAttrsAttrName(result.name),
816  MSFTModuleExternOp::getResAttrsAttrName(result.name));
817 
818  // Extern modules carry an empty region to work with
819  // HWModuleImplementation.h.
820  result.addRegion();
821 
822  return success();
823 }

859 LogicalResult MSFTModuleExternOp::verify() {
860  using namespace mlir::function_interface_impl;
861  auto typeAttr = (*this)->getAttrOfType<TypeAttr>(getFunctionTypeAttrName());
862  auto moduleType = typeAttr.getValue().cast<FunctionType>();
863  auto argNames = (*this)->getAttrOfType<ArrayAttr>("argNames");
864  auto resultNames = (*this)->getAttrOfType<ArrayAttr>("resultNames");
865  if (argNames.size() != moduleType.getNumInputs())
866  return emitOpError("incorrect number of argument names");
867  if (resultNames.size() != moduleType.getNumResults())
868  return emitOpError("incorrect number of result names");
869 
870  SmallPtrSet<Attribute, 4> paramNames;
871 
872  // Check parameter default values are sensible.
873  for (auto param : (*this)->getAttrOfType<ArrayAttr>("parameters")) {
874  auto paramAttr = param.cast<hw::ParamDeclAttr>();
875 
876  // Check that we don't have any redundant parameter names. These are
877  // resolved by string name: reuse of the same name would cause
878  // ambiguities.
879  if (!paramNames.insert(paramAttr.getName()).second)
880  return emitOpError("parameter ")
881  << paramAttr << " has the same name as a previous parameter";
882 
883  // Default values are allowed to be missing, check them if present.
884  auto value = paramAttr.getValue();
885  if (!value)
886  continue;
887 
888  auto typedValue = value.dyn_cast<mlir::TypedAttr>();
889  if (!typedValue)
890  return emitOpError("parameter ")
891  << paramAttr << " should have a typed value; has value " << value;
892 
893  if (typedValue.getType() != paramAttr.getType())
894  return emitOpError("parameter ")
895  << paramAttr << " should have type " << paramAttr.getType()
896  << "; has type " << typedValue.getType();
897 
898  // Verify that this is a valid parameter value, disallowing parameter
899  // references. We could allow parameters to refer to each other in the
900  // future with lexical ordering if there is a need.
901  if (failed(checkParameterInContext(value, *this, *this,
902  /*disallowParamRefs=*/true)))
903  return failure();
904  }
905  return success();
906 }

964 ParseResult SystolicArrayOp::parse(OpAsmParser &parser,
965  OperationState &result) {
966  uint64_t numRows, numColumns;
967  Type rowType, columnType;
968  OpAsmParser::UnresolvedOperand rowInputs, columnInputs;
969  llvm::SMLoc loc = parser.getCurrentLocation();
970  if (parser.parseLSquare() || parser.parseOperand(rowInputs) ||
971  parser.parseColon() || parser.parseInteger(numRows) ||
972  parser.parseKeyword("x") || parser.parseType(rowType) ||
973  parser.parseRSquare() || parser.parseLSquare() ||
974  parser.parseOperand(columnInputs) || parser.parseColon() ||
975  parser.parseInteger(numColumns) || parser.parseKeyword("x") ||
976  parser.parseType(columnType) || parser.parseRSquare())
977  return failure();
978 
979  hw::ArrayType rowInputType = hw::ArrayType::get(rowType, numRows);
980  hw::ArrayType columnInputType = hw::ArrayType::get(columnType, numColumns);
981  SmallVector<Value> operands;
982  if (parser.resolveOperands({rowInputs, columnInputs},
983  {rowInputType, columnInputType}, loc, operands))
984  return failure();
985  result.addOperands(operands);
986 
987  Type peOutputType;
988  SmallVector<OpAsmParser::Argument> peArgs;
989  if (parser.parseKeyword("pe")) {
990  return failure();
991  }
992  llvm::SMLoc peLoc = parser.getCurrentLocation();
993  if (parser.parseArgumentList(peArgs, AsmParser::Delimiter::Paren)) {
994  return failure();
995  }
996  if (peArgs.size() != 2) {
997  return parser.emitError(peLoc, "expected two operands");
998  }
999 
1000  peArgs[0].type = rowType;
1001  peArgs[1].type = columnType;
1002 
1003  if (parser.parseArrow() || parser.parseLParen() ||
1004  parser.parseType(peOutputType) || parser.parseRParen())
1005  return failure();
1006 
1007  result.addTypes({hw::ArrayType::get(
1008  hw::ArrayType::get(peOutputType, numColumns), numRows)});
1009 
1010  Region *pe = result.addRegion();
1011 
1012  peLoc = parser.getCurrentLocation();
1013 
1014  if (parser.parseRegion(*pe, peArgs))
1015  return failure();
1016 
1017  if (pe->getBlocks().size() != 1)
1018  return parser.emitError(peLoc, "expected one block for the PE");
1019  Operation *peTerm = pe->getBlocks().front().getTerminator();
1020  if (peTerm->getOperands().size() != 1)
1021  return peTerm->emitOpError("expected one return value");
1022  if (peTerm->getOperand(0).getType() != peOutputType)
1023  return peTerm->emitOpError("expected return type as given in parent: ")
1024  << peOutputType;
1025 
1026  return success();
1027 }
