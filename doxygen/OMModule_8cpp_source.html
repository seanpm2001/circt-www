22 /// Provides an Object class by simply wrapping the OMObject CAPI.
23 struct Object {
24  // Instantiate an Object with a reference to the underlying OMObject.
25  Object(OMObject object) : object(object) {}
26 
27  /// Get the Type from an Object, which will be a ClassType.
28  MlirType getType() { return omEvaluatorObjectGetType(object); }
29 
30  // Get a field from the Object, using pybind's support for variant to return a
31  // Python object that is either an Object or Attribute.
32  std::variant<Object, MlirAttribute> getField(MlirAttribute name) {
33  // Get the field's ObjectValue via the CAPI.
34  OMObjectValue result = omEvaluatorObjectGetField(object, name);
35 
36  // If the ObjectValue is null, something failed. Diagnostic handling is
37  // implemented in pure Python, so nothing to do here besides throwing an
38  // error to halt execution.
39  if (omEvaluatorObjectValueIsNull(result))
40  throw py::value_error("unable to get field, see previous error(s)");
41 
42  // If the field was an Object, return a new Object.
43  if (omEvaluatorObjectValueIsAObject(result))
44  return Object(omEvaluatorObjectValueGetObject(result));
45 
46  // If the field was a primitive, return the Attribute.
47  assert(omEvaluatorObjectValueIsAPrimitive(result));
48  return omEvaluatorObjectValueGetPrimitive(result);
49  }
50 
51 private:
52  // The underlying CAPI OMObject.
53  OMObject object;
54 };

56 /// Provides an Evaluator class by simply wrapping the OMEvaluator CAPI.
57 struct Evaluator {
58  // Instantiate an Evaluator with a reference to the underlying OMEvaluator.
59  Evaluator(MlirModule mod) : evaluator(omEvaluatorNew(mod)) {}
60 
61  // Instantiate an Object.
62  Object instantiate(MlirAttribute className,
63  std::vector<MlirAttribute> actualParams) {
64  // Instantiate the Object via the CAPI.
65  OMObject result = omEvaluatorInstantiate(
66  evaluator, className, actualParams.size(), actualParams.begin().base());
67 
68  // If the Object is null, something failed. Diagnostic handling is
69  // implemented in pure Python, so nothing to do here besides throwing an
70  // error to halt execution.
71  if (omEvaluatorObjectIsNull(result))
72  throw py::value_error(
73  "unable to instantiate object, see previous error(s)");
74 
75  // Return a new Object.
76  return Object(result);
77  }
78 
79  // Get the Module the Evaluator is built from.
80  MlirModule getModule() { return omEvaluatorGetModule(evaluator); }
81 
82 private:
83  // The underlying CAPI OMEvaluator.
84  OMEvaluator evaluator;
85 };

87 /// Populate the OM Python module.
88 void circt::python::populateDialectOMSubmodule(py::module &m) {
89  m.doc() = "OM dialect Python native extension";
90 
91  // Add the Evaluator class definition.
92  py::class_<Evaluator>(m, "Evaluator")
93  .def(py::init<MlirModule>(), py::arg("module"))
94  .def("instantiate", &Evaluator::instantiate, "Instantiate an Object",
95  py::arg("class_name"), py::arg("actual_params"))
96  .def_property_readonly("module", &Evaluator::getModule,
97  "The Module the Evaluator is built from");
98 
99  // Add the Object class definition.
100  py::class_<Object>(m, "Object")
101  .def("get_field", &Object::getField, "Get a field from an Object",
102  py::arg("name"))
103  .def_property_readonly("type", &Object::getType,
104  "The Type of the Object");
105 
106  // Add the ClassType class definition.
107  mlir_type_subclass(m, "ClassType", omTypeIsAClassType);
108 }
