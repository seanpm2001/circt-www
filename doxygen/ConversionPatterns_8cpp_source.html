13 // Converts a function type wrt. the given type converter.
14 static FunctionType convertFunctionType(TypeConverter &typeConverter,
15  FunctionType type) {
16  // Convert the original function types.
17  llvm::SmallVector<Type> res, arg;
18  llvm::transform(type.getResults(), std::back_inserter(res),
19  [&](Type t) { return typeConverter.convertType(t); });
20  llvm::transform(type.getInputs(), std::back_inserter(arg),
21  [&](Type t) { return typeConverter.convertType(t); });
22 
23  return FunctionType::get(type.getContext(), arg, res);
24 }

26 LogicalResult TypeConversionPattern::matchAndRewrite(
27  Operation *op, ArrayRef<Value> operands,
28  ConversionPatternRewriter &rewriter) const {
29  // Convert the TypeAttrs.
30  llvm::SmallVector<NamedAttribute, 4> newAttrs;
31  newAttrs.reserve(op->getAttrs().size());
32  for (auto attr : op->getAttrs()) {
33  if (auto typeAttr = attr.getValue().dyn_cast<TypeAttr>()) {
34  auto innerType = typeAttr.getValue();
35  // TypeConvert::convertType doesn't handle function types, so we need to
36  // handle them manually.
37  if (auto funcType = innerType.dyn_cast<FunctionType>(); innerType)
38  innerType = convertFunctionType(*getTypeConverter(), funcType);
39  else
40  innerType = getTypeConverter()->convertType(innerType);
41  newAttrs.emplace_back(attr.getName(), TypeAttr::get(innerType));
42  } else {
43  newAttrs.push_back(attr);
44  }
45  }
46 
47  // Convert the result types.
48  llvm::SmallVector<Type, 4> newResults;
49  if (failed(
50  getTypeConverter()->convertTypes(op->getResultTypes(), newResults)))
51  return rewriter.notifyMatchFailure(op->getLoc(), "type conversion failed");
52 
53  // Build the state for the edited clone.
54  OperationState state(op->getLoc(), op->getName().getStringRef(), operands,
55  newResults, newAttrs, op->getSuccessors());
56  for (size_t i = 0, e = op->getNumRegions(); i < e; ++i)
57  state.addRegion();
58 
59  // Must create the op before running any modifications on the regions so that
60  // we don't crash with '-debug' and so we have something to 'root update'.
61  Operation *newOp = rewriter.create(state);
62 
63  // Move the regions over, converting the signatures as we go.
64  rewriter.startRootUpdate(newOp);
65  for (size_t i = 0, e = op->getNumRegions(); i < e; ++i) {
66  Region &region = op->getRegion(i);
67  Region *newRegion = &newOp->getRegion(i);
68 
69  // TypeConverter::SignatureConversion drops argument locations, so we need
70  // to manually copy them over (a verifier in e.g. HWModule checks this).
71  llvm::SmallVector<Location, 4> argLocs;
72  for (auto arg : region.getArguments())
73  argLocs.push_back(arg.getLoc());
74 
75  // Move the region and convert the region args.
76  rewriter.inlineRegionBefore(region, *newRegion, newRegion->begin());
77  TypeConverter::SignatureConversion result(newRegion->getNumArguments());
78  if (failed(getTypeConverter()->convertSignatureArgs(
79  newRegion->getArgumentTypes(), result)))
80  return rewriter.notifyMatchFailure(op->getLoc(),
81  "type conversion failed");
82  rewriter.applySignatureConversion(newRegion, result, getTypeConverter());
83 
84  // Apply the argument locations.
85  for (auto [arg, loc] : llvm::zip(newRegion->getArguments(), argLocs))
86  arg.setLoc(loc);
87  }
88  rewriter.finalizeRootUpdate(newOp);
89 
90  rewriter.replaceOp(op, newOp->getResults());
91  return success();
92 }
