31 /// This holds the name, type, direction of a module's ports
32 struct PortInfo {
33  StringAttr name;
34  PortDirection direction;
35  Type type;
36 
37  /// This is the argument index or the result index depending on the direction.
38  /// "0" for an output means the first output, "0" for a in/inout means the
39  /// first argument.
40  size_t argNum = ~0U;
41 
42  /// The optional symbol for this port.
43  InnerSymAttr sym = {};
44  LocationAttr loc = {};
45 
46  StringRef getName() const { return name.getValue(); }
47  bool isInput() const { return direction == PortDirection::INPUT; }
48  bool isOutput() const { return direction == PortDirection::OUTPUT; }
49  bool isInOut() const { return direction == PortDirection::INOUT; }
50 
51  /// Return a unique numeric identifier for this port.
52  ssize_t getId() const { return isOutput() ? argNum : (-1 - argNum); };
53 };

55 /// This holds a decoded list of input/inout and output ports for a module or
56 /// instance.
57 struct ModulePortInfo {
58  explicit ModulePortInfo(ArrayRef<PortInfo> inputs, ArrayRef<PortInfo> outputs)
59  : inputs(inputs.begin(), inputs.end()),
60  outputs(outputs.begin(), outputs.end()) {}
61 
62  explicit ModulePortInfo(ArrayRef<PortInfo> mergedPorts) {
63  inputs.reserve(mergedPorts.size());
64  outputs.reserve(mergedPorts.size());
65  for (auto port : mergedPorts) {
66  if (port.isOutput())
67  outputs.push_back(port);
68  else
69  inputs.push_back(port);
70  }
71  }
72 
73  /// This contains a list of the input and inout ports.
74  SmallVector<PortInfo> inputs;
75  /// This is a list of the output ports.
76  SmallVector<PortInfo> outputs;
77 };

93 /// This trait is for operations that define a scope for resolving InnerRef's,
94 /// and provides verification for InnerRef users (via InnerRefUserOpInterface).
95 template <typename ConcreteType>
96 class InnerRefNamespace : public TraitBase<ConcreteType, InnerRefNamespace> {
97 public:
98  static LogicalResult verifyRegionTrait(Operation *op) {
99  static_assert(
100  ConcreteType::template hasTrait<::mlir::OpTrait::SymbolTable>(),
101  "expected operation to be a SymbolTable");
102 
103  if (op->getNumRegions() != 1)
104  return op->emitError("expected operation to have a single region");
105  if (!op->getRegion(0).hasOneBlock())
106  return op->emitError("expected operation to have a single block");
107 
108  // Verify all InnerSymbolTable's and InnerRef users.
109  return ::circt::hw::detail::verifyInnerRefNamespace(op);
110  }
111 };

113 /// A trait for inner symbol table functionality on an operation.
114 template <typename ConcreteType>
115 class InnerSymbolTable : public TraitBase<ConcreteType, InnerSymbolTable> {
116 public:
117  static LogicalResult verifyRegionTrait(Operation *op) {
118  // Insist that ops with InnerSymbolTable's provide a Symbol, this is
119  // essential to how InnerRef's work.
120  static_assert(
121  ConcreteType::template hasTrait<::mlir::SymbolOpInterface::Trait>(),
122  "expected operation to define a Symbol");
123 
124  // InnerSymbolTable's must be directly nested within an InnerRefNamespace.
125  auto *parent = op->getParentOp();
126  if (!parent || !parent->hasTrait<InnerRefNamespace>())
127  return op->emitError(
128  "InnerSymbolTable must have InnerRefNamespace parent");
129 
130  return success();
131  }
132 };
