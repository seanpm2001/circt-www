94 void PassCommon::getAndSortModulesVisitor(
95  hw::HWModuleLike mod, SmallVectorImpl<hw::HWModuleLike> &mods,
96  DenseSet<Operation *> &modsSeen) {
97  if (modsSeen.contains(mod))
98  return;

107  getAndSortModulesVisitor(modLike, mods, modsSeen);
108  });
