23 struct InferMemoriesPass : public InferMemoriesBase<InferMemoriesPass> {
24  void runOnOperation() override;
25  SmallVector<Operation *> opsToDelete;
26  SmallPtrSet<StringAttr, 2> schemaNames;
27  DenseMap<StringAttr, DictionaryAttr> memoryParams;
28 };

31 void InferMemoriesPass::runOnOperation() {
32  auto module = getOperation();
33  opsToDelete.clear();
34  schemaNames.clear();
35  memoryParams.clear();
36 
37  SymbolCache cache;
38  cache.addDefinitions(module);
39  Namespace names;
40  names.add(cache);
41 
42  // Find the matching generator schemas.
43  for (auto schemaOp : module.getOps<hw::HWGeneratorSchemaOp>()) {
44  if (schemaOp.getDescriptor() == "FIRRTL_Memory") {
45  schemaNames.insert(schemaOp.getSymNameAttr());
46  opsToDelete.push_back(schemaOp);
47  }
48  }
49  LLVM_DEBUG(llvm::dbgs() << "Found " << schemaNames.size() << " schemas\n");
50 
51  // Find generated ops using these schemas.
52  for (auto genOp : module.getOps<hw::HWModuleGeneratedOp>()) {
53  if (!schemaNames.contains(genOp.getGeneratorKindAttr().getAttr()))
54  continue;
55  memoryParams[genOp.getModuleNameAttr()] = genOp->getAttrDictionary();
56  opsToDelete.push_back(genOp);
57  }
58  LLVM_DEBUG(llvm::dbgs() << "Found " << memoryParams.size()
59  << " memory modules\n");
60 
61  // Convert instances of the generated ops into dedicated memories.
62  unsigned numReplaced = 0;
63  module.walk([&](hw::InstanceOp instOp) {
64  auto it = memoryParams.find(instOp.getModuleNameAttr().getAttr());
65  if (it == memoryParams.end())
66  return;
67  ++numReplaced;
68  DictionaryAttr params = it->second;
69  auto width = params.getAs<IntegerAttr>("width").getValue().getZExtValue();
70  auto depth = params.getAs<IntegerAttr>("depth").getValue().getZExtValue();
71  auto maskGranAttr = params.getAs<IntegerAttr>("maskGran");
72  auto maskGran =
73  maskGranAttr ? maskGranAttr.getValue().getZExtValue() : width;
74  auto maskBits = width / maskGran;
75 
76  auto writeLatency =
77  params.getAs<IntegerAttr>("writeLatency").getValue().getZExtValue();
78  auto readLatency =
79  params.getAs<IntegerAttr>("readLatency").getValue().getZExtValue();
80  if (writeLatency != 1) {
81  instOp.emitError("unsupported memory write latency ") << writeLatency;
82  return signalPassFailure();
83  }
84 
85  ImplicitLocOpBuilder builder(instOp.getLoc(), instOp);
86  auto wordType = builder.getIntegerType(width);
87  auto addressTy = dyn_cast<IntegerType>(instOp.getOperand(0).getType());
88  if (!addressTy) {
89  instOp.emitError("expected integer type for memory addressing, got ")
90  << addressTy;
91  return signalPassFailure();
92  }
93  auto memType = MemoryType::get(&getContext(), depth, wordType, addressTy);
94  auto memOp = builder.create<MemoryOp>(memType);
95  if (!instOp.getInstanceName().empty())
96  memOp->setAttr("name", instOp.getInstanceNameAttr());
97 
98  unsigned argIdx = 0;
99  unsigned resultIdx = 0;
100 
101  auto applyReadLatency = [&](Value clock, Value data) {
102  for (unsigned i = 0; i < readLatency; ++i)
103  data = builder.create<seq::CompRegOp>(data, clock, "mem_read_latency");
104  return data;
105  };
106 
107  SmallVector<std::tuple<Value, Value, SmallVector<Value>, bool, bool>>
108  writePorts;
109 
110  // Handle read ports.
111  auto numReadPorts =
112  params.getAs<IntegerAttr>("numReadPorts").getValue().getZExtValue();
113  for (unsigned portIdx = 0; portIdx != numReadPorts; ++portIdx) {
114  auto address = instOp.getOperand(argIdx++);
115  auto enable = instOp.getOperand(argIdx++);
116  auto clock = instOp.getOperand(argIdx++);
117  auto data = instOp.getResult(resultIdx++);
118 
119  if (address.getType() != addressTy) {
120  instOp.emitOpError("expected ")
121  << addressTy << ", but got " << address.getType();
122  return signalPassFailure();
123  }
124 
125  // NOTE: the result of a disabled read port is undefined, currently we
126  // define it to be zero.
127  Value readOp = builder.create<MemoryReadPortOp>(wordType, memOp, address);
128  Value zero = builder.create<hw::ConstantOp>(wordType, 0);
129  readOp = builder.create<comb::MuxOp>(enable, readOp, zero);
130 
131  // NOTE: if the read-latency is 0, the memory read is combinatorial
132  // without any buffer
133  readOp = applyReadLatency(clock, readOp);
134  data.replaceAllUsesWith(readOp);
135  }
136 
137  // Handle read-write ports.
138  auto numReadWritePorts = params.getAs<IntegerAttr>("numReadWritePorts")
139  .getValue()
140  .getZExtValue();
141  for (unsigned portIdx = 0; portIdx != numReadWritePorts; ++portIdx) {
142  auto address = instOp.getOperand(argIdx++);
143  auto enable = instOp.getOperand(argIdx++);
144  auto clock = instOp.getOperand(argIdx++);
145  auto writeMode = instOp.getOperand(argIdx++);
146  auto writeData = instOp.getOperand(argIdx++);
147  auto writeMask = maskBits > 1 ? instOp.getOperand(argIdx++) : Value{};
148  auto readData = instOp.getResult(resultIdx++);
149 
150  if (address.getType() != addressTy) {
151  instOp.emitOpError("expected ")
152  << addressTy << ", but got " << address.getType();
153  return signalPassFailure();
154  }
155 
156  // NOTE: the result of a disabled read port is undefined, currently we
157  // define it to be zero.
158  Value readOp = builder.create<MemoryReadPortOp>(wordType, memOp, address);
159  Value zero = builder.create<hw::ConstantOp>(wordType, 0);
160  readOp = builder.create<comb::MuxOp>(enable, readOp, zero);
161 
162  if (writeMask) {
163  unsigned maskWidth = writeMask.getType().cast<IntegerType>().getWidth();
164  SmallVector<Value> toConcat;
165  for (unsigned i = 0; i < maskWidth; ++i) {
166  Value bit = builder.create<comb::ExtractOp>(writeMask, i, 1);
167  Value replicated = builder.create<comb::ReplicateOp>(bit, maskGran);
168  toConcat.push_back(replicated);
169  }
170  writeMask =
171  builder.create<comb::ConcatOp>(writeData.getType(), toConcat);
172  }
173 
174  // NOTE: if the read-latency is 0, the memory read is combinatorial
175  // without any buffer
176  readOp = applyReadLatency(clock, readOp);
177  readData.replaceAllUsesWith(readOp);
178 
179  auto writeEnable = builder.create<comb::AndOp>(enable, writeMode);
180  SmallVector<Value> inputs({address, writeData, writeEnable});
181  if (writeMask)
182  inputs.push_back(writeMask);
183  writePorts.push_back({memOp, clock, inputs, true, !!writeMask});
184  }
185 
186  // Handle write ports.
187  auto numWritePorts =
188  params.getAs<IntegerAttr>("numWritePorts").getValue().getZExtValue();
189  for (unsigned portIdx = 0; portIdx != numWritePorts; ++portIdx) {
190  auto address = instOp.getOperand(argIdx++);
191  auto enable = instOp.getOperand(argIdx++);
192  auto clock = instOp.getOperand(argIdx++);
193  auto data = instOp.getOperand(argIdx++);
194  auto mask = maskBits > 1 ? instOp.getOperand(argIdx++) : Value{};
195 
196  if (address.getType() != addressTy) {
197  instOp.emitOpError("expected ")
198  << addressTy << ", but got " << address.getType();
199  return signalPassFailure();
200  }
201 
202  if (mask) {
203  unsigned maskWidth = mask.getType().cast<IntegerType>().getWidth();
204  SmallVector<Value> toConcat;
205  for (unsigned i = 0; i < maskWidth; ++i) {
206  Value bit = builder.create<comb::ExtractOp>(mask, i, 1);
207  Value replicated = builder.create<comb::ReplicateOp>(bit, maskGran);
208  toConcat.push_back(replicated);
209  }
210  mask = builder.create<comb::ConcatOp>(data.getType(), toConcat);
211  }
212  SmallVector<Value> inputs({address, data});
213  if (enable)
214  inputs.push_back(enable);
215  if (mask)
216  inputs.push_back(mask);
217  writePorts.push_back({memOp, clock, inputs, !!enable, !!mask});
218  }
219 
220  // Create the actual write ports with a dependency arc to all read
221  // ports.
222  for (auto [memOp, clock, inputs, hasEnable, hasMask] : writePorts) {
223  auto ipSave = builder.saveInsertionPoint();
224  TypeRange types = ValueRange(inputs).getTypes();
225  builder.setInsertionPointToStart(module.getBody());
226  auto defOp = builder.create<DefineOp>(
227  names.newName("mem_write"), builder.getFunctionType(types, types));
228  auto &block = defOp.getBody().emplaceBlock();
229  auto args = block.addArguments(
230  types, SmallVector<Location>(types.size(), builder.getLoc()));
231  builder.setInsertionPointToEnd(&block);
232  builder.create<arc::OutputOp>(SmallVector<Value>(args));
233  builder.restoreInsertionPoint(ipSave);
234  builder.create<MemoryWritePortOp>(memOp, defOp.getName(), inputs, clock,
235  hasEnable, hasMask);
236  }
237 
238  opsToDelete.push_back(instOp);
239  });
240  LLVM_DEBUG(llvm::dbgs() << "Inferred " << numReplaced << " memories\n");
241 
242  for (auto *op : opsToDelete)
243  op->erase();
244 }
