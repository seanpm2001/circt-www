40 /// Print a type with a custom printer implementation.
41 ///
42 /// This only prints a subset of all types in the dialect. Use `printNestedType`
43 /// instead, which will call this function in turn, as appropriate.
44 static LogicalResult customTypePrinter(Type type, AsmPrinter &os) {
45  if (isConst(type)) {
46  os << "const.";
47  }
48 
49  auto printWidthQualifier = [&](std::optional<int32_t> width) {
50  if (width)
51  os << '<' << *width << '>';
52  };
53  bool anyFailed = false;
54  TypeSwitch<Type>(type)
55  .Case<ClockType>([&](auto) { os << "clock"; })
56  .Case<ResetType>([&](auto) { os << "reset"; })
57  .Case<AsyncResetType>([&](auto) { os << "asyncreset"; })
58  .Case<SIntType>([&](auto sIntType) {
59  os << "sint";
60  printWidthQualifier(sIntType.getWidth());
61  })
62  .Case<UIntType>([&](auto uIntType) {
63  os << "uint";
64  printWidthQualifier(uIntType.getWidth());
65  })
66  .Case<AnalogType>([&](auto analogType) {
67  os << "analog";
68  printWidthQualifier(analogType.getWidth());
69  })
70  .Case<BundleType, OpenBundleType>([&](auto bundleType) {
71  if (isa<OpenBundleType>(bundleType))
72  os << "open";
73  os << "bundle<";
74  llvm::interleaveComma(bundleType, os, [&](auto element) {
75  os << element.name.getValue();
76  if (element.isFlip)
77  os << " flip";
78  os << ": ";
79  printNestedType(element.type, os);
80  });
81  os << '>';
82  })
83  .Case<FEnumType>([&](auto fenumType) {
84  os << "enum<";
85  llvm::interleaveComma(fenumType, os,
86  [&](FEnumType::EnumElement element) {
87  os << element.name.getValue();
88  os << ": ";
89  printNestedType(element.type, os);
90  });
91  os << '>';
92  })
93  .Case<FVectorType, OpenVectorType>([&](auto vectorType) {
94  if (isa<OpenVectorType>(vectorType))
95  os << "open";
96  os << "vector<";
97  printNestedType(vectorType.getElementType(), os);
98  os << ", " << vectorType.getNumElements() << '>';
99  })
100  .Case<RefType>([&](auto refType) {
101  if (refType.getForceable())
102  os << "rw";
103  os << "probe<";
104  printNestedType(refType.getType(), os);
105  os << '>';
106  })
107  .Case<StringType>([&](auto stringType) {
108  os << "string";
109  })
110  .Default([&](auto) { anyFailed = true; });
111  return failure(anyFailed);
112 }

128 /// Parse a type with a custom parser implementation.
129 ///
130 /// This only accepts a subset of all types in the dialect. Use `parseType`
131 /// instead, which will call this function in turn, as appropriate.
132 ///
133 /// Returns `std::nullopt` if the type `name` is not covered by the custom
134 /// parsers. Otherwise returns success or failure as appropriate. On success,
135 /// `result` is set to the resulting type.
136 ///
137 /// ```plain
138 /// firrtl-type
139 /// ::= clock
140 /// ::= reset
141 /// ::= asyncreset
142 /// ::= sint ('<' int '>')?
143 /// ::= uint ('<' int '>')?
144 /// ::= analog ('<' int '>')?
145 /// ::= bundle '<' (bundle-elt (',' bundle-elt)*)? '>'
146 /// ::= enum '<' (enum-elt (',' enum-elt)*)? '>'
147 /// ::= vector '<' type ',' int '>'
148 /// ::= const '.' type
149 /// ::= 'property.' firrtl-phased-type
150 /// bundle-elt ::= identifier flip? ':' type
151 /// enum-elt ::= identifier ':' type
152 /// ```
153 static OptionalParseResult customTypeParser(AsmParser &parser, StringRef name,
154  Type &result) {
155  bool isConst = false;
156  const char constPrefix[] = "const.";
157  if (name.starts_with(constPrefix)) {
158  isConst = true;
159  name = name.drop_front(std::size(constPrefix) - 1);
160  }
161 
162  auto *context = parser.getContext();
163  if (name.equals("clock"))
164  return result = ClockType::get(context, isConst), success();
165  if (name.equals("reset"))
166  return result = ResetType::get(context, isConst), success();
167  if (name.equals("asyncreset"))
168  return result = AsyncResetType::get(context, isConst), success();
169 
170  if (name.equals("sint") || name.equals("uint") || name.equals("analog")) {
171  // Parse the width specifier if it exists.
172  int32_t width = -1;
173  if (!parser.parseOptionalLess()) {
174  if (parser.parseInteger(width) || parser.parseGreater())
175  return failure();
176 
177  if (width < 0)
178  return parser.emitError(parser.getNameLoc(), "unknown width"),
179  failure();
180  }
181 
182  if (name.equals("sint"))
183  result = SIntType::get(context, width, isConst);
184  else if (name.equals("uint"))
185  result = UIntType::get(context, width, isConst);
186  else {
187  assert(name.equals("analog"));
188  result = AnalogType::get(context, width, isConst);
189  }
190  return success();
191  }
192 
193  if (name.equals("bundle")) {
194  SmallVector<BundleType::BundleElement, 4> elements;
195 
196  auto parseBundleElement = [&]() -> ParseResult {
197  std::string nameStr;
198  StringRef name;
199  FIRRTLBaseType type;
200 
201  // The 'name' can be an identifier or an integer.
202  uint32_t fieldIntName;
203  auto intName = parser.parseOptionalInteger(fieldIntName);
204  if (intName.has_value()) {
205  if (failed(intName.value()))
206  return failure();
207  nameStr = llvm::utostr(fieldIntName);
208  name = nameStr;
209  } else {
210  // Otherwise must be an identifier.
211  if (parser.parseKeyword(&name))
212  return failure();
213  }
214 
215  bool isFlip = succeeded(parser.parseOptionalKeyword("flip"));
216  if (parser.parseColon() || parseNestedBaseType(type, parser))
217  return failure();
218 
219  elements.push_back({StringAttr::get(context, name), isFlip, type});
220  return success();
221  };
222 
223  if (parser.parseCommaSeparatedList(mlir::AsmParser::Delimiter::LessGreater,
224  parseBundleElement))
225  return failure();
226 
227  return result = BundleType::get(context, elements, isConst), success();
228  }
229  if (name.equals("openbundle")) {
230  SmallVector<OpenBundleType::BundleElement, 4> elements;
231 
232  auto parseBundleElement = [&]() -> ParseResult {
233  std::string nameStr;
234  StringRef name;
235  FIRRTLType type;
236 
237  // The 'name' can be an identifier or an integer.
238  uint32_t fieldIntName;
239  auto intName = parser.parseOptionalInteger(fieldIntName);
240  if (intName.has_value()) {
241  if (failed(intName.value()))
242  return failure();
243  nameStr = llvm::utostr(fieldIntName);
244  name = nameStr;
245  } else {
246  // Otherwise must be an identifier.
247  if (parser.parseKeyword(&name))
248  return failure();
249  }
250 
251  bool isFlip = succeeded(parser.parseOptionalKeyword("flip"));
252  if (parser.parseColon() || parseNestedType(type, parser))
253  return failure();
254 
255  elements.push_back({StringAttr::get(context, name), isFlip, type});
256  return success();
257  };
258 
259  if (parser.parseCommaSeparatedList(mlir::AsmParser::Delimiter::LessGreater,
260  parseBundleElement))
261  return failure();
262 
263  return result = OpenBundleType::get(context, elements, isConst), success();
264  }
265 
266  if (name.equals("enum")) {
267  SmallVector<FEnumType::EnumElement, 4> elements;
268 
269  auto parseEnumElement = [&]() -> ParseResult {
270  std::string nameStr;
271  StringRef name;
272  FIRRTLBaseType type;
273 
274  // The 'name' can be an identifier or an integer.
275  uint32_t fieldIntName;
276  auto intName = parser.parseOptionalInteger(fieldIntName);
277  if (intName.has_value()) {
278  if (failed(intName.value()))
279  return failure();
280  nameStr = llvm::utostr(fieldIntName);
281  name = nameStr;
282  } else {
283  // Otherwise must be an identifier.
284  if (parser.parseKeyword(&name))
285  return failure();
286  }
287 
288  if (parser.parseColon() || parseNestedBaseType(type, parser))
289  return failure();
290 
291  elements.push_back({StringAttr::get(context, name), type});
292  return success();
293  };
294 
295  if (parser.parseCommaSeparatedList(mlir::AsmParser::Delimiter::LessGreater,
296  parseEnumElement))
297  return failure();
298 
299  return result = FEnumType::get(context, elements, isConst), success();
300  }
301 
302  if (name.equals("vector")) {
303  FIRRTLBaseType elementType;
304  uint64_t width = 0;
305 
306  if (parser.parseLess() || parseNestedBaseType(elementType, parser) ||
307  parser.parseComma() || parser.parseInteger(width) ||
308  parser.parseGreater())
309  return failure();
310 
311  return result = FVectorType::get(elementType, width, isConst), success();
312  }
313  if (name.equals("openvector")) {
314  FIRRTLType elementType;
315  uint64_t width = 0;
316 
317  if (parser.parseLess() || parseNestedType(elementType, parser) ||
318  parser.parseComma() || parser.parseInteger(width) ||
319  parser.parseGreater())
320  return failure();
321 
322  return result = OpenVectorType::get(elementType, width, isConst), success();
323  }
324 
325  // For now, support both firrtl.ref and firrtl.probe.
326  if (name.equals("ref") || name.equals("probe")) {
327  FIRRTLBaseType type;
328  // Don't pass `isConst` to `parseNestedBaseType since `ref` can point to
329  // either `const` or non-`const` types
330  if (parser.parseLess() || parseNestedBaseType(type, parser) ||
331  parser.parseGreater())
332  return failure();
333 
334  if (failed(RefType::verify(
335  [&]() { return parser.emitError(parser.getNameLoc()); }, type,
336  false)))
337  return failure();
338 
339  return result = RefType::get(type, false), success();
340  }
341  if (name.equals("rwprobe")) {
342  FIRRTLBaseType type;
343  if (parser.parseLess() || parseNestedBaseType(type, parser) ||
344  parser.parseGreater())
345  return failure();
346 
347  if (failed(RefType::verify(
348  [&]() { return parser.emitError(parser.getNameLoc()); }, type,
349  true)))
350  return failure();
351 
352  return result = RefType::get(type, true), success();
353  }
354  if (name.equals("string")) {
355  if (isConst) {
356  parser.emitError(parser.getNameLoc(), "strings cannot be const");
357  return failure();
358  }
359  result = StringType::get(parser.getContext());
360  return success();
361  }
362 
363  return {};
364 }

366 /// Parse a type defined by this dialect.
367 ///
368 /// This will first try the generated type parsers and then resort to the custom
369 /// parser implementation. Emits an error and returns failure if `name` does not
370 /// refer to a type defined in this dialect.
371 static ParseResult parseType(Type &result, StringRef name, AsmParser &parser) {
372  // Try the custom type parser.
373  OptionalParseResult parseResult = customTypeParser(parser, name, result);
374  if (parseResult.has_value())
375  return parseResult.value();
376 
377  // None of the above recognized the type, so we bail.
378  parser.emitError(parser.getNameLoc(), "unknown FIRRTL dialect type: \"")
379  << name << "\"";
380  return failure();
381 }

383 /// Parse a `FIRRTLType` with a `name` that has already been parsed.
384 ///
385 /// Note that only a subset of types defined in the FIRRTL dialect inherit from
386 /// `FIRRTLType`. Use `parseType` to parse *any* of the defined types.
387 static ParseResult parseFIRRTLType(FIRRTLType &result, StringRef name,
388  AsmParser &parser) {
389  Type type;
390  if (failed(parseType(type, name, parser)))
391  return failure();
392  result = type.dyn_cast<FIRRTLType>();
393  if (result)
394  return success();
395  parser.emitError(parser.getNameLoc(), "unknown FIRRTL type: \"")
396  << name << "\"";
397  return failure();
398 }

400 static ParseResult parseFIRRTLBaseType(FIRRTLBaseType &result, StringRef name,
401  AsmParser &parser) {
402  FIRRTLType type;
403  if (failed(parseFIRRTLType(type, name, parser)))
404  return failure();
405  if (auto base = type.dyn_cast<FIRRTLBaseType>()) {
406  result = base;
407  return success();
408  }
409  parser.emitError(parser.getNameLoc(), "expected base type, found ") << type;
410  return failure();
411 }

724 /// Helper to implement the equivalence logic for a pair of bundle elements.
725 /// Note that the FIRRTL spec requires bundle elements to have the same
726 /// orientation, but this only compares their passive types. The FIRRTL dialect
727 /// differs from the spec in how it uses flip types for module output ports and
728 /// canonicalizes flips in bundles, so only passive types can be compared here.
729 static bool areBundleElementsEquivalent(BundleType::BundleElement destElement,
730  BundleType::BundleElement srcElement,
731  bool destOuterTypeIsConst,
732  bool srcOuterTypeIsConst,
733  bool requiresSameWidth) {
734  if (destElement.name != srcElement.name)
735  return false;
736  if (destElement.isFlip != srcElement.isFlip)
737  return false;
738 
739  if (destElement.isFlip) {
740  std::swap(destElement, srcElement);
741  std::swap(destOuterTypeIsConst, srcOuterTypeIsConst);
742  }
743 
744  return areTypesEquivalent(destElement.type, srcElement.type,
745  destOuterTypeIsConst, srcOuterTypeIsConst,
746  requiresSameWidth);
747 }

749 /// Returns whether the two types are equivalent. This implements the exact
750 /// definition of type equivalence in the FIRRTL spec. If the types being
751 /// compared have any outer flips that encode FIRRTL module directions (input or
752 /// output), these should be stripped before using this method.
753 bool firrtl::areTypesEquivalent(FIRRTLType destFType, FIRRTLType srcFType,
754  bool destOuterTypeIsConst,
755  bool srcOuterTypeIsConst,
756  bool requireSameWidths) {
757  auto destType = destFType.dyn_cast<FIRRTLBaseType>();
758  auto srcType = srcFType.dyn_cast<FIRRTLBaseType>();
759 
760  // For non-base types, only equivalent if identical.
761  if (!destType || !srcType)
762  return destFType == srcFType;
763 
764  bool srcIsConst = srcOuterTypeIsConst || srcFType.isConst();
765  bool destIsConst = destOuterTypeIsConst || destFType.isConst();
766 
767  // Vector types can be connected if they have the same size and element type.
768  auto destVectorType = destType.dyn_cast<FVectorType>();
769  auto srcVectorType = srcType.dyn_cast<FVectorType>();
770  if (destVectorType && srcVectorType)
771  return destVectorType.getNumElements() == srcVectorType.getNumElements() &&
772  areTypesEquivalent(destVectorType.getElementType(),
773  srcVectorType.getElementType(), destIsConst,
774  srcIsConst, requireSameWidths);
775 
776  // Bundle types can be connected if they have the same size, element names,
777  // and element types.
778  auto destBundleType = destType.dyn_cast<BundleType>();
779  auto srcBundleType = srcType.dyn_cast<BundleType>();
780  if (destBundleType && srcBundleType) {
781  auto destElements = destBundleType.getElements();
782  auto srcElements = srcBundleType.getElements();
783  size_t numDestElements = destElements.size();
784  if (numDestElements != srcElements.size())
785  return false;
786 
787  for (size_t i = 0; i < numDestElements; ++i) {
788  auto destElement = destElements[i];
789  auto srcElement = srcElements[i];
790  if (!areBundleElementsEquivalent(destElement, srcElement, destIsConst,
791  srcIsConst, requireSameWidths))
792  return false;
793  }
794  return true;
795  }
796 
797  // Enum types can be connected if they have the same size, element names, and
798  // element types.
799  auto dstEnumType = destType.dyn_cast<FEnumType>();
800  auto srcEnumType = srcType.dyn_cast<FEnumType>();
801  if (dstEnumType && srcEnumType) {
802  if (dstEnumType.getNumElements() != srcEnumType.getNumElements())
803  return false;
804  // Enums requires the types to match exactly.
805  for (const auto &[dst, src] : llvm::zip(dstEnumType, srcEnumType)) {
806  // The variant names must match.
807  if (dst.name != src.name)
808  return false;
809  // Enumeration types can only be connected if the inner types have the
810  // same width.
811  if (!areTypesEquivalent(dst.type, src.type, destIsConst, srcIsConst,
812  true))
813  return false;
814  }
815  return true;
816  }
817 
818  // Ground type connections must be const compatible.
819  if (destIsConst && !srcIsConst)
820  return false;
821 
822  // Reset types can be driven by UInt<1>, AsyncReset, or Reset types.
823  if (destType.isa<ResetType>())
824  return srcType.isResetType();
825 
826  // Reset types can drive UInt<1>, AsyncReset, or Reset types.
827  if (srcType.isa<ResetType>())
828  return destType.isResetType();
829 
830  // If we can implicitly truncate or extend the bitwidth, or either width is
831  // currently uninferred, then compare the widthless version of these types.
832  if (!requireSameWidths || destType.getBitWidthOrSentinel() == -1)
833  srcType = srcType.getWidthlessType();
834  if (!requireSameWidths || srcType.getBitWidthOrSentinel() == -1)
835  destType = destType.getWidthlessType();
836 
837  // Ground types can be connected if their constless types are the same
838  return destType.getConstType(false) == srcType.getConstType(false);
839 }

841 /// Returns whether the two types are weakly equivalent.
842 bool firrtl::areTypesWeaklyEquivalent(FIRRTLType destFType, FIRRTLType srcFType,
843  bool destFlip, bool srcFlip,
844  bool destOuterTypeIsConst,
845  bool srcOuterTypeIsConst) {
846  auto destType = destFType.dyn_cast<FIRRTLBaseType>();
847  auto srcType = srcFType.dyn_cast<FIRRTLBaseType>();
848 
849  // For non-base types, only equivalent if identical.
850  if (!destType || !srcType)
851  return destFType == srcFType;
852 
853  bool srcIsConst = srcOuterTypeIsConst || srcFType.isConst();
854  bool destIsConst = destOuterTypeIsConst || destFType.isConst();
855 
856  // Vector types can be connected if their element types are weakly equivalent.
857  // Size doesn't matter.
858  auto destVectorType = destType.dyn_cast<FVectorType>();
859  auto srcVectorType = srcType.dyn_cast<FVectorType>();
860  if (destVectorType && srcVectorType)
861  return areTypesWeaklyEquivalent(destVectorType.getElementType(),
862  srcVectorType.getElementType(), destFlip,
863  srcFlip, destIsConst, srcIsConst);
864 
865  // Bundle types are weakly equivalent if all common elements are weakly
866  // equivalent. Non-matching fields are ignored. Flips are "pushed" into
867  // recursive weak type equivalence checks.
868  auto destBundleType = destType.dyn_cast<BundleType>();
869  auto srcBundleType = srcType.dyn_cast<BundleType>();
870  if (destBundleType && srcBundleType)
871  return llvm::all_of(destBundleType, [&](auto destElt) -> bool {
872  auto destField = destElt.name.getValue();
873  auto srcElt = srcBundleType.getElement(destField);
874  // If the src doesn't contain the destination's field, that's okay.
875  if (!srcElt)
876  return true;
877 
878  return areTypesWeaklyEquivalent(
879  destElt.type, srcElt->type, destFlip ^ destElt.isFlip,
880  srcFlip ^ srcElt->isFlip, destOuterTypeIsConst, srcOuterTypeIsConst);
881  });
882 
883  // Ground types require leaf flippedness and const compatibility
884  if (destFlip != srcFlip)
885  return false;
886  if (destFlip && srcIsConst && !destIsConst)
887  return false;
888  if (srcFlip && destIsConst && !srcIsConst)
889  return false;
890 
891  // Reset types can be driven by UInt<1>, AsyncReset, or Reset types.
892  if (destType.isa<ResetType>())
893  return srcType.isResetType();
894 
895  // Reset types can drive UInt<1>, AsyncReset, or Reset types.
896  if (srcType.isa<ResetType>())
897  return destType.isResetType();
898 
899  // Ground types can be connected if their passive, widthless versions
900  // are equal and are const and flip compatible
901  auto widthlessDestType = destType.getWidthlessType();
902  auto widthlessSrcType = srcType.getWidthlessType();
903  return widthlessDestType.getConstType(false) ==
904  widthlessSrcType.getConstType(false);
905 }

907 /// Returns whether the srcType can be const-casted to the destType.
908 bool firrtl::areTypesConstCastable(FIRRTLType destFType, FIRRTLType srcFType,
909  bool srcOuterTypeIsConst) {
910  // Identical types are always castable
911  if (destFType == srcFType)
912  return true;
913 
914  auto destType = destFType.dyn_cast<FIRRTLBaseType>();
915  auto srcType = srcFType.dyn_cast<FIRRTLBaseType>();
916 
917  // For non-base types, only castable if identical.
918  if (!destType || !srcType)
919  return false;
920 
921  // Types must be passive
922  if (!destType.isPassive() || !srcType.isPassive())
923  return false;
924 
925  bool srcIsConst = srcType.isConst() || srcOuterTypeIsConst;
926 
927  // Cannot cast non-'const' src to 'const' dest
928  if (destType.isConst() && !srcIsConst)
929  return false;
930 
931  // Vector types can be casted if they have the same size and castable element
932  // type.
933  auto destVectorType = destType.dyn_cast<FVectorType>();
934  auto srcVectorType = srcType.dyn_cast<FVectorType>();
935  if (destVectorType && srcVectorType)
936  return destVectorType.getNumElements() == srcVectorType.getNumElements() &&
937  areTypesConstCastable(destVectorType.getElementType(),
938  srcVectorType.getElementType(), srcIsConst);
939  if (destVectorType != srcVectorType)
940  return false;
941 
942  // Bundle types can be casted if they have the same size, element names,
943  // and castable element types.
944  auto destBundleType = destType.dyn_cast<BundleType>();
945  auto srcBundleType = srcType.dyn_cast<BundleType>();
946  if (destBundleType && srcBundleType) {
947  auto destElements = destBundleType.getElements();
948  auto srcElements = srcBundleType.getElements();
949  size_t numDestElements = destElements.size();
950  if (numDestElements != srcElements.size())
951  return false;
952 
953  return llvm::all_of_zip(
954  destElements, srcElements,
955  [&](const auto &destElement, const auto &srcElement) {
956  return destElement.name == srcElement.name &&
957  areTypesConstCastable(destElement.type, srcElement.type,
958  srcIsConst);
959  });
960  }
961  if (destBundleType != srcBundleType)
962  return false;
963 
964  // Ground types can be casted if the source type is a const
965  // version of the destination type
966  return destType == srcType.getConstType(destType.isConst());
967 }

970 bool firrtl::isTypeLarger(FIRRTLBaseType dstType, FIRRTLBaseType srcType) {
971  return TypeSwitch<FIRRTLBaseType, bool>(dstType)
972  .Case<BundleType>([&](auto dstBundle) {

2106 // Get the bit width for this type, return None if unknown. Unlike
2107 // getBitWidthOrSentinel(), this can recursively compute the bitwidth of
2108 // aggregate types. For bundle and vectors, recursively get the width of each
2109 // field element and return the total bit width of the aggregate type. This
2110 // returns None, if any of the bundle fields is a flip type, or ground type with
2111 // unknown bit width.
2112 std::optional<int64_t> firrtl::getBitWidth(FIRRTLBaseType type,
2113  bool ignoreFlip) {
2114  std::function<std::optional<int64_t>(FIRRTLBaseType)> getWidth =
2115  [&](FIRRTLBaseType type) -> std::optional<int64_t> {
2116  return TypeSwitch<FIRRTLBaseType, std::optional<int64_t>>(type)
2117  .Case<BundleType>([&](BundleType bundle) -> std::optional<int64_t> {
2118  int64_t width = 0;
2119  for (auto &elt : bundle) {
2120  if (elt.isFlip && !ignoreFlip)
2121  return std::nullopt;
2122  auto w = getBitWidth(elt.type);
2123  if (!w.has_value())
2124  return std::nullopt;
2125  width += *w;
2126  }
2127  return width;
2128  })
2129  .Case<FEnumType>([&](FEnumType fenum) -> std::optional<int64_t> {
2130  int64_t width = 0;
2131  for (auto &elt : fenum) {
2132  auto w = getBitWidth(elt.type);
2133  if (!w.has_value())
2134  return std::nullopt;
2135  width = std::max(width, *w);
2136  }
2137  return width + llvm::Log2_32_Ceil(fenum.getNumElements());
2138  })
2139  .Case<FVectorType>([&](auto vector) -> std::optional<int64_t> {
2140  auto w = getBitWidth(vector.getElementType());
2141  if (!w.has_value())
2142  return std::nullopt;
2143  return *w * vector.getNumElements();
2144  })
2145  .Case<IntType>([&](IntType iType) { return iType.getWidth(); })
2146  .Case<ClockType, ResetType, AsyncResetType>([](Type) { return 1; })
2147  .Default([&](auto t) { return std::nullopt; });
2148  };
2149  return getWidth(type);
2150 }
