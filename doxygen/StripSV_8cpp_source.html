31 void StripSVPass::runOnOperation() {
32  auto mlirModule = getOperation();
33  opsToDelete.clear();
34  clockGateModuleNames.clear();
35  externModuleNames.clear();
36 
37  auto expectedClockGateInputs =
38  ArrayAttr::get(&getContext(), {StringAttr::get(&getContext(), "in"),
39  StringAttr::get(&getContext(), "test_en"),
40  StringAttr::get(&getContext(), "en")});
41  auto expectedClockGateOutputs =
42  ArrayAttr::get(&getContext(), {StringAttr::get(&getContext(), "out")});
43  auto i1Type = IntegerType::get(&getContext(), 1);
44 
45  for (auto extModOp : mlirModule.getOps<hw::HWModuleExternOp>()) {
46  if (extModOp.getVerilogModuleName() == "EICG_wrapper") {
47  if (extModOp.getArgNames() != expectedClockGateInputs ||
48  extModOp.getResultNames() != expectedClockGateOutputs) {
49  extModOp.emitError("clock gate module `")
50  << extModOp.getModuleName() << "` has incompatible port names "
51  << extModOp.getArgNamesAttr() << " -> "
52  << extModOp.getResultNamesAttr();
53  return signalPassFailure();
54  }
55  if (extModOp.getArgumentTypes() !=
56  ArrayRef<Type>{i1Type, i1Type, i1Type} ||
57  extModOp.getResultTypes() != ArrayRef<Type>{i1Type}) {
58  extModOp.emitError("clock gate module `")
59  << extModOp.getModuleName() << "` has incompatible port types "
60  << extModOp.getArgumentTypes() << " -> "
61  << extModOp.getResultTypes();
62  return signalPassFailure();
63  }
64  clockGateModuleNames.insert(extModOp.getModuleNameAttr());
65  opsToDelete.push_back(extModOp);
66  continue;
67  }
68 
69  externModuleNames.insert(extModOp.getModuleNameAttr());
70  if (replaceExtModuleOutputs)
71  opsToDelete.push_back(extModOp);
72  }
73  LLVM_DEBUG(llvm::dbgs() << "Found " << clockGateModuleNames.size()
74  << " clock gates, " << externModuleNames.size()
75  << " other extern modules\n");
76 
77  // Remove `sv.*` operation attributes.
78  mlirModule.walk([](Operation *op) {
79  auto isSVAttr = [](NamedAttribute attr) {
80  return attr.getName().getValue().startswith("sv.");
81  };
82  if (llvm::any_of(op->getAttrs(), isSVAttr)) {
83  SmallVector<NamedAttribute> newAttrs;
84  newAttrs.reserve(op->getAttrs().size());
85  for (auto attr : op->getAttrs())
86  if (!isSVAttr(attr))
87  newAttrs.push_back(attr);
88  op->setAttrs(newAttrs);
89  }
90  });
91 
92  // Remove ifdefs and verbatim.
93  for (auto verb : mlirModule.getOps<sv::VerbatimOp>())
94  opsToDelete.push_back(verb);
95  for (auto verb : mlirModule.getOps<sv::IfDefOp>())
96  opsToDelete.push_back(verb);
97  for (auto verb : mlirModule.getOps<sv::MacroDeclOp>())
98  opsToDelete.push_back(verb);
99 
100  for (auto module : mlirModule.getOps<hw::HWModuleOp>()) {
101  for (Operation &op : *module.getBodyBlock()) {
102  // Remove ifdefs and verbatim.
103  if (isa<sv::IfDefOp, sv::CoverOp, sv::CoverConcurrentOp>(&op)) {
104  opsToDelete.push_back(&op);
105  continue;
106  }
107  if (isa<sv::VerbatimOp, sv::AlwaysOp>(&op)) {
108  opsToDelete.push_back(&op);
109  continue;
110  }
111 
112  // Remove wires.
113  if (auto assign = dyn_cast<sv::AssignOp>(&op)) {
114  auto wire = assign.getDest().getDefiningOp<sv::WireOp>();
115  if (!wire) {
116  assign.emitOpError("expected wire lhs");
117  return signalPassFailure();
118  }
119  for (Operation *user : wire->getUsers()) {
120  if (user == assign)
121  continue;
122  auto readInout = dyn_cast<sv::ReadInOutOp>(user);
123  if (!readInout) {
124  user->emitOpError("has user that is not `sv.read_inout`");
125  return signalPassFailure();
126  }
127  readInout.replaceAllUsesWith(assign.getSrc());
128  opsToDelete.push_back(readInout);
129  }
130  opsToDelete.push_back(assign);
131  opsToDelete.push_back(wire);
132  continue;
133  }
134 
135  // Canonicalize registers.
136  if (auto reg = dyn_cast<seq::FirRegOp>(&op)) {
137  OpBuilder builder(reg);
138  Value next;
139  // Note: this register will have an sync reset regardless.
140  if (reg.hasReset())
141  next = builder.create<comb::MuxOp>(reg.getLoc(), reg.getReset(),
142  reg.getResetValue(), reg.getNext(),
143  false);
144  else
145  next = reg.getNext();
146 
147  Value compReg = builder.create<seq::CompRegOp>(
148  reg.getLoc(), next.getType(), next, reg.getClk(), reg.getNameAttr(),
149  Value{}, Value{}, reg.getInnerSymAttr());
150  reg.replaceAllUsesWith(compReg);
151  opsToDelete.push_back(reg);
152  continue;
153  }
154 
155  // Replace clock gate instances with the dedicated arc op and stub
156  // out other external modules.
157  if (auto instOp = dyn_cast<hw::InstanceOp>(&op)) {
158  auto modName = instOp.getModuleNameAttr().getAttr();
159  ImplicitLocOpBuilder builder(instOp.getLoc(), instOp);
160  if (clockGateModuleNames.contains(modName)) {
161  auto enable = builder.createOrFold<comb::OrOp>(
162  instOp.getOperand(1), instOp.getOperand(2), true);
163  auto gated =
164  builder.create<arc::ClockGateOp>(instOp.getOperand(0), enable);
165  instOp.replaceAllUsesWith(gated);
166  opsToDelete.push_back(instOp);
167  } else if (externModuleNames.contains(modName)) {
168  if (replaceExtModuleOutputs) {
169  instOp->emitWarning("StripSV: outputs of external module instance "
170  "replaced with zero value!");
171  for (auto result : instOp.getResults())
172  result.replaceAllUsesWith(
173  builder.create<hw::ConstantOp>(result.getType(), 0));
174  opsToDelete.push_back(instOp);
175  }
176  }
177  continue;
178  }
179  }
180  }
181  for (auto *op : opsToDelete)
182  op->erase();
183 }
